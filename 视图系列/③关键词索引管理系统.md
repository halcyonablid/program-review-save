function ensureUiStateSets() {
    if (!(state.collapsedIds instanceof Set)) state.collapsedIds = new Set();
    if (!(state.notesOpenIds instanceof Set)) state.notesOpenIds = new Set();
}

function buildCapsuleExpandedMap() {
    ensureUiStateSets();
    const capsuleExpanded = {};
    Object.keys(state.keywords).forEach(id => {
        capsuleExpanded[id] = !state.collapsedIds.has(id);
    });
    return capsuleExpanded;
}

function applyCapsuleExpandedMap(capsuleExpanded) {
    ensureUiStateSets();
    state.collapsedIds.clear();
    if (!capsuleExpanded || typeof capsuleExpanded !== 'object') return;

    Object.entries(capsuleExpanded).forEach(([id, expanded]) => {
        if (state.keywords[id] && expanded === false) {
            state.collapsedIds.add(id);
        }
    });
}

function pruneUiStateByKeywords() {
    ensureUiStateSets();
    const validIds = new Set(Object.keys(state.keywords));
    state.collapsedIds = new Set([...state.collapsedIds].filter(id => validIds.has(id)));
    state.notesOpenIds = new Set([...state.notesOpenIds].filter(id => validIds.has(id)));
}

function persistUiState() {
    return saveIndex().catch(e => {
        console.error('дҝқеӯҳUIзҠ¶жҖҒеӨұиҙҘ:', e);
    });
}

async function loadIndex() {
    try {
        if (await app.vault.adapter.exists(INDEX_PATH)) {
            const content = await app.vault.adapter.read(INDEX_PATH);
            const json = JSON.parse(content);
            state.keywords = json.keywords || {};
            applyCapsuleExpandedMap(json.capsuleExpanded);
            pruneUiStateByKeywords();
            return state.keywords;
        }
    } catch (e) {
        console.error('еҠ иҪҪзҙўеј•еӨұиҙҘ:', e);
    }
    state.keywords = {};
    ensureUiStateSets();
    state.collapsedIds.clear();
    state.notesOpenIds.clear();
    return {};
}

async function saveIndex() {
    const json = {
        keywords: state.keywords,
        capsuleExpanded: buildCapsuleExpandedMap(), // жүҒе№іеӯ—е…ёпјҡ{ [keywordId]: true/false }
        version: "3.8",
        lastModified: moment().format("YYYY-MM-DD HH:mm:ss"),
        metadata: {
            total: Object.keys(state.keywords).length,
            topLevel: Object.values(state.keywords).filter(k => !k.parentId).length
        }
    };
    const content = JSON.stringify(json, null, 2);

    const dir = INDEX_PATH.substring(0, INDEX_PATH.lastIndexOf('/'));
    if (dir && !(await app.vault.adapter.exists(dir))) {
        await app.vault.adapter.mkdir(dir);
    }

    if (await app.vault.adapter.exists(INDEX_PATH)) {
        const file = app.vault.getAbstractFileByPath(INDEX_PATH);
        await app.vault.modify(file, content);
    } else {
        await app.vault.create(INDEX_PATH, content);
    }
}

/**
 * и§Јжһҗе…ій”®иҜҚеӯ—з¬ҰдёІпјҢжҸҗеҸ–еҹәзЎҖе…ій”®иҜҚе’ҢеҗҺзјҖ
 * дҫӢеҰӮпјҡ'MMKV-AдёҠйЎ№+ж–°ејҖе§Ӣ' вҶ’ { base: 'MMKV-AдёҠйЎ№', suffix: '+ж–°ејҖе§Ӣ' }
 * дҫӢеҰӮпјҡ'з§‘жҠҖе§”е‘ҳдјҡ' вҶ’ { base: 'з§‘жҠҖе§”е‘ҳдјҡ', suffix: '' }
 */
function parseKeywordWithSuffix(kwString) {
    const plusIndex = kwString.indexOf('+');
    if (plusIndex > 0) {
        return {
            base: kwString.substring(0, plusIndex),
            suffix: kwString.substring(plusIndex)  // еҢ…еҗ«+еҸ·
        };
    }
    return { base: kwString, suffix: '' };
}

/**
 * з”ҹжҲҗе…ій”®иҜҚIDпјҲеҹәдәҺеҹәзЎҖе…ій”®иҜҚзҡ„hashпјү
 */
function generateKeywordId(baseKeyword) {
    let hash = 0;
    for (let i = 0; i < baseKeyword.length; i++) {
        hash = ((hash << 5) - hash) + baseKeyword.charCodeAt(i);
        hash = hash & hash;
    }
    return 'kw-' + Math.abs(hash).toString(36);
}

/**
 * иҺ·еҸ–жҺ’еәҸеӯ—жҜҚ
 */
function getSortLetter(keyword) {
    const cleaned = keyword.replace(/^[_\.+\s]+/, '');
    const firstChar = cleaned.charAt(0);
    return (firstChar && /[A-Za-z]/.test(firstChar)) ? firstChar.toUpperCase() : '#';
}

/**
 * иҺ·еҸ–жҳҫзӨәж–Үжң¬
 * и§„еҲҷпјҡеҰӮжһңе…ій”®иҜҚд»ҘеӨ§еҶҷеӯ—жҜҚејҖеӨҙпјҢеҸӘз§»йҷӨз¬¬дёҖдёӘеӯ—жҜҚпјҲз”ЁдәҺalphabeticalзҙўеј•е®ҡдҪҚпјү
 * дҫӢеҰӮпјҡ'EEVAзі»з»ҹи®ҫи®Ў' вҶ’ 'EVAзі»з»ҹи®ҫи®Ў'пјҲз§»йҷӨз¬¬дёҖдёӘEпјү
 *      'AATEC7' вҶ’ 'ATEC7'пјҲз§»йҷӨз¬¬дёҖдёӘAпјү
 *      'Cи¶…и¶ҠжҖ§' вҶ’ 'и¶…и¶ҠжҖ§'пјҲз§»йҷӨCпјү
 *      'з§‘жҠҖе§”е‘ҳдјҡ' вҶ’ 'з§‘жҠҖе§”е‘ҳдјҡ'пјҲдёҚд»ҘеӨ§еҶҷеӯ—жҜҚејҖеӨҙпјҢдҝқжҢҒдёҚеҸҳпјү
 */
function getDisplayText(keyword) {
    // жЈҖжҹҘз¬¬дёҖдёӘеӯ—з¬ҰжҳҜеҗҰдёәеӨ§еҶҷеӯ—жҜҚA-Z
    if (keyword.length > 1 && /^[A-Z]/.test(keyword)) {
        return keyword.substring(1);  // з§»йҷӨз¬¬дёҖдёӘеӯ—з¬Ұ
    }
    return keyword;
}

function compareKeywordName(a, b) {
    const aName = a.displayText || a.baseKeyword || '';
    const bName = b.displayText || b.baseKeyword || '';
    return aName.localeCompare(bName, 'zh-CN', { numeric: true, sensitivity: 'base' });
}

function compareNoteName(a, b) {
    const aName = a.name || '';
    const bName = b.name || '';
    return aName.localeCompare(bName, 'zh-CN', { numeric: true, sensitivity: 'base' });
}

/**
 * и®Ўз®—sortKeyпјҲеҹәдәҺе…іиҒ”з¬”и®°зҡ„жңҖе°ҸATOMзј–з Ғпјү
 */
function computeSortKey(notes) {
    const codes = [];
    for (const note of notes) {
        const m = note.path.match(/ATOM@([\d.]+)/);
        if (m) codes.push(m[1]);
    }
    if (codes.length === 0) return 'ZZZZZZ';

    codes.sort((a, b) => {
        const pa = a.split('.').map(Number);
        const pb = b.split('.').map(Number);
        for (let i = 0; i < Math.max(pa.length, pb.length); i++) {
            const diff = (pa[i] || 0) - (pb[i] || 0);
            if (diff !== 0) return diff;
        }
        return 0;
    });

    return codes[0].split('.').map(p => String(p).padStart(3, '0')).join('.');
}

async function fullScan() {
    // дёҙж—¶mapпјҡbaseKeyword -> { notes: [], type }
    const keywordMap = new Map();
    const files = app.vault.getMarkdownFiles();

    for (const file of files) {
        if (SCAN_FOLDER && !file.path.startsWith(SCAN_FOLDER)) continue;

        try {
            // 1. д»ҺyamlжҸҗеҸ– - жіЁж„Ҹеӯ—ж®өеҗҚжҳҜдёӯж–Ү"е…ій”®иҜҚз®ЎзҗҶ"
            const cache = app.metadataCache.getFileCache(file);
            const yamlKws = cache?.frontmatter?.['е…ій”®иҜҚз®ЎзҗҶ'] || [];
            if (Array.isArray(yamlKws)) {
                yamlKws.forEach(kw => {
                    if (typeof kw === 'string' && kw.trim()) {
                        const { base, suffix } = parseKeywordWithSuffix(kw.trim());
                        addToKeywordMap(keywordMap, base, file, suffix, 'yaml');
                    }
                });
            }

            // 2. д»Һж–Үд»¶еҗҚжҸҗеҸ– ++...++
            // жіЁж„ҸпјҡеҶ…е®№дёӯеҸҜиғҪеҢ…еҗ«еҚ•дёӘ+еҸ·дҪңдёәеҗҺзјҖеҲҶйҡ”з¬ҰпјҢжүҖд»ҘдёҚиғҪз”Ё[^+]+
            // дҪҝз”ЁйқһиҙӘе©ӘеҢ№й…ҚпјҢеҢ№й…ҚеҲ°дёӢдёҖдёӘ++дёәжӯў
            const fileName = file.basename;
            const plusRegex = /\+\+(.+?)\+\+/g;
            let match;
            while ((match = plusRegex.exec(fileName)) !== null) {
                const { base, suffix } = parseKeywordWithSuffix(match[1].trim());
                addToKeywordMap(keywordMap, base, file, suffix, 'plus');
            }

            // 3. д»Һж–Үд»¶еҗҚжҸҗеҸ– В·В·...В·В·
            const dotRegex = /В·В·([^В·]+)В·В·/g;
            while ((match = dotRegex.exec(fileName)) !== null) {
                const { base, suffix } = parseKeywordWithSuffix(match[1].trim());
                addToKeywordMap(keywordMap, base, file, suffix, 'dot');
            }

        } catch (e) {
            // и·іиҝҮ
        }
    }

    // дҝқз•ҷзҺ°жңүзҡ„parentIdе’Ңorder
    const oldKeywords = { ...state.keywords };

    // иҪ¬жҚўдёәstate.keywordsж јејҸ
    state.keywords = {};
    for (const [baseKeyword, info] of keywordMap.entries()) {
        const id = generateKeywordId(baseKeyword);
        const oldData = oldKeywords[id] || {};

        state.keywords[id] = {
            id: id,
            baseKeyword: baseKeyword,
            displayText: getDisplayText(baseKeyword),
            sortLetter: getSortLetter(baseKeyword),
            type: info.type,
            notes: Array.from(info.notes.values()),
            sortKey: computeSortKey(Array.from(info.notes.values())),
            // дҝқз•ҷз”ЁжҲ·и®ҫзҪ®зҡ„еұӮзә§е…ізі»
            parentId: oldData.parentId || null,
            order: oldData.order ?? 0
        };
    }
    pruneUiStateByKeywords();

    await saveIndex();
    return state.keywords;
}

function addToKeywordMap(map, baseKeyword, file, suffix, type) {
    if (!map.has(baseKeyword)) {
        map.set(baseKeyword, { notes: [], type: type });
    }
    const entry = map.get(baseKeyword);

    // е…Ғи®ёеҗҢдёҖж–Үд»¶/еҗҢдёҖжқҘжәҗзҡ„еӨҡдёӘеҗҺзјҖе№¶еӯҳпјҢз”Ёж•°з»„еӯҳеӮЁ
    entry.notes.push({
        path: file.path,
        name: file.basename,
        suffix: suffix,
        sourceType: type
    });

    // зұ»еһӢдјҳе…Ҳзә§: yaml > plus > dotпјҲз”ЁдәҺе…ій”®иҜҚж•ҙдҪ“й»ҳи®ӨжҳҫзӨәзұ»еһӢпјү
    const priority = { dot: 1, plus: 2, yaml: 3 };
    if (priority[type] > priority[entry.type]) {
        entry.type = type;
    }
}

/**
 * еҸҚеҗ‘еҗҢжӯҘпјҡжӣҙж–°еҺҹе§Ӣз¬”и®°ж–Үд»¶дёӯзҡ„е…ій”®иҜҚ
 * @param {string} oldKeyword - еҺҹе…ій”®иҜҚпјҲеҹәзЎҖйғЁеҲҶпјү
 * @param {string} newKeyword - ж–°е…ій”®иҜҚ
 * @param {Array} notes - е…іиҒ”зҡ„з¬”и®°еҲ—иЎЁ [{path, name, suffix}]
 * @param {string} type - е…ій”®иҜҚзұ»еһӢ 'yaml' | 'plus' | 'dot'
 */
async function syncKeywordToNotes(oldKeyword, newKeyword, notes, type) {
    if (oldKeyword === newKeyword) return { updated: 0, failed: 0 };

    let updated = 0;
    let failed = 0;

    for (const note of notes) {
        try {
            const file = app.vault.getAbstractFileByPath(note.path);
            if (!file) continue;

            let content = await app.vault.read(file);
            let modified = false;

            // 1. жӣҙж–°YAMLдёӯзҡ„е…ій”®иҜҚз®ЎзҗҶеӯ—ж®ө
            if (type === 'yaml') {
                // еҢ№й…Қyaml frontmatterдёӯзҡ„е…ій”®иҜҚз®ЎзҗҶеӯ—ж®ө
                const yamlRegex = /^---\n([\s\S]*?)\n---/;
                const yamlMatch = content.match(yamlRegex);
                if (yamlMatch) {
                    const yamlContent = yamlMatch[1];
                    // жһ„е»әе®Ңж•ҙзҡ„еҺҹе…ій”®иҜҚпјҲеҢ…еҗ«еҗҺзјҖпјү
                    const oldFull = note.suffix ? oldKeyword + note.suffix : oldKeyword;
                    const newFull = note.suffix ? newKeyword + note.suffix : newKeyword;

                    // жӣҝжҚўyamlдёӯзҡ„е…ій”®иҜҚ
                    const newYaml = yamlContent.replace(
                        new RegExp(`(е…ій”®иҜҚз®ЎзҗҶ:[\\s\\S]*?)${escapeRegex(oldFull)}`, 'g'),
                        `$1${newFull}`
                    );
                    if (newYaml !== yamlContent) {
                        content = content.replace(yamlMatch[1], newYaml);
                        modified = true;
                    }
                }
            }

            // 2. жӣҙж–°ж–Үд»¶еҗҚдёӯзҡ„ ++...++ жҲ– В·В·...В·В·
            if (type === 'plus' || type === 'dot') {
                const delimiter = type === 'plus' ? '++' : 'В·В·';
                const oldFull = note.suffix ? oldKeyword + note.suffix : oldKeyword;
                const newFull = note.suffix ? newKeyword + note.suffix : newKeyword;
                const oldPattern = `${delimiter}${oldFull}${delimiter}`;
                const newPattern = `${delimiter}${newFull}${delimiter}`;

                // жЈҖжҹҘж–Үд»¶еҗҚжҳҜеҗҰеҢ…еҗ«иҜҘе…ій”®иҜҚ
                if (note.name.includes(oldPattern)) {
                    const newName = note.name.replace(oldPattern, newPattern);
                    const newPath = note.path.replace(note.name, newName);

                    // йҮҚе‘ҪеҗҚж–Үд»¶
                    await app.fileManager.renameFile(file, newPath);
                    modified = true;
                }
            }

            // еҰӮжһңеҶ…е®№жңүдҝ®ж”№пјҲyamlжғ…еҶөпјүпјҢдҝқеӯҳж–Үд»¶
            if (modified && type === 'yaml') {
                await app.vault.modify(file, content);
            }

            if (modified) updated++;
        } catch (e) {
            console.error(`жӣҙж–°з¬”и®°еӨұиҙҘ: ${note.path}`, e);
            failed++;
        }
    }

    return { updated, failed };
}

/**
 * иҪ¬д№үжӯЈеҲҷиЎЁиҫҫејҸзү№ж®Ҡеӯ—з¬Ұ
 */
function escapeRegex(str) {
    return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}

/**
 * жҢү+еҸ·еҲҶеүІsuffixпјҢжһ„е»әз¬”и®°зҡ„еұӮзә§ж ‘з»“жһ„
 * @param {Array} notes - з¬”и®°ж•°з»„ [{path, name, suffix, sourceType}, ...]
 * @returns {Object} ж ‘з»“жһ„ { "з¬¬дёҖеұӮ": { children: { "з¬¬дәҢеұӮ": { children: { ... }, notes: [...] } }, notes: [...] } }
 */
function buildSuffixTree(notes) {
    const root = { children: new Map(), notes: [] };

    notes.forEach(note => {
        const suffix = note.suffix || '';
        // еҲҶеүІsuffixпјҢз§»йҷӨжҜҸдёӘйғЁеҲҶеүҚйқўзҡ„+еҸ·
        const parts = suffix.split('+').filter(p => p.trim());

        if (parts.length === 0) {
            // ж— suffixпјҢзӣҙжҺҘж”ҫеңЁж №иҠӮзӮ№
            root.notes.push(note);
        } else {
            let current = root;
            // йҒҚеҺҶжҜҸдёҖеұӮ
            parts.forEach((part, idx) => {
                const trimmedPart = part.trim();
                if (!trimmedPart) return;

                if (!current.children.has(trimmedPart)) {
                    current.children.set(trimmedPart, { children: new Map(), notes: [] });
                }
                current = current.children.get(trimmedPart);

                // жңҖеҗҺдёҖеұӮж”ҫз¬”и®°
                if (idx === parts.length - 1) {
                    current.notes.push(note);
                }
            });
        }
    });

    return root;
}

/**
 * з»ҹи®ЎиҠӮзӮ№зҡ„жүҖжңүеҗҺд»Јз¬”и®°ж•°йҮҸпјҲеҢ…жӢ¬иҮӘиә«пјү
 */
function countAllNotes(node) {
    let count = node.notes.length;
    node.children.forEach(child => {
        count += countAllNotes(child);
    });
    return count;
}

/**
 * ж”¶йӣҶиҠӮзӮ№зҡ„жүҖжңүз¬”и®°пјҲжүҒе№іеҢ–пјҢеҢ…жӢ¬иҮӘиә«е’ҢжүҖжңүеҗҺд»Јпјү
 */
function collectAllNotes(node) {
    let allNotes = [...node.notes];
    node.children.forEach(child => {
        allNotes = allNotes.concat(collectAllNotes(child));
    });
    return allNotes;
}

                noteToggle.style.transform = isNoteExpanded ? 'rotate(90deg)' : '';
                notesContainer.style.display = isNoteExpanded ? 'block' : 'none';

                noteWrapper.onclick = (e) => {
                    e.stopPropagation();
                    isNoteExpanded = !isNoteExpanded;
                    noteToggle.style.transform = isNoteExpanded ? 'rotate(90deg)' : '';
                    notesContainer.style.display = isNoteExpanded ? 'block' : 'none';
                };

                const sortedNotes = [...childNode.notes].sort(compareNoteName);
                // жёІжҹ“з¬”и®°йЎ№
                sortedNotes.forEach(note => {
                    const noteItem = notesContainer.createEl('div', { cls: 'kw-drawer-note' });
                    noteItem.style.marginLeft = '20px';
                    noteItem.createEl('span', { text: 'рҹ“„' });
                    noteItem.createEl('span', { text: note.name });
                    noteItem.onclick = (e) => {
                        e.stopPropagation();
                        app.workspace.openLinkText(note.path, '');
                    };
                });
            }

            // йҖ’еҪ’жёІжҹ“еӯҗеҶ…е®№
            renderDrawerTree(childContainer, childNode, level + 1, type, expandNotesByDefault);
        } else {
            // === жңҖж·ұеұӮзә§зҡ„иҠӮзӮ№пјҲж— еӯҗеұӮзә§пјүпјҡжҳҫзӨәеұӮзә§еҗҚ + еҸҜжҠҳеҸ зҡ„"рҹ“қ ж №з¬”и®°"жҢүй’® ===
            const leafLabel = container.createEl('div', { cls: 'kw-drawer-parent' });
            leafLabel.style.marginLeft = `${level * 20}px`;

            const toggle = leafLabel.createEl('span', { cls: 'kw-drawer-toggle', text: 'в–ј' });
            const label = leafLabel.createEl('span', { cls: 'kw-drawer-label', text: childName });
            const count = leafLabel.createEl('span', { cls: 'kw-drawer-count', text: `(${childNode.notes.length})` });

            // з¬”и®°е®№еҷЁпјҲй»ҳи®ӨжҠҳеҸ пјү
            const notesContainer = leafLabel.createEl('div', { cls: 'kw-drawer-children' });
            notesContainer.style.display = 'none';

            const sortedNotes = [...childNode.notes].sort(compareNoteName);
            // жёІжҹ“иҜҘеҸ¶еӯҗиҠӮзӮ№зҡ„з¬”и®°
            sortedNotes.forEach(note => {
                const noteItem = notesContainer.createEl('div', { cls: 'kw-drawer-note' });
                noteItem.style.marginLeft = '20px';
                noteItem.createEl('span', { text: 'рҹ“„' });
                noteItem.createEl('span', { text: note.name });
                noteItem.onclick = (e) => {
                    e.stopPropagation();
                    app.workspace.openLinkText(note.path, '');
                };
            });

            // зӮ№еҮ»дәӢд»¶пјҡеҲҮжҚўз¬”и®°е®№еҷЁжҳҫзӨә/йҡҗи—Ҹ
            let isNotesExpanded = expandNotesByDefault;
            toggle.style.transform = isNotesExpanded ? 'rotate(90deg)' : '';
            leafLabel.onclick = (e) => {
                e.stopPropagation();
                isNotesExpanded = !isNotesExpanded;
                toggle.style.transform = isNotesExpanded ? 'rotate(90deg)' : '';
                notesContainer.style.display = isNotesExpanded ? 'block' : 'none';
            };

            container.appendChild(leafLabel);
        }
    });

    // --- жёІжҹ“еҪ“еүҚеұӮзә§зҡ„з¬”и®°пјҲзӣҙеұһпјү- д»…й’ҲеҜ№ Level 0 ---
    // v3.7: Level 0 зҡ„з¬”и®°еңЁжүҖжңүеӯҗеұӮзә§жёІжҹ“е®ҢеҗҺжҳҫзӨәпјҢеҸҜжҠҳеҸ "рҹ“қ ж №з¬”и®°"
    if (node.notes.length > 0 && level === 0) {
        const noteWrapper = container.createEl('div', { cls: 'kw-drawer-parent' });
        noteWrapper.style.marginLeft = `${level * 20}px`;

        const noteToggle = noteWrapper.createEl('span', { cls: 'kw-drawer-toggle', text: 'в–ј' });
        noteWrapper.createEl('span', { cls: 'kw-drawer-label', text: 'рҹ“қ ж №з¬”и®°' });
        noteWrapper.createEl('span', { cls: 'kw-drawer-count', text: `(${node.notes.length})` });

        // з¬”и®°е®№еҷЁпјҲй»ҳи®ӨжҠҳеҸ пјү
        const notesContainer = noteWrapper.createEl('div', { cls: 'kw-drawer-children' });
        notesContainer.style.display = 'none';

        // з»‘е®ҡзӮ№еҮ»дәӢд»¶
        let isExpanded = expandNotesByDefault;
        noteToggle.style.transform = isExpanded ? 'rotate(90deg)' : '';

        noteWrapper.onclick = (e) => {
            e.stopPropagation();
            isExpanded = !isExpanded;
            noteToggle.style.transform = isExpanded ? 'rotate(90deg)' : '';
            notesContainer.style.display = isExpanded ? 'block' : 'none';
        };

        const sortedNotes = [...node.notes].sort(compareNoteName);
        // жёІжҹ“з¬”и®°йЎ№
        sortedNotes.forEach(note => {
            const noteItem = notesContainer.createEl('div', { cls: 'kw-drawer-note' });
            noteItem.dataset.level = level;
            noteItem.style.marginLeft = '20px';
            noteItem.createEl('span', { text: 'рҹ“„' });
            noteItem.createEl('span', { text: note.name });
            noteItem.onclick = (e) => {
                e.stopPropagation();
                app.workspace.openLinkText(note.path, '');
            };
        });
    }
}

/**
 * з»ҹи®ЎиҠӮзӮ№зҡ„жүҖжңүеҗҺд»Јз¬”и®°ж•°йҮҸ
 */
function countDescendants(node) {
    let count = node.notes.length;
    node.children.forEach(child => {
        count += countDescendants(child);
    });
    return count;
}

 * @param {HTMLElement} drawer - жҠҪеұүе®№еҷЁ
 * @param {Array} notes - иҜҘзұ»еһӢзҡ„з¬”и®°ж•°з»„

function renderMultiLevelDrawer(drawer, notes, type, showGroupHeader = false, expandNotesByDefault = false) {
    if (notes.length === 0) return;

    // жһ„е»әеұӮзә§ж ‘

    // v3.7: дј йҖ’ expandNotesByDefault еҸӮж•°пјҲй»ҳи®ӨжҠҳеҸ пјү
    renderDrawerTree(drawer, tree, 0, type, expandNotesByDefault);

/**
 * жёІжҹ“еҚ•дёӘе…ій”®иҜҚиғ¶еӣҠпјҲеҘ—еңҲејҸпјҢй»ҳи®Өеұ•ејҖпјү
 */
function renderCapsule(container, kw) {
    if (!kw || !kw.type) {
        console.error('ж— ж•Ҳзҡ„е…ій”®иҜҚеҜ№иұЎ:', kw);
        return;
    }

    const children = getChildKeywords(kw.id);
    const hasChildren = children.length > 0;
    ensureUiStateSets();
    const isCollapsed = state.collapsedIds.has(kw.id);
    const isNotesOpen = state.notesOpenIds?.has?.(kw.id) ?? false;

    // жЈҖжҹҘжҳҜеҗҰжңүеӨҡз§ҚжқҘжәҗзұ»еһӢ
    const sourceTypes = getKeywordSourceTypes(kw);
    const isMixedType = sourceTypes.size > 1;

    const capsule = container.createEl('div', {
        cls: `kw-capsule ${hasChildren ? 'has-children' : ''} ${isCollapsed ? 'collapsed' : 'expanded'} ${isNotesOpen ? 'notes-open' : ''}`
    });

    // еӨҙйғЁ - ж··еҗҲзұ»еһӢдҪҝз”Ёзү№ж®Ҡж ·ејҸ
    const headerCls = isMixedType ? 'kw-capsule-header type-mixed' : `kw-capsule-header type-${kw.type}`;
    const header = capsule.createEl('div', { cls: headerCls });

    if (hasChildren) {
        const toggle = header.createEl('span', {
            cls: 'kw-capsule-toggle',
            text: 'в–¶'
        });
        toggle.onclick = (e) => {
            e.stopPropagation();
            ensureUiStateSets();
            if (state.collapsedIds.has(kw.id)) {
                state.collapsedIds.delete(kw.id);
            } else {
                state.collapsedIds.add(kw.id);
            }
            renderViewFn();
            persistUiState();
        };
    }

    // еӣҫж Ү - ж··еҗҲзұ»еһӢжҳҫзӨәеӨҡдёӘеӣҫж Ү
    if (isMixedType) {
        const typeOrder = ['yaml', 'plus', 'dot'];
        const icons = typeOrder.filter(t => sourceTypes.has(t)).map(t => COLORS[t].icon);
        header.createEl('span', { cls: 'kw-capsule-icon kw-capsule-icon-mixed', text: icons.join('') });
    } else {
        header.createEl('span', { cls: 'kw-capsule-icon', text: COLORS[kw.type].icon });
    }

    header.createEl('span', { cls: 'kw-capsule-label', text: kw.displayText });

    if (kw.notes.length > 0) {
        header.createEl('span', { cls: 'kw-capsule-badge', text: `${kw.notes.length}` });
    }

    // зӮ№еҮ»дәӢд»¶ - жүҖжңүиғ¶еӣҠз»ҹдёҖи§ҰеҸ‘дәӨдә’
    header.onclick = (e) => {
        e.stopPropagation();
        handleCapsuleClick(kw);
    };

    // з»ҹдёҖзҡ„зӮ№еҮ»еӨ„зҗҶеҮҪж•°
    function handleCapsuleClick(keyword) {
        ensureUiStateSets();

        if (keyword.notes.length === 1) {
            // еҸӘжңүдёҖдёӘз¬”и®°пјҢзӣҙжҺҘжү“ејҖ
            app.workspace.openLinkText(keyword.notes[0].path, '');
        } else if (keyword.notes.length > 1) {
            // еӨҡдёӘз¬”и®°пјҢеҲҮжҚўз¬”и®°жҠҪеұү
            if (state.notesOpenIds.has(keyword.id)) {
                state.notesOpenIds.delete(keyword.id);
            } else {
                state.notesOpenIds.add(keyword.id);
            }
            renderViewFn();
        }
        // еӯҗиҠӮзӮ№жҠҳеҸ /еұ•ејҖз”ұе·Ұдҫ§дёүи§’ејҖе…іиҙҹиҙЈ
    }

    // з¬”и®°жҠҪеұүпјҲv3.4: еӨҡеұӮзә§жҢү+еҸ·еҲҶеүІеұ•зӨәпјү
    // ж”ҫеңЁеӯҗе…ій”®иҜҚе®№еҷЁд№ӢеүҚпјҢзЎ®дҝқеұ•ејҖеҗҺзӣҙжҺҘжҳҫзӨәеңЁеҪ“еүҚиғ¶еӣҠдёӢж–№
    if (kw.notes.length > 0) {
        const drawer = capsule.createEl('div', {
            cls: 'kw-notes-drawer',
            style: isNotesOpen ? '' : 'display: none;' // й»ҳи®Өйҡҗи—Ҹз¬”и®°жҠҪеұү
        });

        // жҢүжқҘжәҗзұ»еһӢеҲҶз»„з¬”и®°
        const typeOrder = ['yaml', 'plus', 'dot'];
        const typeNames = {
            yaml: 'в—Ҷ YAMLе…ій”®иҜҚз®ЎзҗҶ',
            plus: 'в–І еұӮзә§е…ій”®иҜҚ ++...++',
            dot: 'в—Ҹ иЎҢеҶ…е…ій”®иҜҚ В·В·...В·В·'
        };
        const groupedNotes = { yaml: [], plus: [], dot: [] };

        kw.notes.forEach(note => {
            const sourceType = note.sourceType || kw.type; // е…је®№ж—§ж•°жҚ®
            if (groupedNotes[sourceType]) {
                groupedNotes[sourceType].push(note);
            }
        });

        // жЈҖжҹҘжҳҜеҗҰжңүеӨҡз§ҚжқҘжәҗзұ»еһӢ
        const activeTypes = typeOrder.filter(t => groupedNotes[t].length > 0);
        const hasMultipleTypes = activeTypes.length > 1;

        // жҢүзұ»еһӢйЎәеәҸжёІжҹ“
        typeOrder.forEach(type => {
            const notes = groupedNotes[type];
            if (notes.length === 0) return;

            // еҰӮжһңжңүеӨҡз§Қзұ»еһӢпјҢжҳҫзӨәеҲҶз»„ж Үйўҳ
            if (hasMultipleTypes) {
                const groupHeader = drawer.createEl('div', { cls: `kw-drawer-group-header type-${type}` });
                groupHeader.createEl('span', { text: COLORS[type].icon });
                groupHeader.createEl('span', { text: typeNames[type] });
                groupHeader.createEl('span', { cls: 'kw-drawer-group-count', text: `${notes.length}` });
            }

            // дҪҝз”ЁеӨҡеұӮзә§жёІжҹ“еҮҪж•°
            // v3.7: з¬”и®°й»ҳи®ӨжҠҳеҸ пјҢз”ЁжҲ·зӮ№еҮ»"рҹ“қ ж №з¬”и®°"еұ•ејҖ
            renderMultiLevelDrawer(drawer, notes, type, false, false);
        });
    }

    // еӯҗе…ій”®иҜҚе®№еҷЁпјҲеҘ—еңҲж•ҲжһңпјҢй»ҳи®Өеұ•ејҖпјү
    if (hasChildren) {
        const childrenContainer = capsule.createEl('div', { cls: 'kw-capsule-children' });
        children.forEach(child => {
            renderCapsule(childrenContainer, child);
        });
    }
}

        const descendantIds = new Set(getDescendantIds(kwId));
        const availableParents = Object.values(state.keywords)
            .filter(k => k.id !== kwId && !descendantIds.has(k.id))
            .sort((a, b) => a.displayText.localeCompare(b.displayText, 'zh-CN'));

        // еҲӣе»әиҮӘе®ҡд№үйҖүжӢ©еҷЁе®№еҷЁ
        const selectorContainer = parentRow.createEl('div', { cls: 'kw-parent-selector' });

        // иҫ“е…ҘжЎҶеҢ…иЈ…еҷЁ
        const inputWrapper = selectorContainer.createEl('div', { cls: 'kw-parent-input-wrapper' });
        const parentInput = inputWrapper.createEl('input', {
            type: 'text',
            cls: 'kw-parent-input',
            attr: { placeholder: 'иҫ“е…Ҙе…ій”®иҜҚжҗңзҙў...' }
        });
        const clearBtn = inputWrapper.createEl('span', { cls: 'kw-parent-clear', text: 'Г—' });

        // дёӢжӢүеҲ—иЎЁ
        const dropdown = selectorContainer.createEl('div', { cls: 'kw-parent-dropdown' });

        // йҖүдёӯзҡ„зҲ¶зә§IDпјҲз”ЁдәҺдҝқеӯҳпјү
        let selectedParentId = kw.parentId || null;
        let highlightedIndex = -1;

        // еҲқе§ӢеҢ–жҳҫзӨә
        if (selectedParentId) {
            const selectedKw = state.keywords[selectedParentId];
            if (selectedKw) {
                parentInput.value = selectedKw.displayText;
            }
        }

        // й«ҳдә®еҢ№й…Қж–Үжң¬зҡ„еҮҪж•°
        function highlightMatch(text, query) {
            if (!query) return text;
            const lowerText = text.toLowerCase();
            const lowerQuery = query.toLowerCase();
            const index = lowerText.indexOf(lowerQuery);
            if (index === -1) return text;
            return text.substring(0, index) +
                   '<span class="kw-parent-option-match">' +
                   text.substring(index, index + query.length) +
                   '</span>' +
                   text.substring(index + query.length);
        }

        // жёІжҹ“дёӢжӢүйҖүйЎ№
        function renderOptions(query = '') {
            dropdown.innerHTML = '';
            const lowerQuery = query.toLowerCase().trim();

            // зӯӣйҖүеҢ№й…Қзҡ„йҖүйЎ№
            const filtered = availableParents.filter(k => {
                if (!lowerQuery) return true;
                return k.displayText.toLowerCase().includes(lowerQuery) ||
                       k.baseKeyword.toLowerCase().includes(lowerQuery);
            });

            // йЎ¶еұӮйҖүйЎ№пјҲе§Ӣз»ҲжҳҫзӨәпјү
            const topOption = dropdown.createEl('div', {
                cls: `kw-parent-option ${!selectedParentId ? 'selected' : ''}`,
                attr: { 'data-value': '' }
            });
            topOption.innerHTML = `<span class="kw-parent-option-icon">рҹҸ </span><span class="kw-parent-option-text">-- йЎ¶еұӮе…ій”®иҜҚ --</span>`;
            topOption.onclick = () => selectOption('', '-- йЎ¶еұӮе…ій”®иҜҚ --');

            if (filtered.length === 0 && lowerQuery) {
                dropdown.createEl('div', { cls: 'kw-parent-no-results', text: `жңӘжүҫеҲ°еҢ№й…Қ "${query}" зҡ„е…ій”®иҜҚ` });
            } else {
                filtered.forEach((k, idx) => {
                    const option = dropdown.createEl('div', {
                        cls: `kw-parent-option ${selectedParentId === k.id ? 'selected' : ''} ${idx === highlightedIndex ? 'highlighted' : ''}`,
                        attr: { 'data-value': k.id }
                    });

                    option.innerHTML = `<span class="kw-parent-option-icon">${COLORS[k.type].icon}</span><span class="kw-parent-option-text">${highlightedText}</span>`;

function renderManagerItem(container, kw, onEdit, level = 0) {

    const item = container.createEl('div', {
        cls: 'kw-mgr-item',

        state.dragId = null;
        item.classList.remove('dragging');

        }

        e.preventDefault();
        e.stopPropagation();
        item.classList.remove('drag-over');
        const draggedId = state.dragId;
        if (draggedId && draggedId !== kw.id && state.keywords[draggedId]) {
            // жЈҖжҹҘеҫӘзҺҜеј•з”Ё
            const descendants = getDescendantIds(draggedId);
            if (descendants.includes(kw.id)) {
                new Notice('вҡ пёҸ дёҚиғҪе°ҶзҲ¶зә§жӢ–е…Ҙе…¶еӯҗзә§');

   - жҜҸдёӘжңүеӯҗзә§зҡ„иғ¶еӣҠе·Ұдҫ§еўһеҠ дёүи§’ејҖе…іпјҢеҚ•зӢ¬жҺ§еҲ¶еӯҗзә§еұ•ејҖ/жҠҳеҸ 
   - жҠҳеҸ зҠ¶жҖҒдјҡеҶҷе…Ҙ `data/keyword_index.json` зҡ„жүҒе№іеӯ—е…ё `capsuleExpanded`
   - жҠҳеҸ жҖҒйўңиүІжӣҙж·ұпјҢдҫҝдәҺеҝ«йҖҹиҜҶеҲ«еҪ“еүҚеӨ„дәҺж”¶иө·зҠ¶жҖҒзҡ„иғ¶еӣҠ

## дҪҝз”ЁиҜҙжҳҺ

### 1. йғЁзҪІ

1. еңЁObsidianдёӯеҲӣе»әдёҖдёӘж–°з¬”и®°
2. е°ҶдёҠйқўзҡ„д»Јз Ғеқ—е®Ңж•ҙеӨҚеҲ¶иҝӣеҺ»
3. еҲҮжҚўеҲ°йҳ…иҜ»жЁЎејҸеҚіеҸҜзңӢеҲ°е…ій”®иҜҚзҙўеј•

### 2. дёүз§Қе…ій”®иҜҚзұ»еһӢ

| зұ»еһӢ | ж јејҸ | йўңиүІ | иҜҙжҳҺ |
|------|------|------|------|
| в—Ҷ YAMLз®ЎзҗҶ | `е…ій”®иҜҚз®ЎзҗҶ:` еӯ—ж®ө | и“қиүІ | yaml frontmatterдёӯе®ҡд№ү |
| в–І еұӮзә§е…ій”®иҜҚ | `++е…ій”®иҜҚ++` | ж©ҷиүІ | ж–Үд»¶еҗҚдёӯз”ЁеҸҢеҠ еҸ·еҢ…иЈ№ |
| в—Ҹ иЎҢеҶ…е…ій”®иҜҚ | `В·В·е…ій”®иҜҚВ·В·` | зҙ«иүІ | ж–Үд»¶еҗҚдёӯз”ЁеҸҢзӮ№еҸ·еҢ…иЈ№ |

### 3. е…ій”®иҜҚжҸҗеҸ–и§„еҲҷ

**йҮҚиҰҒ**пјҡж–Үд»¶еҗҚдёӯзҡ„`+`еҸ·жҳҜ**еҗҺзјҖжіЁи§Ј**пјҢдёҚжҳҜеұӮзә§еҲҶйҡ”з¬ҰпјҒ

зӨәдҫӢпјҡ
- ж–Үд»¶еҗҚ `В·В·MMKV-AдёҠйЎ№+ж–°ејҖе§ӢВ·В·`
  - еҹәзЎҖе…ій”®иҜҚпјҡ`MMKV-AдёҠйЎ№`
  - еҗҺзјҖпјҡ`+ж–°ејҖе§Ӣ`
- ж–Үд»¶еҗҚ `В·В·MMKV-AдёҠйЎ№+иө°дёҠйЎ№жөҒзЁӢзҡ„пјҢжқҗж–ҷВ·В·`
  - еҹәзЎҖе…ій”®иҜҚпјҡ`MMKV-AдёҠйЎ№`пјҲдёҺдёҠйқўзӣёеҗҢпјү
  - еҗҺзјҖпјҡ`+иө°дёҠйЎ№жөҒзЁӢзҡ„пјҢжқҗж–ҷ`

иҝҷдёӨдёӘж–Үд»¶йғҪдјҡиў«еҪ’е…ҘеҗҢдёҖдёӘ`MMKV-AдёҠйЎ№`иғ¶еӣҠдёӯпјҢжү“ејҖжҠҪеұүеҸҜд»ҘзңӢеҲ°еҗ„иҮӘзҡ„еҗҺзјҖгҖӮ

### 4. е…ій”®иҜҚеұӮзә§е…ізі»

еұӮзә§е…ізі»жҳҜ**е…ій”®иҜҚд№Ӣй—ҙ**зҡ„еұӮзә§пјҢдёҺж–Үд»¶еҗҚдёӯзҡ„`+`еҸ·ж— е…ігҖӮ

з”ЁжҲ·еҸҜд»ҘйҖҡиҝҮпјҡ
1. зӮ№еҮ»"вҡҷпёҸ з®ЎзҗҶе…ій”®иҜҚ"иҝӣе…Ҙз®ЎзҗҶз•Ңйқў
2. жӢ–жӢҪе…ій”®иҜҚеҲ°е…¶д»–е…ій”®иҜҚдёҠпјҢе»әз«ӢзҲ¶еӯҗе…ізі»
3. жӢ–жӢҪеҲ°йЎ¶йғЁеҢәеҹҹпјҢдҪҝе…¶жҲҗдёәйЎ¶еұӮе…ій”®иҜҚ
4. зӮ№еҮ»зј–иҫ‘жҢүй’®дҝ®ж”№еҗҚз§°жҲ–йҖүжӢ©зҲ¶зә§

### 5. дёӨз§Қи§Ҷеӣҫ

- **еұ•зӨәи§Ҷеӣҫ**пјҲй»ҳи®ӨпјүпјҡеҘ—еңҲејҸUIпјҢй«ҳеұӮзә§е…ій”®иҜҚеҢ…еҗ«дҪҺеұӮзә§
- **з®ЎзҗҶи§Ҷеӣҫ**пјҡGTDйЈҺж јзҡ„жӢ–жӢҪз•ҢйқўпјҢеҸҜи°ғж•ҙеұӮзә§е’ҢйҮҚе‘ҪеҗҚ

### 6. ж•°жҚ®ж–Үд»¶

зҙўеј•ж•°жҚ®дҝқеӯҳеңЁ `data/keyword_index.json`

### 7. v3.7еҠҹиғҪпјҡеӨҡеұӮзә§жҠҪеұүпјҲеҸҜжҠҳеҸ ж №з¬”и®°пјү

зӮ№еҮ»иғ¶еӣҠеұ•ејҖз¬”и®°жҠҪеұүж—¶пјҢз¬”и®°дјҡжҢү`+`еҸ·еҲҶеүІжҳҫзӨәеӨҡеұӮзә§з»“жһ„пјҡ

**зӨәдҫӢж•Ҳжһң**пјҡ
```
е…ій”®иҜҚз®ЎзҗҶ жҠҪеұүпјҡ
в”ңв”Җв”Җ иҮӘе·ұеҒҡзҡ„и°ғжҹҘ
в”Ӯ   в””в”Җв”Җ рҹ“қ ж №з¬”и®° (зӮ№еҮ»еұ•ејҖ)
в”Ӯ       в””в”Җв”Җ еӯҰд№ и®ЎеҲ’.md
в”ңв”Җв”Җ жҘһдёҘз»ҸеҲҶжһҗ
в”Ӯ   в””в”Җв”Җ рҹ“қ ж №з¬”и®° (зӮ№еҮ»еұ•ејҖ)
в”Ӯ       в””в”Җв”Җ гҖҠжҘһдёҘз»ҸгҖӢеӯҰд№ з¬”и®°.md
в”ңв”Җв”Җ и°ғжҹҘй—®еҚ·и®ҫи®Ў.mdпјҲж— +еҗҺзјҖпјҢзӣҙжҺҘжҳҫзӨәпјү
в”ңв”Җв”Җ еҸҰдёҖдёӘйЎ№зӣ®
в”Ӯ   в””в”Җв”Җ рҹ“қ ж №з¬”и®° (зӮ№еҮ»еұ•ејҖ)
в”Ӯ       в””в”Җв”Җ йЎ№зӣ®и®ЎеҲ’.md
в””в”Җв”Җ еҹәзЎҖз¬”и®°.mdпјҲж— +еҗҺзјҖпјү
```

**v3.7дәӨдә’ж–№ејҸ**пјҡ
- й»ҳи®ӨеҸӘжҳҫзӨәеұӮзә§еҗҚпјҢз¬”и®°й»ҳи®ӨжҠҳеҸ 
- еҰӮжһңеұӮзә§жңүзӣҙеұһз¬”и®°пјҢдјҡжҳҫзӨәеҸҜжҠҳеҸ зҡ„"рҹ“қ ж №з¬”и®°"жҢүй’®
- зӮ№еҮ»"рҹ“қ ж №з¬”и®°"жҢүй’®еҸҜеұ•ејҖ/жҠҳеҸ жҹҘзңӢиҜҘеұӮзә§зҡ„з¬”и®°
- зӮ№еҮ»еұӮзә§ж ҮзӯҫпјҲв–¶ иҮӘе·ұеҒҡзҡ„и°ғжҹҘпјүеҸҜжҠҳеҸ /еұ•ејҖеӯҗеұӮзә§
- жҜҸеұӮжҳҫзӨәиҜҘеұӮзә§еҸҠжүҖжңүеҗҺд»Јдёӯзҡ„з¬”и®°ж•°йҮҸ
- зӣҙжҺҘзӮ№еҮ»з¬”и®°еҗҚз§°еҸҜжү“ејҖз¬”и®°

**еұӮзә§и§„еҲҷ**пјҡ
- `+`еҸ·еҲҶеүІзҡ„жҜҸдёҖзә§дҪңдёәдёҖдёӘеұӮзә§иҠӮзӮ№
- ж— еҗҺзјҖзҡ„з¬”и®°ж”ҫеңЁж №иҠӮзӮ№пјҲзӣҙеұһе…ій”®иҜҚпјүпјҢжҳҫзӨәдёә"рҹ“қ ж №з¬”и®°"
- зӣёеҗҢеҗҺзјҖи·Ҝеҫ„зҡ„з¬”и®°еҪ’е…ҘеҗҢдёҖеҲҶж”Ҝ

### 8. v3.8еҠҹиғҪпјҡиғ¶еӣҠдёүи§’ејҖе…ідёҺзҠ¶жҖҒи®°еҝҶ

- жҜҸдёӘжңүеӯҗе…ій”®иҜҚзҡ„иғ¶еӣҠе·Ұдҫ§дјҡжҳҫзӨә `в–¶` дёүи§’ејҖе…і
- дёүи§’ж–№еҗ‘иЎЁзӨәзҠ¶жҖҒпјҡеҸіеҗ‘дёәжҠҳеҸ пјҢдёӢеҗ‘дёәеұ•ејҖ
- зӮ№еҮ»дёүи§’еҸӘжҺ§еҲ¶иҜҘиғ¶еӣҠзҡ„еӯҗзә§пјҢдёҚеҪұе“Қиғ¶еӣҠзӮ№еҮ»жү“ејҖз¬”и®°жҠҪеұүзҡ„йҖ»иҫ‘
- вҖңе…ЁйғЁжҠҳеҸ  / е…ЁйғЁеұ•ејҖвҖқдјҡеҗҢжӯҘжӣҙж–° `capsuleExpanded` еӯ—е…ё
