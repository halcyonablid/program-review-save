// 缓存 key 基于 JSON 文件修改时间
const CACHE_KEY = 'evaTreeCache_EVA';
let cachedData = null;
let cachedMtime = null;

function getCacheKey() {
    return evaJsonFile ? `${CACHE_KEY}_${evaJsonFile.stat?.mtime || 0}` : CACHE_KEY;
}

function loadFromCache() {
    try {
        const cached = window[CACHE_KEY];
        if (cached && cached.mtime === evaJsonFile?.stat?.mtime) {
            return cached;
        }
    } catch (e) {
        console.log('缓存读取失败:', e);
    }
    return null;
}

function saveToCache(data) {
    try {
        window[CACHE_KEY] = {
            mtime: evaJsonFile?.stat?.mtime,
            pages: data.pages,
            parentHierarchyCodes: Array.from(data.parentHierarchyCodes),
            maxExistingLevel: data.maxExistingLevel,
            timestamp: Date.now()
        };
    } catch (e) {
        console.log('缓存保存失败:', e);
    }
}

async function ensureLatestEvaData(forceReload = false) {
    // 每次都重新定位文件，避免重建 JSON 后对象引用失效
    const latestJsonFile = app.vault.getFiles().find(f => f.name === JSON_FILENAME);
    if (!latestJsonFile) {
        dv.paragraph(`⚠️ 全库未找到 <code>${JSON_FILENAME}</code>，请先运行 EVA 生成脚本。`);
        return false;
    }

    evaJsonFile = latestJsonFile;
    const latestMtime = evaJsonFile.stat?.mtime || 0;

    if (!forceReload && evaData && evaDataMtime === latestMtime) {
        return true;
    }

    try {
        const content = await app.vault.read(evaJsonFile);
        evaData = JSON.parse(content);
        evaDataMtime = latestMtime;

        // JSON 变化后清理旧缓存，保证同步
        const cached = window[CACHE_KEY];
        if (cached && cached.mtime !== latestMtime) {
            delete window[CACHE_KEY];
        }
        return true;
    } catch (e) {
        dv.paragraph(`❌ JSON 读取失败: ${e.message}`);
        return false;
    }
}

// --- 4. 核心逻辑函数 ---

function getHierarchyCode(fileName) {
    // 从文件名中提取层级编码，如 "ATOM@121.001.007- xxx" -> "121.001.007"
    const atIndex = fileName.indexOf('@');
    if (atIndex === -1) return null;
    const dashIndex = fileName.indexOf('-', atIndex);
    if (dashIndex === -1) return null;
    return fileName.substring(atIndex + 1, dashIndex).trim();
}

function precomputeParentNodes() {
    const allCodes = new Set(pages.map(p => getHierarchyCode(p.file_name)).filter(Boolean));
    parentHierarchyCodes.clear();
    allCodes.forEach(code => {
        const hasChildren = Array.from(allCodes).some(otherCode =>
            otherCode.startsWith(code + '.') && otherCode !== code
        );
        if (hasChildren) parentHierarchyCodes.add(code);
    });
}

function toggleFold(hierarchyCode) {
    if (foldedNodes.has(hierarchyCode)) foldedNodes.delete(hierarchyCode);
    else foldedNodes.add(hierarchyCode);
    // 折叠操作后清除关键词高亮定位，避免自动滚回
    viewState.currentHighlightedFile = null;
    renderTree();
}

function unfoldAncestors(hierarchyCode) {
    if (!hierarchyCode) return;
    const parts = hierarchyCode.split('.');
    for (let i = 1; i < parts.length; i++) {
        const ancestorCode = parts.slice(0, i).join('.');
        if (foldedNodes.has(ancestorCode)) foldedNodes.delete(ancestorCode);
    }
}

function isAncestorFolded(hierarchyCode) {
    if (!hierarchyCode) return false;
    const parts = hierarchyCode.split('.');
    for (let i = 1; i < parts.length; i++) {
        const ancestorCode = parts.slice(0, i).join('.');
        if (foldedNodes.has(ancestorCode)) return true;
    }
    return false;
}

function processKeywords(text) {
    // 处理 ++...++ 和 ··...·· 关键词高亮
    const patterns = [
        {
            regex: /··(.+?)··/g,
            className: 'keyword-highlight',
            marker: '··'
        },
        {
            regex: /\+\+(.+?)\+\+/g,
            className: 'emphasis-highlight',
            marker: '++'
        }
    ];

    const hasMarkers = patterns.some(p => text.includes(p.marker));
    if (!hasMarkers) return document.createTextNode(text);

    const fragment = document.createDocumentFragment();
    let segments = [{ text: text, type: 'normal' }];

    patterns.forEach(pattern => {
        const newSegments = [];

        segments.forEach(segment => {
            if (segment.type !== 'normal') {
                newSegments.push(segment);
                return;
            }

            let lastIndex = 0;
            let match;
            const regex = new RegExp(pattern.regex.source, 'g');

            while ((match = regex.exec(segment.text)) !== null) {
                if (match.index > lastIndex) {
                    newSegments.push({
                        text: segment.text.substring(lastIndex, match.index),
                        type: 'normal'
                    });
                }

                let keyword = match[1];
                if (keyword && /[A-Z]/.test(keyword.charAt(0))) {
                    keyword = keyword.substring(1);
                }

                newSegments.push({
                    text: keyword,
                    type: pattern.className
                });

                lastIndex = regex.lastIndex;
            }

            if (lastIndex < segment.text.length) {
                newSegments.push({
                    text: segment.text.substring(lastIndex),
                    type: 'normal'
                });
            }
        });

        segments = newSegments;
    });

    segments.forEach(segment => {
        if (segment.type === 'normal') {
            fragment.appendChild(document.createTextNode(segment.text));
        } else {
            const span = document.createElement('span');
            span.textContent = segment.text;
            span.className = segment.type;
            fragment.appendChild(span);
        }
    });

    return fragment;
}

// --- 5. 数据加载 (带缓存) ---

async function reloadPagesData(forceReload = false) {
    const loaded = await ensureLatestEvaData(forceReload);
    if (!loaded) return false;

    // 尝试从缓存加载
    const cached = loadFromCache();
    if (cached) {
        console.log('从缓存加载树状图数据, pages数量:', cached.pages?.length || 0);
        pages = cached.pages || [];
        parentHierarchyCodes = new Set(cached.parentHierarchyCodes || []);
        maxExistingLevel = cached.maxExistingLevel || 0;
        return true;
    }

    try {
        console.log('重新解析树状图数据');
        // 从 EVA_Notes.json 读取数据
        const notes = Object.values(evaData?.notes || {});

        // 过滤并排序
        const filteredNotes = notes
            .filter(note => {
                // 检查是否有 树的结构 属性
                const hasTreeStructure = note.yaml && note.yaml["树的结构"] === true;
                // 检查文件名前缀
                const hasPrefix = note.file_name.startsWith(FILE_PREFIX_FILTER);
                return hasTreeStructure && hasPrefix;
            })
            .sort((a, b) => a.file_name.localeCompare(b.file_name, undefined, { numeric: true, sensitivity: 'base' }));

        pages = filteredNotes;

        // 计算最大层级
        maxExistingLevel = 0;
        pages.forEach(page => {
            const code = getHierarchyCode(page.file_name);
            if (code) {
                const effectiveLevels = Math.max(0, code.split('.').length - (START_LEVEL - 1));
                maxExistingLevel = Math.max(maxExistingLevel, effectiveLevels);
            }
        });

        return true;
    } catch (error) {
        console.error('重新加载数据失败:', error);
        return false;
    }
}

function getCurrentMaxVisibleLevel() {
    return Math.min(viewState.maxLevel, maxExistingLevel);
}

// --- 6. 渲染函数 ---

function renderTree() {
    container.innerHTML = "";

    const currentMaxVisibleLevel = getCurrentMaxVisibleLevel();
    levelInfo.innerHTML = `当前层级: ${currentMaxVisibleLevel}/${maxExistingLevel} | ${viewState.highlightLevel > 0 ? `高亮: ${viewState.highlightLevel}${viewState.highlightSubLevels ? " (含下级)" : ""}` : ''} ${viewState.currentHighlightedFile ? `| 定位: ${viewState.currentHighlightedFile}` : ''}`;

    let renderedCount = 0;
    let skippedNoCode = 0;
    let skippedLevelDepth = 0;
    let skippedMaxLevel = 0;
    let skippedFolded = 0;

    pages.forEach(page => {
        const fileName = page.file_name;
        const filePath = page.file_path;
        const hierarchyCode = getHierarchyCode(fileName);
        let levelDepth = 0;

        if (hierarchyCode) {
            levelDepth = Math.max(0, hierarchyCode.split('.').length - (START_LEVEL - 1));
        }

        if (!hierarchyCode) {
            skippedNoCode++;
            // console.log('程序①解析失败:', fileName);
            return;
        }

        if (levelDepth === 0) {
            skippedLevelDepth++;
            return;
        }

        if (levelDepth > viewState.maxLevel) {
            skippedMaxLevel++;
            return;
        }

        if (isAncestorFolded(hierarchyCode)) {
            skippedFolded++;
            return;
        }

        const fileItemDiv = document.createElement("div");
        fileItemDiv.className = "tree-item tree-item-link";
        fileItemDiv.dataset.filename = fileName;
        fileItemDiv.dataset.filepath = filePath;
        fileItemDiv.dataset.level = levelDepth;

        // 检查是否有 树状图划重点 属性
        if (page.yaml && page.yaml["树状图划重点"] === true) {
            fileItemDiv.classList.add("supermemo-entry");
        }

        const displayLevel = levelDepth > 0 ? levelDepth : 1;

        if (viewState.highlightLevel > 0) {
            if (viewState.highlightLevel === displayLevel) {
                fileItemDiv.classList.add("tree-item-highlighted");
            } else if (viewState.highlightSubLevels && displayLevel > viewState.highlightLevel) {
                fileItemDiv.classList.add("tree-item-subhighlighted");
            }
        }

        if (fileName === viewState.currentHighlightedFile) {
            fileItemDiv.classList.add("tree-item-h-highlighted");
        }

        // 折叠控制按钮
        const foldControl = document.createElement('span');
        if (parentHierarchyCodes.has(hierarchyCode)) {
            foldControl.className = 'fold-button';
            foldControl.textContent = foldedNodes.has(hierarchyCode) ? '▶' : '▼';
            foldControl.onclick = (e) => {
                e.stopPropagation();
                toggleFold(hierarchyCode);
            };
        } else {
            foldControl.className = 'fold-placeholder';
            foldControl.innerHTML = '&nbsp;';
        }
        fileItemDiv.appendChild(foldControl);

        // 树形前缀
        let prefix = levelDepth > 1 ? "│ ".repeat(levelDepth - 1) + "├─ " : (levelDepth === 1 ? "● " : "● ");
        const prefixSpan = document.createElement('span');
        prefixSpan.className = 'tree-prefix';
        prefixSpan.textContent = prefix;
        fileItemDiv.appendChild(prefixSpan);

        // 文件名（带关键词高亮）
        fileItemDiv.appendChild(processKeywords(fileName));

        // 点击事件 - 直接打开笔记
        fileItemDiv.addEventListener("click", (e) => {
            e.preventDefault();
            // 直接打开笔记文件
            app.workspace.openLinkText(filePath, "", false);
        });

        // 右键事件 - 高亮同级
        fileItemDiv.addEventListener("contextmenu", (e) => {
            e.preventDefault();
            viewState.highlightLevel = displayLevel;
            viewState.highlightSubLevels = true;
            // 右键高亮操作后清除关键词高亮定位，避免自动滚回
            viewState.currentHighlightedFile = null;
            renderTree();
        });

        container.appendChild(fileItemDiv);
        renderedCount++;
    });

    // 调试：打印渲染统计
    if (skippedNoCode > 0 || skippedMaxLevel > 0) {
        console.log(`程序①渲染统计: 总数=${pages.length}, 渲染=${renderedCount}, 无code=${skippedNoCode}, levelDepth=${skippedLevelDepth}, 超过maxLevel=${skippedMaxLevel}, 折叠祖先=${skippedFolded}`);
    }

    // 恢复滚动位置
    if (typeof viewState.scrollTop === 'number' && viewState.scrollTop > 0) {
        setTimeout(() => { container.scrollTop = viewState.scrollTop; }, 0);
    }

    // 高亮定位
    if (viewState.currentHighlightedFile) {
        setTimeout(() => {
            const highlightedItem = container.querySelector('.tree-item-h-highlighted');
            if (highlightedItem) {
                highlightedItem.scrollIntoView({ behavior: 'smooth', block: 'center' });
                highlightedItem.classList.add('pulse-animation');
                setTimeout(() => highlightedItem.classList.remove('pulse-animation'), 2000);
            }
        }, 100);
    }
}

window.myTreeViewKeydownHandler_EVA = (e) => {
    let updated = false;
    const key = e.key.toLowerCase();

    if (key === 'k') {
        viewState.maxLevel = Math.min(viewState.maxLevel + 1, maxExistingLevel);
        updated = true;
    }
    if (key === 'j') {
        viewState.maxLevel = Math.max(1, viewState.maxLevel - 1);
        if (viewState.highlightLevel > viewState.maxLevel) {
            viewState.highlightLevel = viewState.maxLevel;
        }
        updated = true;
    }
    if (key === 'n') {
        viewState.highlightLevel = Math.max(0, viewState.highlightLevel - 1);
        viewState.highlightSubLevels = false;
        updated = true;
    }
    if (key === 'm') {
        if (viewState.highlightLevel < getCurrentMaxVisibleLevel()) {
            viewState.highlightLevel++;
        }
        viewState.highlightSubLevels = false;
        updated = true;
    }

    if (updated) {
        // 键盘操作后清除关键词高亮定位，避免自动滚回
        viewState.currentHighlightedFile = null;
        renderTree();
    }
};

window.evaTreeScrollToFile = function(fileName) {
    // 去掉可能的 .md 后缀（EVA_Notes.json 中 file_name 通常不带后缀）
    const cleanFileName = fileName.replace(/\.md$/i, '');
    console.log('程序①收到滚动请求:', cleanFileName, '| pages数量:', pages.length);

    // 调试：检查 pages 中是否存在匹配的文件
    const matchingPage = pages.find(p => p.file_name === cleanFileName || p.file_name === fileName);
    if (matchingPage) {
        console.log('程序①找到匹配page:', matchingPage.file_name, '| hierarchyCode:', getHierarchyCode(matchingPage.file_name));
    } else {
        // 尝试模糊匹配
        const fuzzyMatch = pages.find(p => p.file_name.includes(cleanFileName) || cleanFileName.includes(p.file_name));
        if (fuzzyMatch) {
            console.log('程序①模糊匹配到:', fuzzyMatch.file_name);
        } else {
            // 打印前3个 pages 的 file_name 用于调试
            console.log('程序①pages前3个:', pages.slice(0, 3).map(p => p.file_name));
        }
    }

    // 先展开祖先节点
    const hierarchyCode = getHierarchyCode(cleanFileName);
    console.log('程序①解析的hierarchyCode:', hierarchyCode);
    if (hierarchyCode) {
        unfoldAncestors(hierarchyCode);
    }

    // 设置高亮并重新渲染
    viewState.currentHighlightedFile = cleanFileName;
    renderTree();

    // 滚动到目标位置（带重试机制，因为从外部调用时 DOM 更新可能有延迟）
    const maxRetries = 15;
    let retryCount = 0;

    function tryScroll() {
        // 详细调试：检查 container 的属性
        const allItems = container.querySelectorAll('[data-filename]');

        // 方法1: 精确匹配
        let targetItem = container.querySelector(`[data-filename="${CSS.escape(cleanFileName)}"]`);
        if (targetItem) {
            targetItem.scrollIntoView({ behavior: 'smooth', block: 'center' });
            targetItem.classList.add('pulse-animation');
            setTimeout(() => targetItem.classList.remove('pulse-animation'), 2000);
            console.log('程序①精确匹配成功:', cleanFileName);
            return true;
        }

        // 方法2: 遍历查找（更宽松的匹配）
        for (const item of allItems) {
            const fn = item.dataset.filename;
            // 尝试多种匹配方式
            if (fn === cleanFileName ||
                fn === cleanFileName + '.md' ||
                fn + '.md' === cleanFileName ||
                fn.includes(cleanFileName) ||
                cleanFileName.includes(fn)) {
                targetItem = item;
                console.log('程序①遍历匹配成功:', fn, '| 原始查找:', cleanFileName);
                targetItem.scrollIntoView({ behavior: 'smooth', block: 'center' });
                targetItem.classList.add('pulse-animation');
                setTimeout(() => targetItem.classList.remove('pulse-animation'), 2000);
                return true;
            }
        }

        retryCount++;
        if (retryCount < maxRetries) {
            // 等待后重试
            setTimeout(tryScroll, 50);
        } else {
            console.log('程序①最终无法找到元素:', cleanFileName);
            console.log('程序①data-filename示例:', Array.from(allItems).slice(0, 3).map(i => i.dataset.filename));
        }
        return false;
    }

    // 首次尝试延迟多一点，给 renderTree 完成
    setTimeout(tryScroll, 200);

    return true;
};

function isTargetJson(fileOrPath) {
    if (!fileOrPath) return false;
    if (typeof fileOrPath === 'string') {
        const name = fileOrPath.split('/').pop();
        return name === JSON_FILENAME;
    }
    return fileOrPath.name === JSON_FILENAME;
}

let refreshTimer = null;
let isRefreshing = false;

function scheduleRefresh(delayMs = 120, forceReload = true) {
    if (refreshTimer) window.clearTimeout(refreshTimer);
    refreshTimer = window.setTimeout(() => {
        refreshTree(forceReload);
    }, delayMs);
}

async function refreshTree(forceReload = false) {
    if (isRefreshing) return;
    isRefreshing = true;

    try {
        const ok = await reloadPagesData(forceReload);
        if (!ok) return;

        precomputeParentNodes();

        // 保存缓存
        saveToCache({
            pages: pages,
            parentHierarchyCodes: parentHierarchyCodes,
            maxExistingLevel: maxExistingLevel
        });

        // 防止数据变化后层级状态越界
        viewState.maxLevel = Math.max(1, Math.min(viewState.maxLevel, Math.max(maxExistingLevel, 1)));
        if (viewState.highlightLevel > viewState.maxLevel) {
            viewState.highlightLevel = viewState.maxLevel;
        }

        renderTree();
    } finally {
        isRefreshing = false;
    }
}

function setupJsonAutoRefresh() {
    // 先清理旧监听，避免 Dataview 反复执行导致重复触发
    if (window.myTreeViewVaultRefs_EVA) {
        window.myTreeViewVaultRefs_EVA.forEach(ref => app.vault.offref(ref));
    }

    const vaultRefs = [];
    vaultRefs.push(app.vault.on("modify", (file) => {
        if (isTargetJson(file)) scheduleRefresh(120, true);
    }));
    vaultRefs.push(app.vault.on("create", (file) => {
        if (isTargetJson(file)) scheduleRefresh(120, true);
    }));
    vaultRefs.push(app.vault.on("delete", (file) => {
        if (isTargetJson(file)) scheduleRefresh(120, true);
    }));
    vaultRefs.push(app.vault.on("rename", (file, oldPath) => {
        if (isTargetJson(file) || isTargetJson(oldPath)) scheduleRefresh(120, true);
    }));

    window.myTreeViewVaultRefs_EVA = vaultRefs;
}

async function initializeView() {
    await refreshTree(true);
    setupJsonAutoRefresh();
}

await initializeView();
