function ensureUiStateSets() {
    if (!(state.collapsedIds instanceof Set)) state.collapsedIds = new Set();
    if (!(state.notesOpenIds instanceof Set)) state.notesOpenIds = new Set();
}

function buildCapsuleExpandedMap() {
    ensureUiStateSets();
    const capsuleExpanded = {};
    Object.keys(state.keywords).forEach(id => {
        capsuleExpanded[id] = !state.collapsedIds.has(id);
    });
    return capsuleExpanded;
}

function applyCapsuleExpandedMap(capsuleExpanded) {
    ensureUiStateSets();
    state.collapsedIds.clear();
    if (!capsuleExpanded || typeof capsuleExpanded !== 'object') return;

    Object.entries(capsuleExpanded).forEach(([id, expanded]) => {
        if (state.keywords[id] && expanded === false) {
            state.collapsedIds.add(id);
        }
    });
}

function pruneUiStateByKeywords() {
    ensureUiStateSets();
    const validIds = new Set(Object.keys(state.keywords));
    state.collapsedIds = new Set([...state.collapsedIds].filter(id => validIds.has(id)));
    state.notesOpenIds = new Set([...state.notesOpenIds].filter(id => validIds.has(id)));
}

function persistUiState() {
    return saveIndex().catch(e => {
        console.error('保存UI状态失败:', e);
    });
}

async function loadIndex() {
    try {
        if (await app.vault.adapter.exists(INDEX_PATH)) {
            const content = await app.vault.adapter.read(INDEX_PATH);
            const json = JSON.parse(content);
            state.keywords = json.keywords || {};
            applyCapsuleExpandedMap(json.capsuleExpanded);
            pruneUiStateByKeywords();
            return state.keywords;
        }
    } catch (e) {
        console.error('加载索引失败:', e);
    }
    state.keywords = {};
    ensureUiStateSets();
    state.collapsedIds.clear();
    state.notesOpenIds.clear();
    return {};
}

async function saveIndex() {
    const json = {
        keywords: state.keywords,
        capsuleExpanded: buildCapsuleExpandedMap(), // 扁平字典：{ [keywordId]: true/false }
        version: "3.9",
        lastModified: moment().format("YYYY-MM-DD HH:mm:ss"),
        metadata: {
            total: Object.keys(state.keywords).length,
            topLevel: Object.values(state.keywords).filter(k => !k.parentId).length
        }
    };
    const content = JSON.stringify(json, null, 2);

    const dir = INDEX_PATH.substring(0, INDEX_PATH.lastIndexOf('/'));
    if (dir && !(await app.vault.adapter.exists(dir))) {
        await app.vault.adapter.mkdir(dir);
    }

    if (await app.vault.adapter.exists(INDEX_PATH)) {
        const file = app.vault.getAbstractFileByPath(INDEX_PATH);
        await app.vault.modify(file, content);
    } else {
        await app.vault.create(INDEX_PATH, content);
    }
}

// ============ 关键词提取逻辑 ============

/**
 * 解析关键词字符串，提取基础关键词和后缀
 * 例如：'MMKV-A上项+新开始' → { base: 'MMKV-A上项', suffix: '+新开始' }
 * 例如：'科技委员会' → { base: '科技委员会', suffix: '' }
 */
function parseKeywordWithSuffix(kwString) {
    const plusIndex = kwString.indexOf('+');
    if (plusIndex > 0) {
        return {
            base: kwString.substring(0, plusIndex),
            suffix: kwString.substring(plusIndex)  // 包含+号
        };
    }
    return { base: kwString, suffix: '' };
}

/**
 * 生成关键词ID（基于基础关键词的hash）
 */
function generateKeywordId(baseKeyword) {
    let hash = 0;
    for (let i = 0; i < baseKeyword.length; i++) {
        hash = ((hash << 5) - hash) + baseKeyword.charCodeAt(i);
        hash = hash & hash;
    }
    return 'kw-' + Math.abs(hash).toString(36);
}

/**
 * 获取排序字母
 */
function getSortLetter(keyword) {
    const cleaned = keyword.replace(/^[_\.+\s]+/, '');
    const firstChar = cleaned.charAt(0);
    return (firstChar && /[A-Za-z]/.test(firstChar)) ? firstChar.toUpperCase() : '#';
}

/**
 * 获取显示文本
 * 规则：如果关键词以大写字母开头，只移除第一个字母（用于alphabetical索引定位）
 * 例如：'EEVA系统设计' → 'EVA系统设计'（移除第一个E）
 *      'AATEC7' → 'ATEC7'（移除第一个A）
 *      'C超越性' → '超越性'（移除C）
 *      '科技委员会' → '科技委员会'（不以大写字母开头，保持不变）
 */
function getDisplayText(keyword) {
    // 检查第一个字符是否为大写字母A-Z
    if (keyword.length > 1 && /^[A-Z]/.test(keyword)) {
        return keyword.substring(1);  // 移除第一个字符
    }
    return keyword;
}

function compareKeywordName(a, b) {
    const aName = a.displayText || a.baseKeyword || '';
    const bName = b.displayText || b.baseKeyword || '';
    return aName.localeCompare(bName, 'zh-CN', { numeric: true, sensitivity: 'base' });
}

function compareNoteName(a, b) {
    const aName = a.name || '';
    const bName = b.name || '';
    return aName.localeCompare(bName, 'zh-CN', { numeric: true, sensitivity: 'base' });
}

function normalizeKeywordListValue(raw) {
    if (Array.isArray(raw)) {
        return raw
            .map(v => String(v || '').trim())
            .filter(Boolean);
    }
    if (typeof raw === 'string') {
        const text = raw.trim();
        return text ? [text] : [];
    }
    return [];
}

function toYamlSafeText(value) {
    const text = String(value || '').trim();
    if (!text) return "''";
    if (/^[\u4e00-\u9fa5A-Za-z0-9_\-+.·]+$/.test(text)) return text;
    return `'${text.replace(/'/g, "''")}'`;
}

function buildKeywordYamlBlock(values) {
    const uniq = [];
    const seen = new Set();
    for (const raw of values || []) {
        const text = String(raw || '').trim();
        if (!text || seen.has(text)) continue;
        seen.add(text);
        uniq.push(text);
    }
    if (uniq.length === 0) return '关键词管理: []';
    return ['关键词管理:', ...uniq.map(v => `  - ${toYamlSafeText(v)}`)].join('\n');
}

function replaceOrAppendKeywordYamlField(yamlBody, keywordValues) {
    const lines = String(yamlBody || '').split('\n');
    const start = lines.findIndex(line => /^\s*关键词管理\s*:/.test(line));
    const replacement = buildKeywordYamlBlock(keywordValues).split('\n');

    if (start >= 0) {
        let end = start + 1;
        while (end < lines.length) {
            const line = lines[end];
            if (/^\s*$/.test(line)) { end += 1; continue; }
            if (/^[^\s#][^:]*\s*:/.test(line)) break;
            if (/^\s/.test(line)) { end += 1; continue; }
            break;
        }
        lines.splice(start, end - start, ...replacement);
        return lines.join('\n');
    }

    const trimmed = String(yamlBody || '').trim();
    return trimmed ? `${yamlBody}\n${buildKeywordYamlBlock(keywordValues)}` : buildKeywordYamlBlock(keywordValues);
}

function normalizeNoteIndexItems(items) {
    return (items || []).map((item) => {
        const path = String(item?.path || '').trim();
        const name = String(item?.name || item?.basename || path.split('/').pop() || '').replace(/\.md$/i, '');
        return {
            path,
            name,
            nameLower: name.toLowerCase(),
            pathLower: path.toLowerCase(),
            search: String(item?.search || `${name.toLowerCase()} ${path.toLowerCase()}`).trim()
        };
    }).filter((item) => item.path);
}

function buildMarkdownNoteIndex() {
    const uniq = new Map();
    Object.values(state.keywords).forEach((kw) => {
        (kw.notes || []).forEach((note) => {
            if (!note?.path || uniq.has(note.path)) return;
            const derivedName = note.name || note.path.replace(/^.*\//, '').replace(/\.md$/i, '');
            uniq.set(note.path, {
                path: note.path,
                name: derivedName
            });
        });
    });
    const active = app.workspace.getActiveFile();
    if (active?.path && !uniq.has(active.path)) {
        uniq.set(active.path, { path: active.path, name: active.basename });
    }
    return normalizeNoteIndexItems(Array.from(uniq.values()))
        .sort((a, b) => compareNoteName(a, b));
}

function buildVaultIndexFromFiles(files) {
    return normalizeNoteIndexItems((files || []).map((f) => ({
        path: f.path,
        name: f.basename,
        search: `${String(f.basename || '').toLowerCase()} ${String(f.path || '').toLowerCase()}`
    }))).sort((a, b) => compareNoteName(a, b));
}

function buildVaultIndexFromEvaNotes(evaNotes) {
    const items = [];
    Object.values(evaNotes || {}).forEach((note) => {
        const path = typeof note?.file_path === 'string' ? note.file_path : null;
        if (!path) return;
        const fileName = typeof note?.file_name === 'string' ? note.file_name : path.split('/').pop();
        const name = (fileName || path).replace(/\.md$/i, '');
        const title = typeof note?.title === 'string' ? note.title : '';
        items.push({
            path,
            name,
            search: `${name.toLowerCase()} ${path.toLowerCase()} ${title.toLowerCase()}`.trim()
        });
    });
    return normalizeNoteIndexItems(items).sort((a, b) => compareNoteName(a, b));
}

async function loadEvaNoteIndex() {
    const candidates = [resolveLocalPath('EVA_Notes.json')].concat(EVA_NOTES_CANDIDATE_PATHS);
    const seen = new Set();
    for (const path of candidates) {
        const normalized = String(path || '').trim();
        if (!normalized || seen.has(normalized)) continue;
        seen.add(normalized);
        try {
            if (!await app.vault.adapter.exists(normalized)) continue;
            const content = await app.vault.adapter.read(normalized);
            const parsed = JSON.parse(content);
            if (parsed && parsed.notes && typeof parsed.notes === 'object') {
                const items = buildVaultIndexFromEvaNotes(parsed.notes);
                if (items.length > 0) {
                    const aliasMap = new Map();
                    for (const item of items) {
                        const full = String(item.path || '').trim().replace(/^\/+/, '');
                        const noExt = full.replace(/\.md$/i, '');
                        const base = noExt.split('/').pop() || noExt;
                        if (!full) continue;
                        aliasMap.set(full.toLowerCase(), full);
                        aliasMap.set(noExt.toLowerCase(), full);
                        aliasMap.set(base.toLowerCase(), full);
                    }
                    window._kwIndexEvaNoteAliasCache = {
                        builtAt: Date.now(),
                        sourcePath: normalized,
                        aliasMap
                    };
                    return { path: normalized, items };
                }
            }
        } catch (e) {
            console.error('读取 EVA_Notes.json 失败:', normalized, e);
        }
    }
    return null;
}

async function getVaultNoteIndex() {
    if (!window._kwIndexVaultNoteIndexCache) {
        window._kwIndexVaultNoteIndexCache = { items: [], builtAt: 0, source: 'none', sourcePath: null };
    }
    const cache = window._kwIndexVaultNoteIndexCache;
    const now = Date.now();
    if (cache.items.length > 0 && (now - cache.builtAt) <= 30000) return cache.items;

    const eva = await loadEvaNoteIndex();
    if (eva && eva.items.length > 0) {
        cache.items = eva.items;
        cache.builtAt = now;
        cache.source = 'eva';
        cache.sourcePath = eva.path;
        return cache.items;
    }

    const markdownFiles = app.vault.getMarkdownFiles();
    cache.items = buildVaultIndexFromFiles(markdownFiles);
    cache.builtAt = now;
    cache.source = 'vault';
    cache.sourcePath = null;
    return cache.items;
}

function sanitizeNoteInput(raw) {
    const text = String(raw || '').trim();
    if (!text) return '';
    return text
        .replace(/^\[\[/, '')
        .replace(/\]\]$/, '')
        .split(/[｜|]/)[0]
        .split('#')[0]
        .trim();
}

function resolveExistingNoteFile(rawPath) {
    const clean = sanitizeNoteInput(rawPath);
    if (!clean) return null;

    const normalizedClean = clean.replace(/^\/+/, '');
    const noExt = normalizedClean.replace(/\.md$/i, '');
    const withExt = normalizedClean.endsWith('.md') ? normalizedClean : `${normalizedClean}.md`;
    const lowerClean = normalizedClean.toLowerCase();
    const lowerNoExt = noExt.toLowerCase();
    const lowerWithExt = withExt.toLowerCase();

    if (!window._kwIndexResolvedNoteFileCache || (Date.now() - window._kwIndexResolvedNoteFileCache.builtAt) > 30000) {
        window._kwIndexResolvedNoteFileCache = { builtAt: Date.now(), map: new Map() };
    }
    const resolvedCache = window._kwIndexResolvedNoteFileCache.map;
    const cacheKey = lowerWithExt || lowerClean;
    if (resolvedCache.has(cacheKey)) return resolvedCache.get(cacheKey);

    const aliasMap = window._kwIndexEvaNoteAliasCache?.aliasMap || new Map();
    const aliasedPath = aliasMap.get(lowerClean)
        || aliasMap.get(lowerWithExt)
        || aliasMap.get(lowerNoExt)
        || aliasMap.get(lowerNoExt.split('/').pop())
        || normalizedClean;
    const aliasedNoExt = aliasedPath.replace(/\.md$/i, '');

    const hit = app.vault.getAbstractFileByPath(normalizedClean)
        || app.vault.getAbstractFileByPath(withExt)
        || app.vault.getAbstractFileByPath(aliasedPath)
        || app.metadataCache.getFirstLinkpathDest?.(normalizedClean, HOST_NOTE_PATH || '')
        || app.metadataCache.getFirstLinkpathDest?.(noExt, HOST_NOTE_PATH || '')
        || app.metadataCache.getFirstLinkpathDest?.(noExt.split('/').pop() || noExt, HOST_NOTE_PATH || '')
        || app.metadataCache.getFirstLinkpathDest?.(aliasedNoExt, HOST_NOTE_PATH || '')
        || null;
    resolvedCache.set(cacheKey, hit || null);
    return hit;
}

function resolveNotePathFromInput(rawInput, noteIndex) {
    const raw = String(rawInput || '').trim();
    if (!raw) return null;

    const pathMatch = raw.match(/(?:｜|\|)\s*(.+?)\s*$/);
    if (pathMatch && pathMatch[1]) {
        const rightPart = String(pathMatch[1] || '').trim();
        const fromRight = resolveExistingNoteFile(rightPart);
        if (fromRight?.path) return fromRight.path;
    }

    const direct = resolveExistingNoteFile(raw);
    if (direct?.path) return direct.path;

    const index = normalizeNoteIndexItems(noteIndex || []);
    const normalized = sanitizeNoteInput(raw).replace(/\.md$/i, '');
    const lower = normalized.toLowerCase();
    const byName = index.filter(n => n.nameLower === lower);
    if (byName.length === 1) return byName[0].path;

    const byContains = index.filter(n =>
        n.nameLower.includes(lower) ||
        n.pathLower.includes(lower)
    );
    if (byContains.length === 1) return byContains[0].path;

    return null;
}

function resolveParentKeywordIdFromInput(rawInput) {
    const raw = String(rawInput || '').trim();
    if (!raw) return { ok: true, id: null };

    const idMatch = raw.match(/(?:｜|\|)\s*([A-Za-z0-9_\-]+)\s*$/);
    if (idMatch) {
        const id = idMatch[1];
        if (id === '__top__') return { ok: true, id: null };
        if (state.keywords[id]) return { ok: true, id };
    }

    if (raw === '-- 顶层关键词 --') return { ok: true, id: null };
    if (state.keywords[raw]) return { ok: true, id: raw };

    const lower = raw.toLowerCase();
    const exact = Object.values(state.keywords).filter(kw =>
        String(kw.displayText || '').toLowerCase() === lower ||
        String(kw.baseKeyword || '').toLowerCase() === lower
    );
    if (exact.length === 1) return { ok: true, id: exact[0].id };

    return { ok: false, id: null };
}

function parseYamlScalarText(raw) {
    const text = String(raw || '').trim();
    if (!text) return '';
    if (text.startsWith("'") && text.endsWith("'")) {
        return text.substring(1, text.length - 1).replace(/''/g, "'").trim();
    }
    if (text.startsWith('"') && text.endsWith('"')) {
        return text.substring(1, text.length - 1).replace(/\\"/g, '"').trim();
    }
    return text;
}

function extractKeywordValuesFromYamlBody(yamlBody) {
    const lines = String(yamlBody || '').split('\n');
    const start = lines.findIndex(line => /^\s*关键词管理\s*:/.test(line));
    if (start < 0) return [];

    const inline = lines[start].replace(/^\s*关键词管理\s*:\s*/, '').trim();
    if (inline) {
        if (inline === '[]') return [];
        if (inline.startsWith('[') && inline.endsWith(']')) {
            return inline
                .slice(1, -1)
                .split(',')
                .map(v => parseYamlScalarText(v))
                .filter(Boolean);
        }
        const value = parseYamlScalarText(inline);
        return value ? [value] : [];
    }

    const values = [];
    for (let i = start + 1; i < lines.length; i++) {
        const line = lines[i];
        if (/^\s*$/.test(line)) continue;
        if (/^[^\s#][^:]*\s*:/.test(line)) break;
        const listItem = line.match(/^\s*-\s*(.+?)\s*$/);
        if (listItem) {
            const value = parseYamlScalarText(listItem[1]);
            if (value) values.push(value);
            continue;
        }
        if (!/^\s/.test(line)) break;
    }
    return values;
}

async function mountKeywordToNoteYaml(notePath, keywordRaw) {
    const keywordText = String(keywordRaw || '').trim();
    if (!keywordText) throw new Error('关键词不能为空');

    const file = app.vault.getAbstractFileByPath(notePath);
    if (!file || !file.path) throw new Error(`未找到笔记: ${notePath}`);

    const content = await app.vault.read(file);
    const yamlMatch = content.match(/^---\r?\n([\s\S]*?)\r?\n---(\r?\n)?/);

    let existingValues = [];
    let newContent = content;
    let hadKeywordField = false;

    if (yamlMatch) {
        const yamlBody = yamlMatch[1];
        existingValues = extractKeywordValuesFromYamlBody(yamlBody);
        hadKeywordField = /^\s*关键词管理\s*:/.test(yamlBody);
        const nextValues = [...existingValues];
        if (!nextValues.includes(keywordText)) nextValues.push(keywordText);
        const newYamlBody = replaceOrAppendKeywordYamlField(yamlBody, nextValues);
        const suffix = content.slice(yamlMatch[0].length);
        newContent = `---\n${newYamlBody}\n---\n${suffix}`;
    } else {
        newContent = `---\n${buildKeywordYamlBlock([keywordText])}\n---\n${content}`;
    }

    if (newContent !== content) {
        await app.vault.modify(file, newContent);
    }

    const wasAdded = !existingValues.includes(keywordText) || !hadKeywordField || !yamlMatch;
    return { changed: newContent !== content, wasAdded };
}

/**
 * 计算sortKey（基于关联笔记的最小ATOM编码）
 */
function computeSortKey(notes) {
    const codes = [];
    for (const note of notes) {
        const m = note.path.match(/ATOM@([\d.]+)/);
        if (m) codes.push(m[1]);
    }
    if (codes.length === 0) return 'ZZZZZZ';

    codes.sort((a, b) => {
        const pa = a.split('.').map(Number);
        const pb = b.split('.').map(Number);
        for (let i = 0; i < Math.max(pa.length, pb.length); i++) {
            const diff = (pa[i] || 0) - (pb[i] || 0);
            if (diff !== 0) return diff;
        }
        return 0;
    });

    return codes[0].split('.').map(p => String(p).padStart(3, '0')).join('.');
}

// ============ 全量扫描 ============

async function fullScan() {
    // 临时map：baseKeyword -> { notes: [], type }
    const keywordMap = new Map();
    const files = app.vault.getMarkdownFiles();

    for (const file of files) {
        if (SCAN_FOLDER && !file.path.startsWith(SCAN_FOLDER)) continue;

        try {
            // 1. 从yaml提取 - 注意字段名是中文"关键词管理"
            const cache = app.metadataCache.getFileCache(file);
            const yamlKws = cache?.frontmatter?.['关键词管理'] || [];
            if (Array.isArray(yamlKws)) {
                yamlKws.forEach(kw => {
                    if (typeof kw === 'string' && kw.trim()) {
                        const { base, suffix } = parseKeywordWithSuffix(kw.trim());
                        addToKeywordMap(keywordMap, base, file, suffix, 'yaml');
                    }
                });
            }

            // 2. 从文件名提取 ++...++
            // 注意：内容中可能包含单个+号作为后缀分隔符，所以不能用[^+]+
            // 使用非贪婪匹配，匹配到下一个++为止
            const fileName = file.basename;
            const plusRegex = /\+\+(.+?)\+\+/g;
            let match;
            while ((match = plusRegex.exec(fileName)) !== null) {
                const { base, suffix } = parseKeywordWithSuffix(match[1].trim());
                addToKeywordMap(keywordMap, base, file, suffix, 'plus');
            }

            // 3. 从文件名提取 ··...··
            const dotRegex = /··([^·]+)··/g;
            while ((match = dotRegex.exec(fileName)) !== null) {
                const { base, suffix } = parseKeywordWithSuffix(match[1].trim());
                addToKeywordMap(keywordMap, base, file, suffix, 'dot');
            }

        } catch (e) {
            // 跳过
        }
    }

    // 保留现有的parentId和order
    const oldKeywords = { ...state.keywords };

    // 转换为state.keywords格式
    state.keywords = {};
    for (const [baseKeyword, info] of keywordMap.entries()) {
        const id = generateKeywordId(baseKeyword);
        const oldData = oldKeywords[id] || {};

        state.keywords[id] = {
            id: id,
            baseKeyword: baseKeyword,
            displayText: getDisplayText(baseKeyword),
            sortLetter: getSortLetter(baseKeyword),
            type: info.type,
            notes: Array.from(info.notes.values()),
            sortKey: computeSortKey(Array.from(info.notes.values())),
            // 保留用户设置的层级关系
            parentId: oldData.parentId || null,
            order: oldData.order ?? 0
        };
    }
    pruneUiStateByKeywords();

    await saveIndex();
    return state.keywords;
}

function addToKeywordMap(map, baseKeyword, file, suffix, type) {
    if (!map.has(baseKeyword)) {
        map.set(baseKeyword, { notes: [], type: type });
    }
    const entry = map.get(baseKeyword);

    // 允许同一文件/同一来源的多个后缀并存，用数组存储
    entry.notes.push({
        path: file.path,
        name: file.basename,
        suffix: suffix,
        sourceType: type
    });

    // 类型优先级: yaml > plus > dot（用于关键词整体默认显示类型）
    const priority = { dot: 1, plus: 2, yaml: 3 };
    if (priority[type] > priority[entry.type]) {
        entry.type = type;
    }
}

// ============ 反向同步功能 ============

/**
 * 反向同步：更新原始笔记文件中的关键词
 * @param {string} oldKeyword - 原关键词（基础部分）
 * @param {string} newKeyword - 新关键词
 * @param {Array} notes - 关联的笔记列表 [{path, name, suffix}]
 * @param {string} type - 关键词类型 'yaml' | 'plus' | 'dot'
 */
async function syncKeywordToNotes(oldKeyword, newKeyword, notes, type) {
    if (oldKeyword === newKeyword) return { updated: 0, failed: 0 };

    let updated = 0;
    let failed = 0;

    for (const note of notes) {
        try {
            const file = app.vault.getAbstractFileByPath(note.path);
            if (!file) continue;

            let content = await app.vault.read(file);
            let modified = false;

            // 1. 更新YAML中的关键词管理字段
            if (type === 'yaml') {
                // 匹配yaml frontmatter中的关键词管理字段
                const yamlRegex = /^---\n([\s\S]*?)\n---/;
                const yamlMatch = content.match(yamlRegex);
                if (yamlMatch) {
                    const yamlContent = yamlMatch[1];
                    // 构建完整的原关键词（包含后缀）
                    const oldFull = note.suffix ? oldKeyword + note.suffix : oldKeyword;
                    const newFull = note.suffix ? newKeyword + note.suffix : newKeyword;

                    // 替换yaml中的关键词
                    const newYaml = yamlContent.replace(
                        new RegExp(`(关键词管理:[\\s\\S]*?)${escapeRegex(oldFull)}`, 'g'),
                        `$1${newFull}`
                    );
                    if (newYaml !== yamlContent) {
                        content = content.replace(yamlMatch[1], newYaml);
                        modified = true;
                    }
                }
            }

            // 2. 更新文件名中的 ++...++ 或 ··...··
            if (type === 'plus' || type === 'dot') {
                const delimiter = type === 'plus' ? '++' : '··';
                const oldFull = note.suffix ? oldKeyword + note.suffix : oldKeyword;
                const newFull = note.suffix ? newKeyword + note.suffix : newKeyword;
                const oldPattern = `${delimiter}${oldFull}${delimiter}`;
                const newPattern = `${delimiter}${newFull}${delimiter}`;

                // 检查文件名是否包含该关键词
                if (note.name.includes(oldPattern)) {
                    const newName = note.name.replace(oldPattern, newPattern);
                    const newPath = note.path.replace(note.name, newName);

                    // 重命名文件
                    await app.fileManager.renameFile(file, newPath);
                    modified = true;
                }
            }

            // 如果内容有修改（yaml情况），保存文件
            if (modified && type === 'yaml') {
                await app.vault.modify(file, content);
            }

            if (modified) updated++;
        } catch (e) {
            console.error(`更新笔记失败: ${note.path}`, e);
            failed++;
        }
    }

    return { updated, failed };
}

    return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}

/**
 * 按+号分割suffix，构建笔记的层级树结构
 * @param {Array} notes - 笔记数组 [{path, name, suffix, sourceType}, ...]
 * @returns {Object} 树结构 { "第一层": { children: { "第二层": { children: { ... }, notes: [...] } }, notes: [...] } }
 */
function buildSuffixTree(notes) {
    const root = { children: new Map(), notes: [] };

    notes.forEach(note => {
        const suffix = note.suffix || '';
        // 分割suffix，移除每个部分前面的+号
        const parts = suffix.split('+').filter(p => p.trim());

        if (parts.length === 0) {
            // 无suffix，直接放在根节点
            root.notes.push(note);
        } else {
            let current = root;
            // 遍历每一层
            parts.forEach((part, idx) => {
                const trimmedPart = part.trim();
                if (!trimmedPart) return;

                if (!current.children.has(trimmedPart)) {
                    current.children.set(trimmedPart, { children: new Map(), notes: [] });
                }
                current = current.children.get(trimmedPart);

                // 最后一层放笔记
                if (idx === parts.length - 1) {
                    current.notes.push(note);
                }
            });
        }
    });

    return root;
}

/**
 * 统计节点的所有后代笔记数量（包括自身）
 */
function countAllNotes(node) {
    let count = node.notes.length;
    node.children.forEach(child => {
        count += countAllNotes(child);
    });
    return count;
}

/**
 * 收集节点的所有笔记（扁平化，包括自身和所有后代）
 */
function collectAllNotes(node) {
    let allNotes = [...node.notes];
    node.children.forEach(child => {
        allNotes = allNotes.concat(collectAllNotes(child));
    });
    return allNotes;
}

                        app.workspace.openLinkText(note.path, '');
                    };
                });
            }

            // 递归渲染子内容
            renderDrawerTree(childContainer, childNode, level + 1, type, expandNotesByDefault);
        } else {
            // === 最深层级的节点（无子层级）：显示层级名 + 可折叠的"📝 根笔记"按钮 ===
            const leafLabel = container.createEl('div', { cls: 'kw-drawer-parent' });
            leafLabel.style.marginLeft = `${level * 20}px`;

            const toggle = leafLabel.createEl('span', { cls: 'kw-drawer-toggle', text: '▼' });
            const label = leafLabel.createEl('span', { cls: 'kw-drawer-label', text: childName });
            const count = leafLabel.createEl('span', { cls: 'kw-drawer-count', text: `(${childNode.notes.length})` });

            // 笔记容器（默认折叠）
            const notesContainer = leafLabel.createEl('div', { cls: 'kw-drawer-children' });
            notesContainer.style.display = 'none';

            const sortedNotes = [...childNode.notes].sort(compareNoteName);
            // 渲染该叶子节点的笔记
            sortedNotes.forEach(note => {
                const noteItem = notesContainer.createEl('div', { cls: 'kw-drawer-note' });
                noteItem.style.marginLeft = '20px';
                noteItem.createEl('span', { text: '📄' });
                noteItem.createEl('span', { text: note.name });
                noteItem.onclick = (e) => {
                    e.stopPropagation();
                    app.workspace.openLinkText(note.path, '');
                };
            });

            // 点击事件：切换笔记容器显示/隐藏
            let isNotesExpanded = expandNotesByDefault;
            toggle.style.transform = isNotesExpanded ? 'rotate(90deg)' : '';
            leafLabel.onclick = (e) => {
                e.stopPropagation();
                isNotesExpanded = !isNotesExpanded;
                toggle.style.transform = isNotesExpanded ? 'rotate(90deg)' : '';
                notesContainer.style.display = isNotesExpanded ? 'block' : 'none';
            };

            container.appendChild(leafLabel);
        }
    });

    // --- 渲染当前层级的笔记（直属）- 仅针对 Level 0 ---
    // v3.7: Level 0 的笔记在所有子层级渲染完后显示，可折叠"📝 根笔记"
    if (node.notes.length > 0 && level === 0) {
        const noteWrapper = container.createEl('div', { cls: 'kw-drawer-parent' });
        noteWrapper.style.marginLeft = `${level * 20}px`;

        const noteToggle = noteWrapper.createEl('span', { cls: 'kw-drawer-toggle', text: '▼' });
        noteWrapper.createEl('span', { cls: 'kw-drawer-label', text: '📝 根笔记' });
        noteWrapper.createEl('span', { cls: 'kw-drawer-count', text: `(${node.notes.length})` });

        // 笔记容器（默认折叠）
        const notesContainer = noteWrapper.createEl('div', { cls: 'kw-drawer-children' });
        notesContainer.style.display = 'none';

        // 绑定点击事件
        let isExpanded = expandNotesByDefault;
        noteToggle.style.transform = isExpanded ? 'rotate(90deg)' : '';

        noteWrapper.onclick = (e) => {
            e.stopPropagation();
            isExpanded = !isExpanded;
            noteToggle.style.transform = isExpanded ? 'rotate(90deg)' : '';
            notesContainer.style.display = isExpanded ? 'block' : 'none';
        };

        const sortedNotes = [...node.notes].sort(compareNoteName);
        // 渲染笔记项
        sortedNotes.forEach(note => {
            const noteItem = notesContainer.createEl('div', { cls: 'kw-drawer-note' });
            noteItem.dataset.level = level;
            noteItem.style.marginLeft = '20px';
            noteItem.createEl('span', { text: '📄' });
            noteItem.createEl('span', { text: note.name });
            noteItem.onclick = (e) => {
                e.stopPropagation();
                app.workspace.openLinkText(note.path, '');
            };
        });
    }
}

/**
 * 统计节点的所有后代笔记数量
 */
function countDescendants(node) {
    let count = node.notes.length;
    node.children.forEach(child => {
        count += countDescendants(child);
    });
    return count;
}

/**
 * 渲染多层级抽屉（按来源类型分组）
 * @param {HTMLElement} drawer - 抽屉容器
 * @param {Array} notes - 该类型的笔记数组
 * @param {string} type - 来源类型
 * @param {boolean} showGroupHeader - 是否显示分组标题
 * @param {boolean} expandNotesByDefault - 笔记默认是否展开（v3.7新增，默认折叠）
 */
function renderMultiLevelDrawer(drawer, notes, type, showGroupHeader = false, expandNotesByDefault = false) {
    if (notes.length === 0) return;

    // v3.4.1: 所有笔记统一使用多层级渲染
    // 构建层级树
    const tree = buildSuffixTree(notes);

    // v3.7: 传递 expandNotesByDefault 参数（默认折叠）
    renderDrawerTree(drawer, tree, 0, type, expandNotesByDefault);
}

/**
 * 渲染单个关键词胶囊（套圈式，默认展开）
 */
function renderCapsule(container, kw) {
    if (!kw || !kw.type) {
        console.error('无效的关键词对象:', kw);
        return;
    }

    const children = getChildKeywords(kw.id);
    const hasChildren = children.length > 0;
    ensureUiStateSets();
    const isCollapsed = state.collapsedIds.has(kw.id);
    const isNotesOpen = state.notesOpenIds?.has?.(kw.id) ?? false;

    // 检查是否有多种来源类型
    const sourceTypes = getKeywordSourceTypes(kw);
    const isMixedType = sourceTypes.size > 1;

    const capsule = container.createEl('div', {
        cls: `kw-capsule ${hasChildren ? 'has-children' : ''} ${isCollapsed ? 'collapsed' : 'expanded'} ${isNotesOpen ? 'notes-open' : ''}`
    });

    // 头部 - 混合类型使用特殊样式
    const headerCls = isMixedType ? 'kw-capsule-header type-mixed' : `kw-capsule-header type-${kw.type}`;
    const header = capsule.createEl('div', { cls: headerCls });

    if (hasChildren) {
        const toggle = header.createEl('span', {
            cls: 'kw-capsule-toggle',
            text: '▶'
        });
        toggle.onclick = (e) => {
            e.stopPropagation();
            ensureUiStateSets();
            if (state.collapsedIds.has(kw.id)) {
                state.collapsedIds.delete(kw.id);
            } else {
                state.collapsedIds.add(kw.id);
            }
            renderViewFn();
            persistUiState();
        };
    }

    // 图标 - 混合类型显示多个图标
    if (isMixedType) {
        const typeOrder = ['yaml', 'plus', 'dot'];
        const icons = typeOrder.filter(t => sourceTypes.has(t)).map(t => COLORS[t].icon);
        header.createEl('span', { cls: 'kw-capsule-icon kw-capsule-icon-mixed', text: icons.join('') });
    } else {
        header.createEl('span', { cls: 'kw-capsule-icon', text: COLORS[kw.type].icon });
    }

    header.createEl('span', { cls: 'kw-capsule-label', text: kw.displayText });

    if (kw.notes.length > 0) {
        header.createEl('span', { cls: 'kw-capsule-badge', text: `${kw.notes.length}` });
    }

    // 点击事件 - 所有胶囊统一触发交互
    header.onclick = (e) => {
        e.stopPropagation();
        handleCapsuleClick(kw);
    };

    // 统一的点击处理函数
    function handleCapsuleClick(keyword) {
        ensureUiStateSets();

        if (keyword.notes.length === 1) {
            // 只有一个笔记，直接打开
            app.workspace.openLinkText(keyword.notes[0].path, '');
        } else if (keyword.notes.length > 1) {
            // 多个笔记，切换笔记抽屉
            if (state.notesOpenIds.has(keyword.id)) {
                state.notesOpenIds.delete(keyword.id);
            } else {
                state.notesOpenIds.add(keyword.id);
            }
            renderViewFn();
        }
        // 子节点折叠/展开由左侧三角开关负责
    }

    // 笔记抽屉（v3.4: 多层级按+号分割展示）
    // 放在子关键词容器之前，确保展开后直接显示在当前胶囊下方
    if (kw.notes.length > 0) {
        const drawer = capsule.createEl('div', {
            cls: 'kw-notes-drawer',
            style: isNotesOpen ? '' : 'display: none;' // 默认隐藏笔记抽屉
        });

        // 按来源类型分组笔记
        const typeOrder = ['yaml', 'plus', 'dot'];
        const typeNames = {
            yaml: '◆ YAML关键词管理',
            plus: '▲ 层级关键词 ++...++',
            dot: '● 行内关键词 ··...··'
        };
        const groupedNotes = { yaml: [], plus: [], dot: [] };

        kw.notes.forEach(note => {
            const sourceType = note.sourceType || kw.type; // 兼容旧数据
            if (groupedNotes[sourceType]) {
                groupedNotes[sourceType].push(note);
            }
        });

        // 检查是否有多种来源类型
        const activeTypes = typeOrder.filter(t => groupedNotes[t].length > 0);
        const hasMultipleTypes = activeTypes.length > 1;

        // 按类型顺序渲染
        typeOrder.forEach(type => {
            const notes = groupedNotes[type];
            if (notes.length === 0) return;

            // 如果有多种类型，显示分组标题
            if (hasMultipleTypes) {
                const groupHeader = drawer.createEl('div', { cls: `kw-drawer-group-header type-${type}` });
                groupHeader.createEl('span', { text: COLORS[type].icon });
                groupHeader.createEl('span', { text: typeNames[type] });
                groupHeader.createEl('span', { cls: 'kw-drawer-group-count', text: `${notes.length}` });
            }

            // 使用多层级渲染函数
            // v3.7: 笔记默认折叠，用户点击"📝 根笔记"展开
            renderMultiLevelDrawer(drawer, notes, type, false, false);
        });
    }

    // 子关键词容器（套圈效果，默认展开）
    if (hasChildren) {
        const childrenContainer = capsule.createEl('div', { cls: 'kw-capsule-children' });
        children.forEach(child => {
            renderCapsule(childrenContainer, child);
        });
    }
}

            type: 'text',
            cls: 'kw-parent-input',
            attr: { placeholder: '输入关键词搜索...' }

        const clearBtn = inputWrapper.createEl('span', { cls: 'kw-parent-clear', text: '×' });

        // 下拉列表

                parentInput.value = selectedKw.displayText;

        function highlightMatch(text, query) {
            if (!query) return text;
            const lowerText = text.toLowerCase();
            const lowerQuery = query.toLowerCase();
            const index = lowerText.indexOf(lowerQuery);
            if (index === -1) return text;
            return text.substring(0, index) +
                   '<span class="kw-parent-option-match">' +
                   text.substring(index, index + query.length) +
                   '</span>' +
                   text.substring(index + query.length);
        }

        // 渲染下拉选项
        function renderOptions(query = '') {
            dropdown.innerHTML = '';
            const lowerQuery = query.toLowerCase().trim();

            // 筛选匹配的选项
            const filtered = availableParents.filter(k => {
                if (!lowerQuery) return true;
                return k.displayText.toLowerCase().includes(lowerQuery) ||
                       k.baseKeyword.toLowerCase().includes(lowerQuery);
            });

            // 顶层选项（始终显示）
            const topOption = dropdown.createEl('div', {
                cls: `kw-parent-option ${!selectedParentId ? 'selected' : ''}`,
                attr: { 'data-value': '' }
            });
            topOption.innerHTML = `<span class="kw-parent-option-icon">🏠</span><span class="kw-parent-option-text">-- 顶层关键词 --</span>`;
            topOption.onclick = () => selectOption('', '-- 顶层关键词 --');

            if (filtered.length === 0 && lowerQuery) {
                dropdown.createEl('div', { cls: 'kw-parent-no-results', text: `未找到匹配 "${query}" 的关键词` });
            } else {
                filtered.forEach((k, idx) => {
                    const option = dropdown.createEl('div', {
                        cls: `kw-parent-option ${selectedParentId === k.id ? 'selected' : ''} ${idx === highlightedIndex ? 'highlighted' : ''}`,
                        attr: { 'data-value': k.id }
                    });
                    const highlightedText = highlightMatch(k.displayText, query);
                    option.innerHTML = `<span class="kw-parent-option-icon">${COLORS[k.type].icon}</span><span class="kw-parent-option-text">${highlightedText}</span>`;
                    option.onclick = () => selectOption(k.id, k.displayText);
                });
            }

            // 统计信息
            const statsText = lowerQuery
                ? `显示 ${filtered.length} / ${availableParents.length} 个可选父级`
                : `共 ${availableParents.length} 个可选父级`;
            dropdown.createEl('div', { cls: 'kw-parent-stats', text: statsText });

            highlightedIndex = -1;
        }

        // 选择选项
        function selectOption(id, displayText) {
            selectedParentId = id || null;
            parentInput.value = id ? displayText : '';
            dropdown.classList.remove('visible');
            parentInput.blur();
        }

        // 显示/隐藏下拉
        function showDropdown() {
            renderOptions(parentInput.value);
            dropdown.classList.add('visible');
        }

        function hideDropdown() {
            setTimeout(() => dropdown.classList.remove('visible'), 150);
        }

        // 事件绑定
        parentInput.addEventListener('focus', showDropdown);
        parentInput.addEventListener('blur', hideDropdown);
        parentInput.addEventListener('input', () => {
            renderOptions(parentInput.value);
            dropdown.classList.add('visible');
        });

        // 键盘导航
        parentInput.addEventListener('keydown', (e) => {
            const options = dropdown.querySelectorAll('.kw-parent-option');
            if (e.key === 'ArrowDown') {
                e.preventDefault();
                highlightedIndex = Math.min(highlightedIndex + 1, options.length - 1);
                options.forEach((opt, i) => opt.classList.toggle('highlighted', i === highlightedIndex));
                if (options[highlightedIndex]) options[highlightedIndex].scrollIntoView({ block: 'nearest' });
            } else if (e.key === 'ArrowUp') {
                e.preventDefault();
                highlightedIndex = Math.max(highlightedIndex - 1, 0);
                options.forEach((opt, i) => opt.classList.toggle('highlighted', i === highlightedIndex));
                if (options[highlightedIndex]) options[highlightedIndex].scrollIntoView({ block: 'nearest' });
            } else if (e.key === 'Enter') {
                e.preventDefault();
                if (highlightedIndex >= 0 && options[highlightedIndex]) {
                    options[highlightedIndex].click();
                }
            } else if (e.key === 'Escape') {
                e.stopPropagation();
                hideDropdown();
                parentInput.blur();
            }
        });

        // 清除按钮
        clearBtn.onclick = (e) => {
            e.stopPropagation();
            selectedParentId = null;
            parentInput.value = '';
            parentInput.focus();
        };

        // 用于保存时获取选中的父级ID
        const getSelectedParentId = () => selectedParentId;

        // 按钮
        const btnRow = body.createEl('div', { cls: 'kw-edit-modal-actions' });

        const cancelBtn = btnRow.createEl('button', { text: '取消' });
        cancelBtn.style.cssText = 'padding:6px 14px; border-radius:4px; border:1px solid var(--background-modifier-border); background:var(--background-secondary); cursor:pointer;';
        cancelBtn.onclick = closeEditModal;

        const saveBtn = btnRow.createEl('button', { text: '保存' });
        saveBtn.style.cssText = 'padding:6px 14px; border-radius:4px; border:none; background:var(--interactive-accent); color:var(--text-on-accent); cursor:pointer;';
        saveBtn.onclick = async () => {
            const newLabel = labelInput.value.trim();
            if (!newLabel) { new Notice('⚠️ 名称不能为空'); return; }

            const oldKeyword = kw.baseKeyword;
            const keywordChanged = oldKeyword !== newLabel;
            const oldId = kw.id;
            const newId = generateKeywordId(newLabel);
            if (newId !== oldId && state.keywords[newId]) {
                new Notice('⚠️ 该名称已存在');
                return;
            }

            // 如果关键词名称改变，执行反向同步
            if (keywordChanged && kw.notes.length > 0) {
                saveBtn.disabled = true;
                saveBtn.textContent = '同步中...';

                const result = await syncKeywordToNotes(oldKeyword, newLabel, kw.notes, kw.type);

                if (result.updated > 0) {
                    new Notice(`📝 已更新 ${result.updated} 个笔记文件`);
                }
                if (result.failed > 0) {
                    new Notice(`⚠️ ${result.failed} 个文件更新失败`);
                }
            }

            // 更新索引中的关键词信息
            kw.baseKeyword = newLabel;
            kw.displayText = getDisplayText(newLabel);
            kw.sortLetter = getSortLetter(newLabel);
            const selectedParentId = getSelectedParentId();
            kw.parentId = selectedParentId === oldId ? null : selectedParentId;

            // 更新notes中的路径信息（如果文件名改变了）
            if (keywordChanged && (kw.type === 'plus' || kw.type === 'dot')) {
                const delimiter = kw.type === 'plus' ? '++' : '··';
                kw.notes = kw.notes.map(note => {
                    const oldFull = note.suffix ? oldKeyword + note.suffix : oldKeyword;
                    const newFull = note.suffix ? newLabel + note.suffix : newLabel;
                    const oldPattern = `${delimiter}${oldFull}${delimiter}`;
                    const newPattern = `${delimiter}${newFull}${delimiter}`;

                    if (note.name.includes(oldPattern)) {
                        const newName = note.name.replace(oldPattern, newPattern);
                        const newPath = note.path.replace(note.name, newName);
                        return { ...note, name: newName, path: newPath };
                    }
                    return note;
                });
            }

            if (newId !== oldId) {
                ensureUiStateSets();
                const wasCollapsed = state.collapsedIds.has(oldId);
                const wasNotesOpen = state.notesOpenIds.has(oldId);
                Object.values(state.keywords).forEach(child => {
                    if (child.parentId === oldId) child.parentId = newId;
                });
                delete state.keywords[oldId];
                kw.id = newId;
                state.keywords[newId] = kw;
                state.collapsedIds.delete(oldId);
                state.notesOpenIds.delete(oldId);
                if (wasCollapsed) state.collapsedIds.add(newId);
                if (wasNotesOpen) state.notesOpenIds.add(newId);
            }

            await saveIndex();
            new Notice('✅ 已保存');
            closeEditModal();
            renderViewFn();
        };

        labelInput.addEventListener('keydown', (e) => {
            if (e.key === 'Enter') {
                e.preventDefault();
                saveBtn.click();
            }
        });

        setTimeout(() => {
            labelInput.focus();
            labelInput.select();
        }, 0);
    };

    // 快速定位编辑模块（输入 + 下拉 + 一键编辑）

# 关键词索引管理系统说明

将以下代码块复制到Obsidian笔记中即可使用。

## 核心设计理念

1. **关键词提取**：
   - YAML字段：`关键词管理`（中文）
   - 文件名`++关键词++`和`··关键词··`
   - `+`号是**后缀注解**，用于抽屉内笔记的多层级展示

2. **层级关系**：
   - 关键词之间的层级通过`parentId`实现（套圈式展示）
   - 单个关键词的笔记抽屉内，支持按`+`号分割显示多层级结构

3. **v3.4新增：多层级抽屉**
   - 点击胶囊展开抽屉后，笔记按`suffix`中的`+`号分割成层级结构
   - 例如：`关键词管理+自己做的调查+楞严经分析` 会显示为嵌套的三层结构
   - 支持折叠/展开交互，每层显示笔记数量统计

4. **v3.7优化：可折叠根笔记**
   - 默认只显示层级名
   - 如果层级有直属笔记，显示可折叠的"📝 根笔记"按钮
   - 点击按钮展开/折叠查看该层级的笔记
   - 保持层级结构的清晰和简洁

5. **反向同步**（v3.1新增）：
   - 在管理界面重命名关键词时，自动更新原始笔记文件
   - 支持YAML、++...++、··...··三种格式的反向更新

6. **父级搜索筛选**（v3.2新增）：
   - 编辑关键词时，父级选择支持输入搜索筛选
   - 支持键盘上下键导航和Enter选择
   - 实时高亮匹配文本，显示筛选统计

7. **混合来源类型支持**（v3.3新增）：
   - 同一关键词可通过不同方式（YAML、++...++、··...··）出现在不同笔记中
   - 每个笔记记录其实际来源类型（sourceType字段）
   - 混合类型关键词显示渐变背景和多图标（如 ◆▲●）
   - 笔记抽屉按来源类型分组显示，带彩色分组标题

8. **v3.8新增：单胶囊折叠状态持久化**：
   - 每个有子级的胶囊左侧增加三角开关，单独控制子级展开/折叠
   - 折叠状态会写入 `data/keyword_index.json` 的扁平字典 `capsuleExpanded`
   - 折叠态颜色更深，便于快速识别当前处于收起状态的胶囊

## 使用说明

### 1. 部署

1. 在Obsidian中创建一个新笔记
2. 将上面的代码块完整复制进去
3. 切换到阅读模式即可看到关键词索引

### 2. 三种关键词类型

| 类型 | 格式 | 颜色 | 说明 |
|------|------|------|------|
| ◆ YAML管理 | `关键词管理:` 字段 | 蓝色 | yaml frontmatter中定义 |
| ▲ 层级关键词 | `++关键词++` | 橙色 | 文件名中用双加号包裹 |
| ● 行内关键词 | `··关键词··` | 紫色 | 文件名中用双点号包裹 |

### 3. 关键词提取规则

**重要**：文件名中的`+`号是**后缀注解**，不是层级分隔符！

示例：
- 文件名 `··MMKV-A上项+新开始··`
  - 基础关键词：`MMKV-A上项`
  - 后缀：`+新开始`
- 文件名 `··MMKV-A上项+走上项流程的，材料··`
  - 基础关键词：`MMKV-A上项`（与上面相同）
  - 后缀：`+走上项流程的，材料`

这两个文件都会被归入同一个`MMKV-A上项`胶囊中，打开抽屉可以看到各自的后缀。

### 4. 关键词层级关系

层级关系是**关键词之间**的层级，与文件名中的`+`号无关。

用户可以通过：
1. 点击"⚙️ 管理关键词"进入管理界面
2. 拖拽关键词到其他关键词上，建立父子关系
3. 拖拽到顶部区域，使其成为顶层关键词
4. 点击编辑按钮修改名称或选择父级

### 5. 两种视图

- **展示视图**（默认）：套圈式UI，高层级关键词包含低层级
- **管理视图**：GTD风格的拖拽界面，可调整层级和重命名

### 6. 数据文件

索引数据保存在 `data/keyword_index.json`

### 7. v3.7功能：多层级抽屉（可折叠根笔记）

点击胶囊展开笔记抽屉时，笔记会按`+`号分割显示多层级结构：

**示例效果**：
```
关键词管理 抽屉：
├── 自己做的调查
│   └── 📝 根笔记 (点击展开)
│       └── 学习计划.md
├── 楞严经分析
│   └── 📝 根笔记 (点击展开)
│       └── 《楞严经》学习笔记.md
├── 调查问卷设计.md（无+后缀，直接显示）
├── 另一个项目
│   └── 📝 根笔记 (点击展开)
│       └── 项目计划.md
└── 基础笔记.md（无+后缀）
```

**v3.7交互方式**：
- 默认只显示层级名，笔记默认折叠
- 如果层级有直属笔记，会显示可折叠的"📝 根笔记"按钮
- 点击"📝 根笔记"按钮可展开/折叠查看该层级的笔记
- 点击层级标签（▶ 自己做的调查）可折叠/展开子层级
- 每层显示该层级及所有后代中的笔记数量
- 直接点击笔记名称可打开笔记

**层级规则**：
- `+`号分割的每一级作为一个层级节点
- 无后缀的笔记放在根节点（直属关键词），显示为"📝 根笔记"
- 相同后缀路径的笔记归入同一分支

### 8. v3.8功能：胶囊三角开关与状态记忆

- 每个有子关键词的胶囊左侧会显示 `▶` 三角开关
- 三角方向表示状态：右向为折叠，下向为展开
- 点击三角只控制该胶囊的子级，不影响胶囊点击打开笔记抽屉的逻辑
- “全部折叠 / 全部展开”会同步更新 `capsuleExpanded` 字典
