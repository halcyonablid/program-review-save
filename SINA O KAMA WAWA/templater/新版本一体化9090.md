// 切换到右侧第一个标签页
async function switchToFirstRightTab() {
    const rightSidebar = app.workspace.rightSplit;
    if (!rightSidebar.expanded) {
        rightSidebar.expand();
    }
    const rightLeaves = rightSidebar.children[0].children;
    if (rightLeaves && rightLeaves.length > 0) {
        await app.workspace.setActiveLeaf(rightLeaves[0], true);
    }
}

// 检查元素是否在视图中
function isElementInViewport(el) {
    const rect = el.getBoundingClientRect();
    return (
        rect.top >= 0 &&
        rect.left >= 0 &&
        rect.bottom <= (window.innerHeight || document.documentElement.clientHeight) &&
        rect.right <= (window.innerWidth || document.documentElement.clientWidth)
    );
}

// 等待滚动完成的函数
function waitForScrollEnd() {
    return new Promise((resolve) => {
        let lastPos = window.scrollY;
        let sameCount = 0;
        
        function checkScrollPosition() {
            const currentPos = window.scrollY;
            if (currentPos === lastPos) {
                sameCount++;
                if (sameCount >= 3) {
                    resolve();
                    return;
                }
            } else {
                sameCount = 0;
                lastPos = currentPos;
            }
            requestAnimationFrame(checkScrollPosition);
        }
        
        requestAnimationFrame(checkScrollPosition);
    });
}

// 查找并滚动到指定笔记
async function findAndScrollToNote() {
    const currentFileName = tp.file.title;
    
    try {
        const customTable = document.querySelector('.custom-table tbody');
        if (!customTable) {
            throw new Error('Custom table not found');
        }

        const targetRow = Array.from(customTable.querySelectorAll('tr')).find(row => {
            const link = row.querySelector('.filename-column a');
            return link && link.textContent === currentFileName;
        });

        if (targetRow) {
            return new Promise(async (resolve) => {
                const observer = new IntersectionObserver(async (entries) => {
                    const entry = entries[0];
                    if (entry.isIntersecting) {
                        observer.disconnect();
                        await waitForScrollEnd();
                        const rowIndex = Array.from(customTable.children).indexOf(targetRow) + 1;
                        new Notice(`已找到当前笔记：${currentFileName}（第 ${rowIndex} 行）`);
                        
                        // 计算需要滚动的距离
                        const rowRect = targetRow.getBoundingClientRect();
                        const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
                        const targetScrollTop = scrollTop + rowRect.top - window.innerHeight / 2 + rowRect.height / 2;
                        
                        // 滚动到元素的中心
                        window.scrollTo({
                            top: targetScrollTop,
                            behavior: 'smooth'
                        });
                        
                        resolve(true);
                    }
                }, {
                    threshold: 1.0,
                    rootMargin: '0px'
                });

                observer.observe(targetRow);

                if (!isElementInViewport(targetRow)) {
                    targetRow.scrollIntoView({ 
                        behavior: 'smooth', 
                        block: 'center' 
                    });
                } else {
                    observer.disconnect();
                    resolve(true);
                }
            });
        } else {
            new Notice(`在编号列表中未找到当前笔记：${currentFileName}`);
            return false;
        }
    } catch (error) {
        console.error('Error:', error);
        new Notice(`查找过程中发生错误：${error.message}`);
        return false;
    }
}


// 主函数：更新YAML并等待加载
async function updateYAMLAndWaitForLoad() {
    // 保存当前活动叶子
    const currentLeaf = app.workspace.activeLeaf;
    
    await switchToFirstRightTab();
    
    const currentFileName = tp.file.title;
    const targetNote = app.vault.getAbstractFileByPath('项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain/新的视图看看.md');
    
    if (!targetNote) {
        new Notice('未找到目标笔记：新的视图看看.md');
        return;
    }

    try {
        // 更新笔记名，保持加载OK为空
        const content = await app.vault.read(targetNote);
        let newContent;
        
        if (content.startsWith('---')) {
            const yamlEndIndex = content.indexOf('---', 3);
            if (yamlEndIndex !== -1) {
                const yamlContent = content.substring(0, yamlEndIndex);
                const restContent = content.substring(yamlEndIndex);
                
                let updatedYaml = yamlContent
                    .replace(/笔记名:.*$/m, `笔记名: ${currentFileName}`)
                    .replace(/加载OK:.*$/m, '加载OK:');
                
                newContent = updatedYaml + restContent;
            }
        } else {
            newContent = `---\n笔记名: ${currentFileName}\n加载OK:\n---\n${content}`;
        }

        await app.vault.modify(targetNote, newContent);
        new Notice(`已更新笔记名属性为：${currentFileName}`);

     
        // 等待视图稳定
        await new Promise(resolve => setTimeout(resolve, 5000));

  

        // 等待额外时间确保加载OK更新完成
        await new Promise(resolve => setTimeout(resolve, 500));

        // 执行滚动
        const scrollResult = await findAndScrollToNote();
        if (!scrollResult) {
            throw new Error('未找到目标笔记');
        }

        // 恢复到原来的活动叶子
        if (currentLeaf) {
            await app.workspace.setActiveLeaf(currentLeaf, true);
        }

    } catch (error) {
        console.error('Error:', error);
        new Notice(`操作失败：${error.message}`);
        throw error;
    }
}

// 执行主流程
try {
    await updateYAMLAndWaitForLoad();
} catch (error) {
    console.error('执行失败:', error);
    new Notice(`操作失败：${error.message}`);
}
