async function createSequentialNote(tp) {
    // 1. Find the latest note
    const files = app.vault.getMarkdownFiles();
    let latestNote = null;
    
    for (const file of files) {
        const content = await app.vault.read(file);
        if (content.includes("车间修理项目: true")) {
            latestNote = file;
            break;
        }
    }
    
    if (!latestNote) {
        await tp.system.prompt("未找到带有'车间修理项目'标记的笔记！");
        return;
    }

    // 2. Parse the current code and generate new code
    const currentTitle = latestNote.basename;
    const codeMatch = currentTitle.match(/ZL@[\d.]+/);
    if (!codeMatch) {
        await tp.system.prompt("无法解析当前笔记的编号！");
        return;
    }

    const currentCode = codeMatch[0];
    const codeParts = currentCode.split('.');
    const lastNumber = parseInt(codeParts[codeParts.length - 1]);
    const newLastNumber = (lastNumber + 1).toString().padStart(3, '0');
    codeParts[codeParts.length - 1] = newLastNumber;
    const newCode = codeParts.join('.');

    // 3. Get new note name
    const newName = await tp.system.prompt("请输入新笔记的名称");
    const newFileName = `${newCode}- ${newName}`;

    // 4. Create new note content with specified YAML frontmatter
    const currentDate = tp.date.now("YYYY-MM-DD");
    const newContent = `---
qishiriqidate: ${currentDate}
atomle: true
antinet: atom
车间修理项目: true
世界的构造: true
---

`;

    // 5. Create new note
    const newFile = await tp.file.create_new(newContent, newFileName);

    // 6. Update old note to remove latest的发现
    const oldContent = await app.vault.read(latestNote);
    const updatedOldContent = oldContent.replace("车间修理项目: true", "车间修理项目: false");
    await app.vault.modify(latestNote, updatedOldContent);

    // 7. Open new note
    await app.workspace.activeLeaf.openFile(newFile);

    // 8. Show confirmation message
    await tp.system.prompt(`已创建新笔记: ${newFileName}\n并更新了原笔记的车间修理项目状态`);

    return newFile;
}

await createSequentialNote(tp);
