async function scrollToFirstListItem() {
    const targetPath = '项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain/本格视图不闪闪.md';
    
    // 获取目标文件
    const targetFile = app.vault.getAbstractFileByPath(targetPath);
    
    if (!targetFile) {
        new Notice('未找到目标文件');
        return;
    }

    try {
        // 查找或打开目标文件
        const leaf = app.workspace.getLeavesOfType('markdown').find(l => 
            l.view.file && l.view.file.path === targetFile.path
        ) || await app.workspace.getLeaf('tab').openFile(targetFile);

        app.workspace.setActiveLeaf(leaf);

        // 等待页面加载
        await new Promise(resolve => setTimeout(resolve, 500));

        // 查找自定义表格
        const customTable = document.querySelector('.custom-table tbody');
        if (!customTable) {
            throw new Error('未找到自定义表格');
        }

        // 获取第一行
        const firstRow = customTable.querySelector('tr');
        if (firstRow) {
            // 滚动到第一行
            firstRow.scrollIntoView({ behavior: 'auto', block: 'start' });
         
        } else {
            new Notice('编号列表为空');
        }
    } catch (error) {
        console.error('Error:', error);
        new Notice(`滚动过程中发生错误：${error.message}`);
    }
}

async function switchToMainView(fileName) {
    // 获取所有markdown叶子
    const markdownLeaves = app.workspace.getLeavesOfType('markdown');

    // 查找包含指定文件名的叶子
    const targetLeaf = markdownLeaves.find(leaf => {
        if (leaf.view && leaf.view.file) {
            return leaf.view.file.name === fileName;
        }
        return false;
    });

    if (targetLeaf) {
        app.workspace.setActiveLeaf(targetLeaf);
        await new Promise(resolve => setTimeout(resolve, 500)); // 短暂延迟以确保切换完成
      
    } else {
    
    }
}

async function updateYAML() {
    const currentFileName = tp.file.title;
    
    // 获取目标笔记（编号列表视图所在的笔记）
    const targetNote = app.vault.getAbstractFileByPath('项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain/本格视图不闪闪.md');
    
    if (!targetNote) {
        new Notice('未找到目标笔记：本格视图不闪闪.md');
        return;
    }

    try {
        // 读取目标笔记的内容
        const content = await app.vault.read(targetNote);
        
        // 解析并更新YAML前置元数据
        let newContent;
        if (content.startsWith('---')) {
            // 如果已经存在YAML前置元数据
            const yamlEndIndex = content.indexOf('---', 3);
            if (yamlEndIndex !== -1) {
                const yamlContent = content.substring(0, yamlEndIndex);
                const restContent = content.substring(yamlEndIndex);
                
                // 更新或添加笔记名属性
                if (yamlContent.includes('笔记名:')) {
                    newContent = yamlContent.replace(/笔记名:.*$/m, `笔记名: ${currentFileName}`) + restContent;
                } else {
                    newContent = yamlContent.trim() + `\n笔记名: ${currentFileName}\n` + restContent;
                }
            }
        } else {
            // 如果不存在YAML前置元数据，添加新的
            newContent = `---\n笔记名: ${currentFileName}\n---\n${content}`;
        }

        // 写入更新后的内容
        await app.vault.modify(targetNote, newContent);
        
    

    } catch (error) {
        console.error('Error:', error);
        new Notice(`更新YAML时发生错误：${error.message}`);
    }
}

// 执行主函数
await scrollToFirstListItem();
await switchToMainView('本格视图不闪闪.md');
await updateYAML();
await switchToMainView(initialFileName); // 最后切换回最初打开的文件
