<%*
async function main() {
    // 添加用户输入对话框
    let startDate = await tp.system.prompt("请输入起始日期 (格式: YYYY-MM-DD)");
    if (!startDate || !/^\d{4}-\d{2}-\d{2}$/.test(startDate)) {
        console.error("无效的日期格式");
        return;
    }

    function extractDailyTasks(content, date) {
        const lines = content.split('\n');
        const tasks = [];
        let isDailyTaskSection = false;

        for (const line of lines) {
            if (line.trim() === '# 日课') {
                isDailyTaskSection = true;
                continue;
            }

            if (isDailyTaskSection && line.startsWith('- "')) {
                const match = line.match(/- "(.*?)",(done|undone)/);
                if (match) {
                    tasks.push({
                        task: match[1],
                        status: match[2]
                    });
                }
            }

            if (isDailyTaskSection && line.startsWith('#')) {
                break; // 遇到下一个标题就停止处理
            }
        }

        return {
            date: date,
            completedTasks: tasks.filter(task => task.status === 'done').map(task => task.task)
        };
    }

    let allDailyTasks = [];

    const relativeDiaryPath = '项目/35-给日程安排的文件夹/35C-日记的文件夹';
    const files = app.vault.getFiles().filter(file => file.path.startsWith(relativeDiaryPath));
    console.log("找到的文件数量:", files.length);

    for (const file of files) {
        console.log("处理文件:", file.name);
        if (/^\d{4}-\d{2}-\d{2}\.md$/.test(file.name)) {
            const fileDate = file.name.slice(0, -3);
            // 只处理起始日期之后的文件
            if (fileDate >= startDate) {
                const content = await app.vault.read(file);
                console.log("文件内容长度:", content.length);
                const dailyTasks = extractDailyTasks(content, fileDate);
                console.log("提取的完成任务数量:", dailyTasks.completedTasks.length);
                allDailyTasks.push(dailyTasks);
            } else {
                console.log("文件日期早于起始日期，跳过:", file.name);
            }
        } else {
            console.log("文件名不符合格式:", file.name);
        }
    }

    allDailyTasks.sort((a, b) => a.date.localeCompare(b.date));

    // 生成YAML格式的日课完成情况
    let newYamlContent = "---\ndaily_tasks:\n";
    allDailyTasks.forEach(day => {
        if (day.completedTasks.length > 0) {
            newYamlContent += `  - date: "${day.date}"\n`;
            newYamlContent += `    completed:\n`;
            day.completedTasks.forEach(task => {
                newYamlContent += `      - "${task}"\n`;
            });
        }
    });
    newYamlContent += "---\n";

    // 读取目标文件
    const targetFilePath = '项目/35-给日程安排的文件夹/35C-日记的文件夹/日课完成情况.md';
    const targetFile = app.vault.getAbstractFileByPath(targetFilePath);
    if (targetFile) {
        let fileContent = await app.vault.read(targetFile);

        // 移除所有现有的YAML frontmatter
        const contentWithoutYaml = fileContent.replace(/^---\n[\s\S]*?\n---\n/, '');

        // 添加新的YAML frontmatter
        const newFileContent = newYamlContent + contentWithoutYaml;

        await app.vault.modify(targetFile, newFileContent);
        console.log("YAML 数据已更新到文件：", targetFilePath);
    } else {
        console.error("目标文件不存在：", targetFilePath);
    }
}

main();
%>
