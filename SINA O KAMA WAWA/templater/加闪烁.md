async function findAndScrollToNote() {
    const currentFileName = tp.file.title;

    // Wait for the page to load
    await new Promise(resolve => setTimeout(resolve, 200));

    try {
        // Check if the filename is anywhere in the page
        const pageContent = document.body.textContent;
        const foundInPage = pageContent.includes(currentFileName);

        // Find the custom table
        const customTable = document.querySelector('.custom-table tbody');
        if (!customTable) {
            throw new Error('Custom table not found');
        }

        // Find the row with the current filename
        const targetRow = Array.from(customTable.querySelectorAll('tr')).find(row => {
            const link = row.querySelector('.filename-column a');
            return link && link.textContent === currentFileName;
        });

        if (targetRow) {
            // Set up IntersectionObserver
            const observer = new IntersectionObserver((entries) => {
                if (!entries[0].isIntersecting) {
                    targetRow.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    observer.disconnect();
                    // Add color change effect after scrolling
                    setTimeout(() => changeBackgroundColor(targetRow), 500);
                }
            }, { threshold: 1.0 });

            observer.observe(targetRow);

            // Trigger initial scroll if not in view
            if (foundInPage && !isElementInViewport(targetRow)) {
                const scrollContainer = document.querySelector('.workspace-leaf-content');
                if (scrollContainer) {
                    const containerHeight = scrollContainer.clientHeight;
                    const newScroll = Math.random() < 0.5 ? containerHeight / 4 : containerHeight * 3 / 4;
                    scrollContainer.scrollTo({
                        top: newScroll,
                        behavior: 'smooth'
                    });
                }
            }

            const rowIndex = Array.from(customTable.children).indexOf(targetRow) + 1;
            new Notice(`已找到并滚动到当前笔记：${currentFileName}（第 ${rowIndex} 行）`);
        } else {
            new Notice(`在编号列表中未找到当前笔记：${currentFileName}`);
        }
    } catch (error) {
        console.error('Error:', error);
        new Notice(`查找过程中发生错误：${error.message}`);
    }
}

function isElementInViewport(el) {
    const rect = el.getBoundingClientRect();
    return (
        rect.top >= 0 &&
        rect.left >= 0 &&
        rect.bottom <= (window.innerHeight || document.documentElement.clientHeight) &&
        rect.right <= (window.innerWidth || document.documentElement.clientWidth)
    );
}

function changeBackgroundColor(element) {
    const originalColor = element.style.backgroundColor;
    element.style.transition = 'background-color 0.3s ease-in-out';
    element.style.backgroundColor = '#0047AB'; // 柯蓝因蓝

    setTimeout(() => {
        element.style.backgroundColor = originalColor;
    }, 5000); // 5秒后恢复原色
}

// Add CSS for smooth transition
const style = document.createElement('style');
style.textContent = `
    .custom-table tr {
        transition: background-color 0.3s ease-in-out;
    }
`;
document.head.appendChild(style);

// Run the main function
await findAndScrollToNote();
