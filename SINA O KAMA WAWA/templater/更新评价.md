async function main() {
    // 添加用户输入对话框
    let startDate = await tp.system.prompt("请输入起始日期 (格式: YYYY-MM-DD)");
    if (!startDate || !/^\d{4}-\d{2}-\d{2}$/.test(startDate)) {
        console.error("无效的日期格式");
        return;
    }

    function extractActivities(content, date) { 
        const lines = content.split('\n'); 
        const activities = []; 
        let currentActivity = null; 
        let isInActivitySection = false; 

        for (const line of lines) { 
            if (line.trim() === '# 二、状态评估（后日评估）：') { 
                isInActivitySection = true; 
                continue; 
            } 

            if (line.trim() === '#### 今日相关笔记') {
                break; // 遇到这个标记就停止处理
            }

            if (!isInActivitySection) continue;

            if (line.startsWith('- 事项：')) { 
                if (currentActivity) activities.push(currentActivity); 
                currentActivity = { date: date, issue: line.substring(5).trim(), category: '', score: 0 }; 
            } else if (line.startsWith('	- 类别：') && currentActivity) { 
                currentActivity.category = line.substring(5).trim(); 
            } else if (line.startsWith('	- 评分：') && currentActivity) { 
                const scoreMatch = line.match(/\d+/);
                if (scoreMatch) {
                    currentActivity.score = parseInt(scoreMatch[0], 10);
                }
                activities.push(currentActivity); 
                currentActivity = null; 
            } 
        }

        if (currentActivity) activities.push(currentActivity);
        return activities;
    }

    let allActivities = [];

    const relativeDiaryPath = '项目/35-给日程安排的文件夹/35C-日记的文件夹';
    const files = app.vault.getFiles().filter(file => file.path.startsWith(relativeDiaryPath));
    console.log("找到的文件数量:", files.length);

    for (const file of files) {
        console.log("处理文件:", file.name);
        if (/^\d{4}-\d{2}-\d{2}\.md$/.test(file.name)) {
            const fileDate = file.name.slice(0, -3);
            // 只处理起始日期之后的文件
            if (fileDate >= startDate) {
                const content = await app.vault.read(file);
                console.log("文件内容长度:", content.length);
                const activities = extractActivities(content, fileDate);
                console.log("提取的活动数量:", activities.length);
                allActivities = allActivities.concat(activities);
            } else {
                console.log("文件日期早于起始日期，跳过:", file.name);
            }
        } else {
            console.log("文件名不符合格式:", file.name);
        }
    }

    allActivities.sort((a, b) => {
        if (a.date !== b.date) {
            return a.date.localeCompare(b.date);
        }
        return a.score - b.score;
    });

    // 生成YAML格式的活动信息
    let newYamlContent = "---\nactivities:\n";
    allActivities.forEach(activity => {
        newYamlContent += `  - date: "${activity.date}"\n`;
        newYamlContent += `    issue: "${activity.issue}"\n`;
        newYamlContent += `    category: "${activity.category.replace(/^：/, '')}"\n`;
        newYamlContent += `    score: ${activity.score}\n`;
    });
    newYamlContent += "---\n";

    // 读取目标文件
    const targetFilePath = '项目/35-给日程安排的文件夹/35C-日记的文件夹/评价的视图.md';
    const targetFile = app.vault.getAbstractFileByPath(targetFilePath);
    if (targetFile) {
        let fileContent = await app.vault.read(targetFile);

        // 移除所有现有的YAML frontmatter
        const contentWithoutYaml = fileContent.replace(/^---\n[\s\S]*?\n---\n/, '');

        // 添加新的YAML frontmatter
        const newFileContent = newYamlContent + contentWithoutYaml;

        await app.vault.modify(targetFile, newFileContent);
        console.log("YAML 数据已更新到文件：", targetFilePath);
    } else {
        console.error("目标文件不存在：", targetFilePath);
    }
}

main();
%>
