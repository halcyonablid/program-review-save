async function updateViewFileYAML() {
    // 获取当前笔记的文件名
    const currentFileName = tp.file.title;

    // 获取当前时间戳作为文件名后缀
    const timestamp = Date.now();
    
    // 指定视图文件路径，加上时间戳后缀
    const viewFilePath = `项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain/当前页面在树上面的位置_${timestamp}.md`;
    
    try {
        // 获取模板文件内容
        const templatePath = "模板/点开新视图的模板.md";
        const templateFile = app.vault.getAbstractFileByPath(templatePath);
        if (!templateFile) {
            new Notice("未找到模板文件！");
            return;
        }
        let templateContent = await app.vault.read(templateFile);

        // 在模板内容的YAML部分添加或更新笔记名属性
        if (templateContent.startsWith('---')) {
            const yamlEndIndex = templateContent.indexOf('---', 3);
            if (yamlEndIndex !== -1) {
                const yamlContent = templateContent.substring(0, yamlEndIndex);
                const restContent = templateContent.substring(yamlEndIndex);
                
                if (yamlContent.includes('最新的事件记录的笔记名:')) {
                    templateContent = yamlContent.replace(/最新的事件记录的笔记名:.*$/m, `最新的事件记录的笔记名: ${currentFileName}`) + restContent;
                } else {
                    templateContent = yamlContent.trim() + `\n最新的事件记录的笔记名: ${currentFileName}\n` + restContent;
                }
            }
        } else {
            templateContent = `---\n最新的事件记录的笔记名: ${currentFileName}\n---\n${templateContent}`;
        }

        // 创建新文件，使用模板内容
        await app.vault.create(viewFilePath, templateContent);

        // 获取新创建的文件
        const viewFile = app.vault.getAbstractFileByPath(viewFilePath);
        if (!viewFile) {
            new Notice("创建文件失败！");
            return;
        }

        // 在新标签页中打开视图文件
        const newLeaf = app.workspace.getLeaf('tab');
        await newLeaf.openFile(viewFile);
        
        // 设置为预览模式
        if (newLeaf.view.getMode() !== 'preview') {
            await newLeaf.setViewState({
                type: 'markdown',
                state: { mode: 'preview' }
            });
        }

        // 锁定标签页
        if (newLeaf.view.leaf) {
            newLeaf.view.leaf.setPinned(true);
        }

        new Notice('已创建新视图文件并打开（预览模式）');
    } catch (error) {
        console.error('Error:', error);
        new Notice(`创建文件时发生错误：${error.message}`);
    }
}

// 执行主函数
await updateViewFileYAML();
%>
