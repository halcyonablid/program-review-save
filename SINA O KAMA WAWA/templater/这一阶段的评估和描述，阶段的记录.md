async function createSequentialNote(tp) {
    // 1. Find the latest note
    const files = app.vault.getMarkdownFiles();
    let latestNote = null;
    
    for (const file of files) {
        const content = await app.vault.read(file);
        if (content.includes("日程阶段分析: true")) {
            latestNote = file;
            break;
        }
    }
    
    if (!latestNote) {
        await tp.system.prompt("未找到带有'日程阶段分析'标记的笔记！");
        return;
    }

    // 2. Parse the current code and generate new code
    const currentTitle = latestNote.basename;
    const codeMatch = currentTitle.match(/ZL@[\d.]+/);
    if (!codeMatch) {
        await tp.system.prompt("无法解析当前笔记的编号！");
        return;
    }

    const currentCode = codeMatch[0];
    const codeParts = currentCode.split('.');
    const lastNumber = parseInt(codeParts[codeParts.length - 1]);
    const newLastNumber = (lastNumber + 1).toString().padStart(3, '0');
    codeParts[codeParts.length - 1] = newLastNumber;
    const newCode = codeParts.join('.');

    // 3. Get new note name
    const newName = await tp.system.prompt("请输入新笔记的名称");
    const newFileName = `${newCode}- ${newName}`;

    // 4. Create new note content with specified YAML frontmatter
    const currentDate = tp.date.now("YYYY-MM-DD");
    const currentTime = tp.date.now("HH:mm:ss");
    
    // 读取上一则笔记的内容
    const latestContent = await app.vault.read(latestNote);
    
    // 提取特定内容
    const marker = "+++++++++++++++++++++++——————————————————++++++++++++++++++++++++++++++";
    const startIndex = latestContent.indexOf(marker);
    const endIndex = latestContent.lastIndexOf(marker);
    
    let copiedContent = "";
    if (startIndex !== -1 && endIndex !== -1 && endIndex > startIndex) {
        copiedContent = latestContent.slice(startIndex + marker.length, endIndex).trim();
    }

    const newContent = `---
qishiriqidate: ${currentDate}
qishiriqitime: ${currentTime}
atomle: true
antinet: atom
日程阶段分析: true
---

${marker}
${copiedContent}
${marker}

`;

    // 5. Create new note using Obsidian API
    const newFilePath = app.vault.getAbstractFileByPath(newFileName + '.md');
    if (newFilePath) {
        await app.vault.modify(newFilePath, newContent);
    } else {
        await app.vault.create(newFileName + '.md', newContent);
    }

    // 6. Update old note to remove latest的发现
    const updatedOldContent = latestContent.replace("日程阶段分析: true", "日程阶段分析: false");
    await app.vault.modify(latestNote, updatedOldContent);

    // 7. Open new note using Obsidian API
    const newFile = app.vault.getAbstractFileByPath(newFileName + '.md');
    if (newFile) {
        await app.workspace.activeLeaf.openFile(newFile);
    }

    // 8. Show confirmation message
    await tp.system.prompt(`已创建新笔记: ${newFileName}\n并更新了原笔记的日程阶段分析状态`);

    return newFile;
}

await createSequentialNote(tp);
