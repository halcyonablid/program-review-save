<%*
// 读取文件内容
let content = tp.file.content;

// 分离非标题内容和带标题的内容块
let firstTitleIndex = content.indexOf("\n# ");
let nonTitleContent = firstTitleIndex > 0 ? content.substring(0, firstTitleIndex).trim() : "";

// 提取所有标题及其内容块
let sections = [];
let lines = content.split('\n');
let currentSection = null;

for (let i = 0; i < lines.length; i++) {
    let line = lines[i];
    if (line.startsWith('# ') && line.includes('*')) {
        // 如果找到新标题，保存之前的section
        if (currentSection) {
            sections.push(currentSection);
        }
        // 创建新的section
        let letter = line.match(/\*([A-Z])$/)[1];
        currentSection = {
            title: line,
            letter: letter,
            content: []
        };
    } else if (currentSection) {
        // 将行添加到当前section的内容中
        currentSection.content.push(line);
    }
}
// 添加最后一个section
if (currentSection) {
    sections.push(currentSection);
}

// 按字母排序
sections.sort((a, b) => a.letter.localeCompare(b.letter));

// 组合最终内容
let result = '';

// 添加非标题内容（如果存在）
if (nonTitleContent) {
    result += nonTitleContent + '\n\n';
}

// 添加排序后的标题及其内容
result += sections
    .map(section => {
        return section.title + '\n' + section.content.join('\n').trim();
    })
    .join('\n\n');

// 替换文件全部内容
await app.vault.modify(tp.file.find_tfile(tp.file.title), result);
%>
