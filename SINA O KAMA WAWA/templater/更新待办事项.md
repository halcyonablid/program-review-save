<%*
async function collectProjectTodos() {
    // 获取所有项目待办集合卡
    const query = '"项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain"';
    const files = app.vault.getMarkdownFiles()
        .filter(file => file.path.startsWith(query.slice(1, -1)));

    let allTodos = [];

    // 遍历所有文件
    for (const file of files) {
        const content = await app.vault.read(file);
        
        // 检查是否是项目待办集合卡
        if (file.name.endsWith('（项目待办集合卡）') && 
            content.includes('索引卡: true')) {
            
            // 分割内容为行
            const lines = content.split('\n');
            let collecting = false;
            let currentTodos = [];
            
            for (let i = 0; i < lines.length; i++) {
                const line = lines[i].trim();
                
                // 检查是否是待办D标签的开始
                if (line.startsWith('# 待办') && line.endsWith('D')) {
                    collecting = true;
                    continue;
                }
                // 检查是否遇到下一个一级标签
                else if (collecting && line.startsWith('# ')) {
                    collecting = false;
                }
                
                // 收集内容
                if (collecting && line && !line.startsWith('#')) {
                    currentTodos.push({
                        content: line,
                        source: file.name,
                        section: lines[i-1].startsWith('# ') ? lines[i-1] : '待办事项'
                    });
                }
            }
            
            allTodos = allTodos.concat(currentTodos);
        }
    }

    // 生成汇总内容
    const targetPath = "项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain/项目待办事项.md";
    const today = tp.date.now("YYYY-MM-DD HH:mm:ss");
    
    let summaryContent = `---
更新时间: ${today}
---

# 项目待办事项汇总

`;

    // 按来源文件分组显示待办事项
    const groupedTodos = {};
    allTodos.forEach(todo => {
        if (!groupedTodos[todo.source]) {
            groupedTodos[todo.source] = [];
        }
        groupedTodos[todo.source].push(todo);
    });

    // 生成分组内容
    for (const source in groupedTodos) {
        summaryContent += `## 来自：${source}\n\n`;
        
        // 按照section分组
        const sectionTodos = {};
        groupedTodos[source].forEach(todo => {
            if (!sectionTodos[todo.section]) {
                sectionTodos[todo.section] = [];
            }
            sectionTodos[todo.section].push(todo.content);
        });

        // 输出每个section的内容
        for (const section in sectionTodos) {
            if (section !== '待办事项') {
                summaryContent += `### ${section}\n`;
            }
            sectionTodos[section].forEach(todo => {
                summaryContent += `- ${todo}\n`;
            });
            summaryContent += '\n';
        }
    }

    // 写入文件
    const targetFile = app.vault.getAbstractFileByPath(targetPath);
    if (targetFile) {
        await app.vault.modify(targetFile, summaryContent);
    } else {
        await app.vault.create(targetPath, summaryContent);
    }

    // 显示完成消息
    const message = `汇总完成！\n共处理了 ${Object.keys(groupedTodos).length} 个文件，${allTodos.length} 个待办事项。`;
    await tp.system.prompt(message);
}

// 执行函数
await collectProjectTodos();
%>
