async function findAndScrollToNote() {
    const currentFileName = tp.file.title;

    // Wait for the page to load
    await new Promise(resolve => setTimeout(resolve, 500));

    try {
        // 获取当前文件
        const currentFile = app.workspace.getActiveFile();
        if (!currentFile) {
            throw new Error('无法获取当前文件');
        }

        // 等待 DataviewJS 视图加载完成
        const attempts = 20; // 最多尝试20次
        let attempt = 0;
        
        while (attempt < attempts) {
            // 查找表格元素
            const customTable = document.querySelector('.custom-table tbody');
            if (customTable) {
                // 找到所有行
                const rows = Array.from(customTable.querySelectorAll('tr'));
                const targetRow = rows.find(row => {
                    const link = row.querySelector('.filename-column a');
                    return link && link.textContent === currentFileName;
                });

                if (targetRow) {
                    // 计算当前行在所有数据中的位置
                    const rowIndex = rows.indexOf(targetRow);
                    
                    // 计算目标页码
                    const ITEMS_PER_PAGE = 50;
                    const targetPage = Math.floor(rowIndex / ITEMS_PER_PAGE);
                    
                    // 执行页面跳转和滚动
                    const script = `
                        currentPage = ${targetPage};
                        renderTable();
                        
                        setTimeout(() => {
                            const tbody = document.querySelector('.custom-table tbody');
                            if (tbody) {
                                const rows = tbody.querySelectorAll('tr');
                                const targetRow = rows[${rowIndex % ITEMS_PER_PAGE}];
                                if (targetRow) {
                                    targetRow.scrollIntoView({ behavior: 'smooth', block: 'center' });
                                    // 添加高亮效果
                                    targetRow.style.transition = 'background-color 0.3s';
                                    targetRow.style.backgroundColor = 'rgba(255, 255, 0, 0.2)';
                                    setTimeout(() => {
                                        targetRow.style.backgroundColor = '';
                                    }, 2000);
                                }
                            }
                        }, 300);
                    `;

                    // 执行脚本
                    eval(script);
                    
                    new Notice(`已定位到当前笔记：${currentFileName}`);
                    return;
                }
            }

            // 如果没找到，等待后重试
            await new Promise(resolve => setTimeout(resolve, 200));
            attempt++;
        }

        if (attempt >= attempts) {
            new Notice(`未能在编号列表中找到当前笔记：${currentFileName}`);
        }

    } catch (error) {
        console.error('Error:', error);
        new Notice(`查找过程中发生错误：${error.message}`);
    }
}

// 执行主函数
await findAndScrollToNote();
