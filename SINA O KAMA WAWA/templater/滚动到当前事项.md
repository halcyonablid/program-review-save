async function findAndScrollToNoteInTodoList() {
    const currentFileName = tp.file.title;

    // 等待页面加载
    await new Promise(resolve => setTimeout(resolve, 500));

    try {
        // 查找atom-files-table表格
        const atomTable = document.querySelector('.atom-files-table tbody');
        if (!atomTable) {
            throw new Error('Atom files table not found');
        }

        // 查找包含当前文件名的行
        const targetRow = Array.from(atomTable.querySelectorAll('tr')).find(row => {
            const link = row.querySelector('.internal-link');
            return link && (link.textContent === currentFileName || 
                          link.getAttribute('href').includes(currentFileName));
        });

        if (targetRow) {
            // 设置 IntersectionObserver
            const observer = new IntersectionObserver((entries) => {
                if (!entries[0].isIntersecting) {
                    targetRow.scrollIntoView({
                        behavior: 'smooth',
                        block: 'center'
                    });
                    // 添加高亮效果
                    targetRow.style.transition = 'background-color 0.3s';
                    targetRow.style.backgroundColor = 'rgba(255, 255, 0, 0.2)';
                    setTimeout(() => {
                        targetRow.style.backgroundColor = '';
                    }, 2000);
                    observer.disconnect();
                }
            }, { threshold: 1.0 });

            observer.observe(targetRow);

            // 如果不在视图中，触发滚动
            if (!isElementInViewport(targetRow)) {
                targetRow.scrollIntoView({
                    behavior: 'smooth',
                    block: 'center'
                });
            }

            // 计算行号并显示通知
            const rowIndex = Array.from(atomTable.children).indexOf(targetRow) + 1;
            new Notice(`已在待办事项列表中找到并滚动到当前笔记（第 ${rowIndex} 行）`);
        } else {
            new Notice(`在待办事项列表中未找到当前笔记：${currentFileName}`);
        }
    } catch (error) {
        console.error('Error:', error);
        new Notice(`查找过程中发生错误：${error.message}`);
    }
}

function isElementInViewport(el) {
    const rect = el.getBoundingClientRect();
    return (
        rect.top >= 0 &&
        rect.left >= 0 &&
        rect.bottom <= (window.innerHeight || document.documentElement.clientHeight) &&
        rect.right <= (window.innerWidth || document.documentElement.clientWidth)
    );
}

// 执行主函数
await findAndScrollToNoteInTodoList();
