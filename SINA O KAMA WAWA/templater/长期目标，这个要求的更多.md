async function scheduleSprints() {
    const goals = await readLongTermGoals();
    const processedGoals = processGoalData(goals);
    const scheduledGoals = scheduleSprintsForGoals(processedGoals);
    const updatedGoals = await updateYAMLFrontmatter(scheduledGoals);
    showResultPopup(updatedGoals);
    return "Sprints scheduled successfully!";
}

async function readLongTermGoals() {
    const files = app.vault.getMarkdownFiles();
    const goals = [];

    for (const file of files) {
        const content = await app.vault.read(file);
        const frontmatter = app.metadataCache.getFileCache(file)?.frontmatter;

        if (frontmatter && 
            frontmatter.qishiriqidate && 
            frontmatter.duedate && 
            (frontmatter.gaocengji === true || frontmatter.gaocengji === 'true')) {
            const goalSteps = extractGoalSteps(content);
            const sprintDates = {};
            
            // 动态读取存在的sprint日期
            for (let i = 1; i <= goalSteps.length; i++) {
                if (frontmatter[`sprint${i}`]) {
                    sprintDates[i] = new Date(frontmatter[`sprint${i}`]);
                }
            }

            goals.push({
                file: file,
                startDate: new Date(frontmatter.qishiriqidate),
                dueDate: new Date(frontmatter.duedate),
                steps: goalSteps,
                existingSprints: frontmatter.sprints || [],
                sprintDates: sprintDates
            });
        }
    }

    return goals;
}

function extractGoalSteps(content) {
    const lines = content.split('\n');
    const steps = [];
    let currentStep = null;
    let isInGoalSection = false;

    for (const line of lines) {
        if (line.trim() === '# 接下来的分解步骤，以及抵达的路点') {
            isInGoalSection = true;
            continue;
        }

        if (isInGoalSection) {
            if (line.startsWith('- goal:')) {
                if (currentStep) steps.push(currentStep);
                currentStep = { goal: line.substring(7).trim(), duration: 0, weight: 1 };
            } else if (line.includes('预计使用时间:') && currentStep) {
                currentStep.duration = parseInt(line.split(':')[1].trim()) || 0;
                currentStep.weight = Math.log(currentStep.duration + 1) + 1;
                steps.push(currentStep);
                currentStep = null;
            } else if (line.trim().startsWith('#')) {
                break;
            }
        }
    }

    if (currentStep) steps.push(currentStep);
    return steps;
}

function processGoalData(goals) {
    return goals.map(goal => ({
        ...goal,
        totalDuration: goal.steps.reduce((sum, step) => sum + step.duration, 0),
        totalWeight: goal.steps.reduce((sum, step) => sum + step.weight, 0),
        availableDays: Math.floor((goal.dueDate - goal.startDate) / (1000 * 60 * 60 * 24))
    }));
}

function scheduleSprintsForGoals(goals) {
    const timeline = createTimeline(goals);
    const scheduledGoals = goals.map(goal => ({ ...goal, sprints: [] }));

    scheduledGoals.sort((a, b) => a.dueDate - b.dueDate);

    for (const goal of scheduledGoals) {
        let currentDate = new Date(goal.startDate);
        let sprintIndex = 1;

        for (const step of goal.steps) {
            const sprintDuration = step.duration;
            const sprintDeadline = goal.sprintDates[sprintIndex] || goal.dueDate;
            let bestStart = findBestStartDate(goal, timeline, sprintDuration, currentDate, sprintDeadline);

            const sprintEnd = new Date(Math.min(
                bestStart.getTime() + sprintDuration * 24 * 60 * 60 * 1000,
                sprintDeadline.getTime(),
                goal.dueDate.getTime()
            ));

            goal.sprints.push({
                start: bestStart.toISOString().split('T')[0],
                end: sprintEnd.toISOString().split('T')[0],
                goal: step.goal,
                weight: step.weight,
                type: 'sprint'
            });

            updateTimeline(timeline, bestStart, (sprintEnd - bestStart) / (24 * 60 * 60 * 1000), step.weight);
            currentDate = new Date(sprintEnd.getTime());
            sprintIndex++;

            // 减少休息时间
            if (step !== goal.steps[goal.steps.length - 1]) {
                const restDuration = Math.min(3, (goal.dueDate - currentDate) / (24 * 60 * 60 * 1000));
                if (restDuration > 0) {
                    const restEnd = new Date(currentDate.getTime() + restDuration * 24 * 60 * 60 * 1000);
                    goal.sprints.push({
                        start: currentDate.toISOString().split('T')[0],
                        end: restEnd.toISOString().split('T')[0],
                        goal: `Rest after: ${step.goal}`,
                        weight: 0,
                        type: 'rest'
                    });
                    currentDate = new Date(restEnd.getTime());
                }
            }
        }
    }

    return scheduledGoals;
}

function createTimeline(goals) {
    const startDate = new Date(Math.min(...goals.map(g => g.startDate)));
    const endDate = new Date(Math.max(...goals.map(g => g.dueDate)));
    const timeline = [];
    for (let d = new Date(startDate); d <= endDate; d.setDate(d.getDate() + 1)) {
        timeline.push({ date: new Date(d), load: 0 });
    }
    return timeline;
}

function findBestStartDate(goal, timeline, duration, lastSprintEnd, sprintDeadline) {
    let bestStart = new Date(Math.max(goal.startDate, lastSprintEnd));
    let minLoad = Infinity;

    const latestPossibleStart = new Date(Math.min(
        sprintDeadline.getTime() - duration * 24 * 60 * 60 * 1000,
        goal.dueDate.getTime() - duration * 24 * 60 * 60 * 1000
    ));

    for (let start = new Date(bestStart); start <= latestPossibleStart; start.setDate(start.getDate() + 1)) {
        const load = calculateLoad(timeline, start, duration);
        if (load < minLoad) {
            minLoad = load;
            bestStart = new Date(start);
        }
    }

    return bestStart;
}

function calculateLoad(timeline, start, duration) {
    let load = 0;
    for (let i = 0; i < duration; i++) {
        const date = new Date(start.getTime() + i * 24 * 60 * 60 * 1000);
        const day = timeline.find(d => d.date.getTime() === date.getTime());
        if (day) {
            load += day.load;
        }
    }
    return load;
}


function updateTimeline(timeline, start, duration, weight) {
    for (let i = 0; i < duration; i++) {
        const date = new Date(start.getTime() + i * 24 * 60 * 60 * 1000);
        const day = timeline.find(d => d.date.getTime() === date.getTime());
        if (day) {
            day.load += weight;
        }
    }
}

async function updateYAMLFrontmatter(scheduledGoals) {
    const updatedGoals = [];
    for (const goal of scheduledGoals) {
        const content = await app.vault.read(goal.file);
        const frontmatter = app.metadataCache.getFileCache(goal.file)?.frontmatter;

        if (frontmatter) {
            // 只保留 sprint 类型的条目，不包括 rest
            const filteredSprints = goal.sprints.filter(sprint => sprint.type === 'sprint');
            
            // 创建新的 sprints 数组，保持原始的开始和结束日期
            const updatedSprints = filteredSprints.map(sprint => ({
                start: sprint.start,
                end: sprint.end,
                goal: sprint.goal
            }));

            const updatedFrontmatter = { ...frontmatter, sprints: updatedSprints };
            const updatedContent = updateYAMLContent(content, updatedFrontmatter);
            await app.vault.modify(goal.file, updatedContent);
            updatedGoals.push(goal.file.basename);
        }
    }
    return updatedGoals;
}

function updateYAMLContent(content, frontmatter) {
    const yamlSeparator = '---\n';
    const [, oldFrontmatter, afterFrontmatter] = content.split(yamlSeparator);
    const newFrontmatter = Object.entries(frontmatter)
        .map(([key, value]) => `${key}: ${JSON.stringify(value)}`)
        .join('\n');
    return `${yamlSeparator}${newFrontmatter}\n${yamlSeparator}${afterFrontmatter}`;
}

function showResultPopup(updatedGoals) {
    const message = `已更新以下长期目标的Sprints:\n${updatedGoals.join('\n')}`;
    new Notice(message, 10000); // 显示10秒
}

        const result = await scheduleSprints();
        console.log(result);

        console.error("An error occurred:", error);
        new Notice("安排Sprints时发生错误，请查看控制台以获取详细信息。", 5000);
    }
})();
