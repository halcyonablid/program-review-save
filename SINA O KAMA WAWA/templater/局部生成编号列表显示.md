// 解析笔记名称
function parseNoteName(name) {
  const match = name.match(/ATOM@(.*?)- (.+)/);
  if (!match) return null;
  
  const addressPart = match[1];
  const namePart = match[2];
  
  const atIndex = addressPart.lastIndexOf('@');
  if (atIndex !== -1) {
    return {
      addressType: addressPart.substring(0, atIndex),
      address: addressPart.substring(atIndex + 1),
      name: namePart
    };
  } else {
    return {
      addressType: '',
      address: addressPart,
      name: namePart
    };
  }
}

function generateColorIndex(addressType, address) {
  const fullAddress = (addressType ? addressType + '@' : '') + address;
  let hash = 0;
  for (let i = 0; i < fullAddress.length; i++) {
    const char = fullAddress.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;
  }
  hash = Math.abs(hash);
  return hash % colorOptions.length;
}

function isSameFormat(addr1, addr2, ignoreDotsUpTo = 0) {
  const processAddr = (addr, ignoreDots) => {
    let processed = addr.substring(0, ignoreDots).replace(/\./g, '') + addr.substring(ignoreDots);
    return processed.replace(/[A-Z]/g, 'L').replace(/\d+/g, 'N');
  };

  const pattern1 = processAddr(addr1, ignoreDotsUpTo);
  const pattern2 = processAddr(addr2, ignoreDotsUpTo);

  return pattern1 === pattern2;
}

function containsDot(address) {
  return address.includes('.');
}

function getLengthBeforeFirstDot(address) {
  const dotIndex = address.indexOf('.');
  return dotIndex === -1 ? address.length : dotIndex;
}

function compareAddresses(addr1, addr2) {
  const parts1 = addr1.split('.');
  const parts2 = addr2.split('.');
  
  for (let i = 0; i < Math.min(parts1.length, parts2.length); i++) {
    if (parts1[i] !== parts2[i]) {
      if (/^\d+$/.test(parts1[i]) && /^\d+$/.test(parts2[i])) {
        return parseInt(parts1[i]) - parseInt(parts2[i]);
      }
      return parts1[i].localeCompare(parts2[i]);
    }
  }
  
  return parts1.length - parts2.length;
}

// 主处理函数
async function processAndDisplayNotes() {
    // 获取当前笔记名
    const currentNoteName = tp.file.title;
    





    if (!currentNoteName) {
        return "无法获取当前笔记名";
    }

    // 获取所有笔记
    const files = app.vault.getMarkdownFiles();
    const pages = files.filter(file => 
        file.name.startsWith("ATOM@") && 
        file.basename.includes("antinet: atom")
    ).sort((a, b) => a.name.localeCompare(b.name));

    // 找到当前笔记的索引
    const currentIndex = pages.findIndex(p => p.name === currentNoteName);

    if (currentIndex === -1) {
        return "当前笔记不在ATOM@列表中";
    }

    // 计算显示范围
    const startIndex = Math.max(0, currentIndex - 20);
    const endIndex = Math.min(pages.length, currentIndex + 21);

    // 颜色分配逻辑
    let colorIndex = 0;
    let addressFormats = new Map();
    let uncoloredPages = [];

    // 第一阶段：初步颜色分配
    pages.forEach((page) => {
        const noteName = parseNoteName(page.name);
        if (!noteName) return;

        const fullAddress = (noteName.addressType ? noteName.addressType + '@' : '') + noteName.address;
        
        if (containsDot(fullAddress)) {
            uncoloredPages.push({ page, fullAddress });
            return;
        }

        let found = false;
        for (let [existingAddress, existingColor] of addressFormats) {
            if (isSameFormat(existingAddress, fullAddress)) {
                addressFormats.set(fullAddress, existingColor);
                found = true;
                break;
            }
        }
        
        if (!found) {
            addressFormats.set(fullAddress, colorIndex);
            colorIndex = (colorIndex + 1) % colorOptions.length;
        }
    });

    // 第二阶段：处理含 "." 的地址
    while (uncoloredPages.length > 0) {
        uncoloredPages.sort((a, b) => compareAddresses(a.fullAddress, b.fullAddress));
        
        const firstPage = uncoloredPages.shift();
        const firstAddress = firstPage.fullAddress;
        const noteName = parseNoteName(firstPage.page.name);
        const ignoreDotsUpTo = getLengthBeforeFirstDot(firstAddress);

        const colorIndex = generateColorIndex(noteName.addressType, noteName.address);
        addressFormats.set(firstAddress, colorIndex);
        let currentColorIndex = (colorIndex + 1) % colorOptions.length;

        let sameLevelPages = [];
        let nextLevelPages = [];
        
        for (let { page, fullAddress } of uncoloredPages) {
            if (isSameFormat(firstAddress, fullAddress, ignoreDotsUpTo)) {
                addressFormats.set(fullAddress, colorIndex);
            } else {
                const firstDotCount = (firstAddress.match(/\./g) || []).length;
                const currentDotCount = (fullAddress.match(/\./g) || []).length;
                
                if (currentDotCount === firstDotCount) {
                    sameLevelPages.push({ page, fullAddress });
                } else {
                    nextLevelPages.push({ page, fullAddress });
                }
            }
        }

        for (let { page, fullAddress } of sameLevelPages) {
            let colorAssigned = false;
            for (let [existingAddress, existingColor] of addressFormats) {
                if (isSameFormat(existingAddress, fullAddress)) {
                    addressFormats.set(fullAddress, existingColor);
                    colorAssigned = true;
                    break;
                }
            }

            if (!colorAssigned) {
                addressFormats.set(fullAddress, currentColorIndex);
                currentColorIndex = (currentColorIndex + 1) % colorOptions.length;
            }
        }

        uncoloredPages = nextLevelPages;
    }

    // 生成HTML
    let tableHtml = `<table class="custom-table">
        <thead>
            <tr>
                <th class="number-column">编号</th>
                <th class="filename-column">文件名</th>
            </tr>
        </thead>
        <tbody>`;

    // 只渲染指定范围内的笔记
    for (let i = startIndex; i < endIndex; i++) {
        const p = pages[i];
        const noteName = parseNoteName(p.name);
        if (!noteName) continue;

        const fullAddress = (noteName.addressType ? noteName.addressType + '@' : '') + noteName.address;
        let colorName = colorOptions[addressFormats.get(fullAddress) % colorOptions.length];
        let color = colorMap[colorName][1];

        const isCurrentRow = i === currentIndex;
        const highlightStyle = isCurrentRow ? 'border: 2px solid #FF0000;' : '';
        const colorClass = ` style="background-color: ${color}; ${highlightStyle}"`;

        tableHtml += `<tr${colorClass}>
            <td class="number-column">${i - startIndex + 1}</td>
            <td class="filename-column"><a href="${p.path}" class="internal-link">${p.name}</a></td>
        </tr>`;
    }

    tableHtml += `</tbody></table>`;

    return tableHtml;
}

// 执行主函数并输出结果
tR += await processAndDisplayNotes();
