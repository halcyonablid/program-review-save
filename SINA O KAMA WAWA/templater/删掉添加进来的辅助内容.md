<%*
// 定义要匹配的模式
const pattern = /《《——《《[\s\S]*?》》——》》/g;

// 获取当前文件的内容
const currentFile = tp.file.find_tfile(tp.file.title);
let content = await app.vault.read(currentFile);

// 计算原始内容的长度
const originalLength = content.length;

// 使用正则表达式替换匹配的内容
content = content.replace(pattern, '');

// 计算新内容的长度
const newLength = content.length;

// 计算删除的字符数
const deletedChars = originalLength - newLength;

try {
    // 将修改后的内容写回文件
    await app.vault.modify(currentFile, content);
    
    // 给用户反馈
    if (deletedChars > 0) {
        new Notice(`成功删除了${deletedChars}个字符。`);
    } else {
        new Notice("没有找到需要删除的内容。");
    }
} catch (error) {
    console.error("更新文件时出错:", error);
    new Notice("更新文件时出错,请查看控制台以获取更多信息。");
}
%>
