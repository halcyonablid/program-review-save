async function openFile() {
    const filePath = '项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain/ATOM@121.001.001.001- 就在这里发动，每一次让系统启动的.md';
    
    const targetFile = app.vault.getAbstractFileByPath(filePath);
    
    if (!targetFile) {
        new Notice('未找到目标文件');

    }

    try {

    } catch (error) {
        console.error('Error:', error);
        new Notice(`打开文件时发生错误：${error.message}`);
    }
}
