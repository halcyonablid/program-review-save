<%*
// ==================== CONFIGURATION ====================
// The tab position to switch to (1 for the first, 3 for the third, etc.)
const TARGET_TAB_INDEX = 1;

// How long to wait for the API to become available (in milliseconds)
// It will check every 200ms up to this total time.
const API_WAIT_TIMEOUT = 5000; // 5 seconds
// =======================================================

new Notice("🚀 自动化全流程启动...", 3000);

// --- 步骤 1: 使用 Obsidian API 跳转到指定标签页 ---
new Notice(`#️⃣ 正在跳转到第 ${TARGET_TAB_INDEX} 个标签页...`);
const allLeaves = app.workspace.getLeavesOfType("markdown");

if (allLeaves.length < TARGET_TAB_INDEX) {
    new Notice(`❌ 错误: 未找到第 ${TARGET_TAB_INDEX} 个标签页。`, 5000);
    return;
}

const targetLeaf = allLeaves[TARGET_TAB_INDEX - 1];
app.workspace.setActiveLeaf(targetLeaf);

// --- 步骤 2: 智能等待 API 加载完成 ---
async function waitForApi(timeout) {
    const startTime = Date.now();
    while (Date.now() - startTime < timeout) {
        // If the API is ready, return it
        if (window.referenceGraphAPI && typeof window.referenceGraphAPI.runAutomatedGeneration === 'function') {
            return window.referenceGraphAPI;
        }
        // Otherwise, wait a short interval before checking again
        await new Promise(resolve => setTimeout(resolve, 200));
    }
    // If the loop finishes without finding the API, return null
    return null;
}

// --- 步骤 3: 执行 API 调用 ---
(async () => {
    new Notice("⚙️ 正在等待视图 API 加载...", 3000);
    const api = await waitForApi(API_WAIT_TIMEOUT);

    if (api) {
        try {
            new Notice("✅ API 已就绪，正在生成图表...");
            await api.runAutomatedGeneration();
            new Notice("🎉 自动化流程执行成功！", 5000);
        } catch (error) {
            new Notice(`❌ API 执行失败: ${error.message}`, 5000);
            console.error("Templater 脚本未能成功执行视图 API 函数:", error);
        }
    } else {
        new Notice("❌ 错误: 等待超时，未在当前页面找到关系图 API。", 6000);
        console.error("找不到 window.referenceGraphAPI，请确认第 " + TARGET_TAB_INDEX + " 个标签页是您的 Dataview 视图且没有错误。");
    }
})();
%>
