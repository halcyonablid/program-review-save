// 主程序执行
async function scheduleSprints() {
    try {
        const goals = await readLongTermGoals();
        const processedGoals = processGoalData(goals);
        const sixDigitCode = generateSixDigitCode();
        
        // 检查是否存在同一天的排期
        const existingSchedule = processedGoals.find(goal => 
            Object.keys(goal).some(key => key === `sprints${sixDigitCode}`)
        );

        const scheduledGoals = scheduleSprintsForGoals(processedGoals, sixDigitCode);
        
        // 如果存在同一天的排期,则替换
        const updatedGoals = await updateYAMLFrontmatter(scheduledGoals, sixDigitCode, existingSchedule);
        
        showResultPopup(updatedGoals);
        displayTimeline(scheduledGoals);
        return "Sprints scheduled successfully!";
    } catch (error) {
        console.error("An error occurred:", error);
        new Notice("安排Sprints时发生错误，请查看控制台以获取详细信息。", 5000);
        throw error;
    }
}

function generateSixDigitCode() {
    const now = new Date();
    return now.getFullYear().toString().slice(-2) + 
           (now.getMonth() + 1).toString().padStart(2, '0') + 
           now.getDate().toString().padStart(2, '0');
}

async function readLongTermGoals() {
    const files = app.vault.getMarkdownFiles();
    const goals = [];

    for (const file of files) {
        const content = await app.vault.read(file);
        const frontmatter = app.metadataCache.getFileCache(file)?.frontmatter;

        if (frontmatter && 
            frontmatter.qishiriqidate && 
            frontmatter.duedate && 
            (frontmatter.gaocengji === true || frontmatter.gaocengji === 'true')) {
            const goalSteps = extractGoalSteps(content);
            const sprintDates = {};
            
            for (let i = 1; i <= goalSteps.length; i++) {
                if (frontmatter[`sprint${i}`]) {
                    sprintDates[i] = new Date(frontmatter[`sprint${i}`]);
                }
            }

            goals.push({
                file: file,
                startDate: new Date(frontmatter.qishiriqidate),
                dueDate: new Date(frontmatter.duedate),
                steps: goalSteps,
                existingSprints: frontmatter.sprints || [],
                sprintDates: sprintDates,
                importance: frontmatter.importance || 5 // 默认为 5 如果没有设置
            });
        }
    }

    return goals;
}

function extractGoalSteps(content) {
    const lines = content.split('\n');
    const steps = [];
    let currentStep = null;
    let isInGoalSection = false;

    for (const line of lines) {
        if (line.trim() === '# 接下来的分解步骤，以及抵达的路点') {
            isInGoalSection = true;
            continue;
        }

        if (isInGoalSection) {
            if (line.startsWith('- goal:')) {
                if (currentStep) steps.push(currentStep);
                currentStep = { goal: line.substring(7).trim(), duration: 0, weight: 1 };
            } else if (line.includes('预计使用时间:') && currentStep) {
                const durationMatch = line.match(/预计使用时间:\s*(\d+)/);
                currentStep.duration = durationMatch ? parseInt(durationMatch[1]) : 0;
                if (isNaN(currentStep.duration)) {
                    console.error(`Invalid duration for step: ${currentStep.goal}`);
                    currentStep.duration = 0;
                }
                currentStep.weight = Math.log(currentStep.duration + 1) + 1;
                steps.push(currentStep);
                currentStep = null;
            } else if (line.trim().startsWith('#')) {
                break;
            }
        }
    }

    if (currentStep) steps.push(currentStep);
    return steps;
}

function processGoalData(goals) {
    return goals.map(goal => {
        const totalDuration = goal.steps.reduce((sum, step) => sum + step.duration, 0);
        const availableDays = Math.floor((goal.dueDate - goal.startDate) / (1000 * 60 * 60 * 24));
        
        const importance = goal.importance || 5; // 默认为 5 如果没有设置
        
        const priorityScore = (totalDuration / availableDays) * importance;
        
        return {
            ...goal,
            totalDuration,
            availableDays,
            priorityScore
        };
    });
}

function calculateSprintWeight(sprint, goal, currentDate) {
    const totalDuration = (goal.dueDate - goal.startDate) / (1000 * 60 * 60 * 24);
    const remainingTime = (goal.dueDate - currentDate) / (1000 * 60 * 60 * 24);
    const timeRatio = 1 - (remainingTime / totalDuration);
    
    const durationWeight = Math.log(sprint.duration + 1) + 1;
    const timeWeight = Math.exp(timeRatio * 2) - 1;
    
    return (durationWeight + timeWeight) / 2; // 归一化并相加
}

function scheduleSprintsForGoals(goals, sixDigitCode) {
    const timeline = createTimeline(goals);
    const scheduledGoals = goals.map(goal => ({ ...goal, sprints: [] }));

    scheduledGoals.sort((a, b) => b.priorityScore - a.priorityScore);

    for (const goal of scheduledGoals) {
        let currentDate = new Date(goal.startDate);
        let sprintIndex = 1;

        for (const step of goal.steps) {
            const sprintDuration = step.duration;
            const sprintDeadline = goal.sprintDates[sprintIndex] || goal.dueDate;
            
            const sprintWeight = calculateSprintWeight(step, goal, currentDate);
            
            let bestStart = findBestStartDateImproved(goal, timeline, sprintDuration, currentDate, sprintDeadline, sprintWeight);

            if (bestStart === null) {
                bestStart = currentDate; // 如果找不到最佳开始日期，就使用当前日期
            }

            const sprintEnd = new Date(bestStart.getTime() + sprintDuration * 24 * 60 * 60 * 1000);

            goal.sprints.push({
                start: bestStart.toISOString().split('T')[0],
                end: sprintEnd.toISOString().split('T')[0],
                goal: step.goal,
                weight: sprintWeight,
                type: 'sprint',
                code: sixDigitCode
            });

            updateTimeline(timeline, bestStart, sprintDuration, sprintWeight, goal.file.basename);
            currentDate = new Date(sprintEnd.getTime() + 3 * 24 * 60 * 60 * 1000); // 添加 3 天休息时间
            sprintIndex++;
        }
    }

    return scheduledGoals;
}

function createTimeline(goals) {
    const startDate = new Date(Math.min(...goals.map(g => g.startDate)));
    const endDate = new Date(Math.max(...goals.map(g => g.dueDate)));




    const timeline = [];
    for (let d = new Date(startDate); d <= endDate; d.setDate(d.getDate() + 1)) {
        timeline.push({ date: new Date(d), load: 0, goals: [] });
    }
    return timeline;
}

function findBestStartDateImproved(goal, timeline, duration, lastSprintEnd, sprintDeadline, sprintWeight) {
    let bestStart = null;
    let minOverlap = Infinity;

    const latestPossibleStart = new Date(Math.min(
        sprintDeadline.getTime() - duration * 24 * 60 * 60 * 1000,
        goal.dueDate.getTime() - duration * 24 * 60 * 60 * 1000
    ));

    for (let start = new Date(Math.max(goal.startDate, lastSprintEnd)); start <= latestPossibleStart; start.setDate(start.getDate() + 1)) {
        const overlap = calculateOverlap(timeline, start, duration, sprintWeight);
        if (overlap < minOverlap) {
            minOverlap = overlap;
            bestStart = new Date(start);
        }
    }

    return bestStart;
}

function calculateOverlap(timeline, start, duration, weight) {
    let overlap = 0;
    for (let i = 0; i < duration; i++) {
        const date = new Date(start.getTime() + i * 24 * 60 * 60 * 1000);
        const day = timeline.find(d => d.date.getTime() === date.getTime());
        if (day) {
            overlap += day.load;
        }
    }
    return overlap * weight;
}

function updateTimeline(timeline, start, duration, weight, goalName) {
    for (let i = 0; i < duration; i++) {
        const date = new Date(start.getTime() + i * 24 * 60 * 60 * 1000);
        const day = timeline.find(d => d.date.getTime() === date.getTime());
        if (day) {
            day.load += weight;
            if (!day.goals.includes(goalName)) {
                day.goals.push(goalName);
            }
        }
    }
}

async function updateYAMLFrontmatter(scheduledGoals, sixDigitCode, existingSchedule) {
    const updatedGoals = [];
    for (const goal of scheduledGoals) {
        const content = await app.vault.read(goal.file);
        const frontmatter = app.metadataCache.getFileCache(goal.file)?.frontmatter;

        if (frontmatter) {
            const sprintsKey = `sprints${sixDigitCode}`;
            const updatedSprints = goal.sprints.map(sprint => ({
                start: sprint.start,
                end: sprint.end,
                goal: sprint.goal
            }));

            let updatedFrontmatter = { ...frontmatter };
            
            // 如果存在同一天的排期,则删除旧的排期
            if (existingSchedule) {
                Object.keys(updatedFrontmatter).forEach(key => {
                    if (key.startsWith('sprints') && key !== sprintsKey) {
                        delete updatedFrontmatter[key];
                    }
                });
            }
            
            // 添加新的排期
            updatedFrontmatter[sprintsKey] = updatedSprints;
            
            const updatedContent = updateYAMLContent(content, updatedFrontmatter);
            await app.vault.modify(goal.file, updatedContent);
            updatedGoals.push(goal.file.basename);
        }
    }
    return updatedGoals;
}

function updateYAMLContent(content, frontmatter) {
    const yamlSeparator = '---\n';
    const [, oldFrontmatter, afterFrontmatter] = content.split(yamlSeparator);
    const newFrontmatter = Object.entries(frontmatter)
        .map(([key, value]) => `${key}: ${JSON.stringify(value)}`)
        .join('\n');
    return `${yamlSeparator}${newFrontmatter}\n${yamlSeparator}${afterFrontmatter}`;
}

function showResultPopup(updatedGoals) {
    const message = `已更新以下长期目标的Sprints:\n${updatedGoals.join('\n')}`;
    new Notice(message, 10000); // 显示10秒
}

function displayTimeline(scheduledGoals) {
    const allDates = scheduledGoals.flatMap(goal => 
        goal.sprints.flatMap(sprint => [new Date(sprint.start), new Date(sprint.end)])
    );
    const startDate = new Date(Math.min(...allDates));
    const endDate = new Date(Math.max(...allDates));

    let timeline = "";
    for (let d = new Date(startDate); d <= endDate; d.setDate(d.getDate() + 1)) {
        const date = d.toISOString().split('T')[0];
        const activeGoals = scheduledGoals.filter(goal => 
            goal.sprints.some(sprint => 
                date >= sprint.start && date <= sprint.end
            )
        );
        timeline += `${date}: ${activeGoals.map(g => g.file.basename).join(', ')}\n`;
    }
    console.log(timeline);
}
