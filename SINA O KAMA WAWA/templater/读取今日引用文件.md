<%*
// 配置
const DAILY_NOTE_FOLDER = "项目/35-给日程安排的文件夹/35C-日记的文件夹";
const QUOTE_HEADING = "# 想要引用的文件们";
const today = tp.date.now("YYYY-MM-DD");
const dailyNotePath = `${DAILY_NOTE_FOLDER}/${today}.md`;

let file = app.vault.getAbstractFileByPath(dailyNotePath);
if (!file) {
    new Notice("今日无日记文件，未读取数据。");
    return;
}

let content = await app.vault.read(file);
let lines = content.split('\n');
const headingIndex = lines.findIndex(line => line.trim() === QUOTE_HEADING.trim());

if (headingIndex === -1) {
    new Notice("未找到『# 想要引用的文件们』标题，未复制。");
    return;
}

// 目标块范围
let startLine = headingIndex + 1;
let endLine = lines.length;
for (let i = startLine; i < lines.length; i++) {
    if (lines[i].trim().startsWith('#') && lines[i].trim() !== QUOTE_HEADING.trim()) {
        endLine = i;
        break;
    }
}
const wikiLinkPattern = /^\s*\[\[.+\]\]\s*$/;
const links = [];
for (let i = startLine; i < endLine; i++) {
    const line = lines[i];
    if (wikiLinkPattern.test(line.trim())) {
        links.push(line.trim());
    }
}

// 写入剪切板
if (links.length === 0) {
    new Notice("未找到任何文件链接内容。");
    return;
}
await navigator.clipboard.writeText(links.join('\n'));
new Notice(`已复制 ${links.length} 条文件链接到剪切板！`);

// 询问是否清空
const choice = await tp.system.prompt(
    "是否要清空『# 想要引用的文件们』下所有 wiki 文件链接？输入 Y 或 y 确认，留空或其他即不删除。",
    { title: "清空确认", placeholder: "Y/y=确认，其他=保留" }
);

if (choice && (choice.trim() === "Y" || choice.trim() === "y")) {
    // 清空目标块内所有 [[...]] 行
    let changed = false;
    for (let i = endLine - 1; i >= startLine; i--) {
        if (wikiLinkPattern.test(lines[i].trim())) {
            lines.splice(i, 1);
            changed = true;
        }
    }
    if (changed) {
        await app.vault.modify(file, lines.join('\n'));
        new Notice("已清空所有wiki文件链接！");
    } else {
        new Notice("无需清空，目标区域无链接。");
    }
} else {
    new Notice("链接已复制，未清空引用区。");
}
%>
