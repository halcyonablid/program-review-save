    function categorizeNotesByType(kw) {

    /**
     * 按+号分割suffix，构建笔记的层级树结构
     */
    function buildPopupSuffixTree(notes) {
        const root = { children: new Map(), notes: [] };

        notes.forEach(note => {
            const suffix = note.suffix || '';
            const parts = suffix.split('+').filter(p => p.trim());

            if (parts.length === 0) {
                root.notes.push(note);
            } else {
                let current = root;
                parts.forEach((part, idx) => {
                    const trimmedPart = part.trim();
                    if (!trimmedPart) return;

                    if (!current.children.has(trimmedPart)) {
                        current.children.set(trimmedPart, { children: new Map(), notes: [] });
                    }
                    current = current.children.get(trimmedPart);

                    if (idx === parts.length - 1) {
                        current.notes.push(note);
                    }
                });
            }
        });

        return root;
    }

    /**
     * 统计节点后代数量
     */
    function countPopupDescendants(node) {
        let count = node.notes.length;
        node.children.forEach(child => {
            count += countPopupDescendants(child);
        });
        return count;
    }

    /**
     * 收集节点的所有笔记（扁平化，包括自身和所有后代）
     */
    function collectPopupAllNotes(node) {
        let allNotes = [...node.notes];
        node.children.forEach(child => {
            allNotes = allNotes.concat(collectPopupAllNotes(child));
        });
        return allNotes;
    }

    // --- 6. 树状图滚动控制函数 ---
    function scrollTreeViewToFile(fileName) {
        let scrolledCount = 0;

        // 方法1: 调用程序①的全局滚动函数 (整库树状图)
        if (window.evaTreeScrollToFile && window.evaTreeScrollToFile.viewType === 'EVA_FULL') {
            try {
                window.evaTreeScrollToFile(fileName);
                scrolledCount++;
            } catch (e) {
                console.log('程序①滚动失败:', e);
            }
        }

        // 方法2: 调用程序②的全局滚动函数 (关键词过滤视图)
        if (window.evaTreeScrollToFile_KEYWORD) {
            try {
                window.evaTreeScrollToFile_KEYWORD(fileName);
                scrolledCount++;
            } catch (e) {
                console.log('程序②滚动失败:', e);
            }
        }

        // 方法3: 通用滚动接口
        if (scrolledCount === 0 && window.evaTreeScrollToFile && !window.evaTreeScrollToFile.viewType) {
            try {
                window.evaTreeScrollToFile(fileName);
                scrolledCount++;
            } catch (e) {
                console.log('通用滚动失败:', e);
            }
        }

        // 方法4: 直接查找树状图容器中的元素（兜底方案）
        if (scrolledCount === 0) {
            const treeContainers = document.querySelectorAll('.hierarchical-tree-structure, .tree-view-container');
            treeContainers.forEach(container => {
                const items = container.querySelectorAll('.tree-item, [data-filename]');
                items.forEach(item => {
                    const itemFileName = item.dataset?.filename || item.getAttribute('data-filename');
                    if (itemFileName === fileName) {
                        item.scrollIntoView({ behavior: 'smooth', block: 'center' });
                        // 高亮动画
                        item.style.transition = 'background-color 0.3s';
                        const originalBg = item.style.backgroundColor;
                        item.style.backgroundColor = 'rgba(255, 215, 0, 0.5)';
                        setTimeout(() => {
                            item.style.backgroundColor = originalBg;
                        }, 2000);
                        scrolledCount++;
                    }
                });
            });
        }

        if (scrolledCount === 0) {
            new Notice(`未找到对应的树节点: ${fileName.substring(0, 40)}...`);
        } else {
            console.log(`已滚动 ${scrolledCount} 个视图到: ${fileName}`);
        }

        return scrolledCount > 0;
    }

    // --- 7. 关键词点击处理 ---
    async function handleKeywordClick(kw) {
        const notes = kw.notes || [];
        if (notes.length === 0) {
            return;
        }

        if (notes.length === 1) {
            scrollTreeViewToFile(notes[0].name);
        } else {
            // 多笔记时，按类型分组显示
            const categorized = await categorizeNotesByType(kw);
            showFileSelectModal(kw.displayText, categorized, kw.type);
        }
    }

    // --- 13. 分析每个关键词的实际来源类型 ---
    // v1.2: 直接使用索引中存储的 sourceType 字段
    function analyzeKeywordTypes(kw) {
        const notes = kw.notes || [];
        const types = new Set();

        notes.forEach(note => {
            // 优先使用索引中存储的 sourceType
            const sourceType = note.sourceType || kw.type || 'yaml';
            types.add(sourceType);
        });

        // 如果没有笔记，使用关键词的默认类型
        if (types.size === 0 && kw.type) {
            types.add(kw.type);
        }

        return Array.from(types);
    }

    // --- 14. 构建层级结构 ---
    // 按 sortLetter 分组
    const groups = {};
    const letters = [];

    // 找出所有顶层关键词（无 parentId）
    const topLevelKws = Object.values(keywords).filter(kw => !kw.parentId);

    // 为每个关键词添加到分组
    topLevelKws.forEach(kw => {
        const letter = kw.sortLetter || '#';
        if (!groups[letter]) {
            groups[letter] = [];
            letters.push(letter);
        }
        groups[letter].push(kw);
    });

    // 排序字母（# 放在 A 前面）
    letters.sort((a, b) => {
        if (a === '#') return -1;
        if (b === '#') return 1;
        return a.localeCompare(b);
    });

    // 字母导航
    if (letters.length > 0) {

    // --- 15. 递归渲染关键词（支持层级）---

    function createKeywordElement(kw) {
        const children = getChildren(kw.id);

        document.removeEventListener('mouseup', onMouseUp);
