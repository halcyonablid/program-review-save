<%*
// 获取用户输入的哈希值
let targetHash = await tp.system.prompt("请输入要查找的哈希值：");
targetHash = parseInt(targetHash);

// 使用 Obsidian API 获取所有文件
const files = app.vault.getMarkdownFiles();

// 异步检查文件函数
async function checkFile(file) {
    const frontmatter = app.metadataCache.getFileCache(file)?.frontmatter;
    if (frontmatter && frontmatter.哈希 === targetHash) {
        return file;
    }
    return null;
}

try {
    // 并行处理所有文件
    const results = await Promise.all(files.map(checkFile));
    const foundFile = results.find(file => file !== null);

    if (foundFile) {
        // 等待一段时间，确保表格已经渲染完成
        await new Promise(resolve => setTimeout(resolve, 500));

        // 直接查找对应的表格行
        const selector = `.custom-table tbody tr .filename-column a[href^="${foundFile.path}"]`;
        const link = document.querySelector(selector);

        if (link) {
            const row = link.closest('tr');
            // 找到匹配的行，滚动到该位置
            row.scrollIntoView({ behavior: 'smooth', block: 'center' });
            const rowIndex = Array.from(row.parentNode.children).indexOf(row) + 1;
            new Notice(`已找到哈希值为 ${targetHash} 的笔记：${foundFile.basename}（第 ${rowIndex} 行）`);
        } else {
            new Notice(`找到了笔记 ${foundFile.basename}，但无法在表格中定位。`);
        }
    } else {
        new Notice(`未找到哈希值为 ${targetHash} 的笔记。`);
    }
} catch (error) {
    console.error('Error:', error);
    new Notice(`查找过程中发生错误：${error.message}`);
}
%>
