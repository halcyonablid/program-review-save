<%*
/**
 * Obsidian Templater 脚本——只更新/新增"crab_canon"字段
 * 格式：crab_canon: [10, 7, {{date+7:YYYY-MM-DD}}, null]
 * 保留其它 YAML 字段，通知自动弹窗
 */
const file = tp.file;
const currentFile = file.find_tfile(file.title);
if (!currentFile) {
  new Notice("未找到当前文件");
  return;
}
const content = await file.content;

// ———YAML 处理———
const yamlMatch = content.match(/^---\n([\s\S]*?)\n---\n?/);
let yaml = yamlMatch ? yamlMatch[1] : "";
let restOfContent = yamlMatch ? content.slice(yamlMatch[0].length) : content;

let yamlLines = yaml ? yaml.split('\n').filter(x => x.trim() !== '') : [];

// 计算7天后的日期
const futureDate = new Date();
futureDate.setDate(futureDate.getDate() + 7);
const dateStr = futureDate.toISOString().split('T')[0]; // YYYY-MM-DD格式

// 新增或更新"crab_canon: [10, 7, 日期, null]"
let found = false;
yamlLines = yamlLines.map(line => {
  if (/^crab_canon\s*:/.test(line)) {
    found = true;
    return `crab_canon: [10, 7, ${dateStr}, null]`;
  }
  return line;
});
if (!found) {
  yamlLines.push(`crab_canon: [10, 7, ${dateStr}, null]`);
}

const newYaml = yamlLines.length ? `---\n${yamlLines.join('\n')}\n---\n` : '';
const newContent = `${newYaml}${restOfContent}`;

// 保存并无打扰通知
await app.vault.modify(currentFile, newContent);
new Notice("『crab_canon』已写入YAML", 3000);
%>
