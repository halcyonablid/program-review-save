<%*
// ============================================================
// 🚀 EVA 悬浮灵动岛 (自动同步版)
// 功能：非模态、可拖拽、颜色逻辑完美复刻、长标题换行
// 新增：切换笔记时自动更新视图并定位
// ============================================================

(async () => {
    // --- 0. Toggle 机制 ---
    const WIDGET_ID = "eva-floating-widget";
    const INDEX_WIDGET_ID = "eva-index-widget";  // 索引页 widget ID

    // 先清理可能存在的索引页 widget
    const existingIndex = document.getElementById(INDEX_WIDGET_ID);
    if (existingIndex) {
        if (existingIndex._cleanup) existingIndex._cleanup();
        existingIndex.remove();
    }

    // 检查自己的 widget
    const existing = document.getElementById(WIDGET_ID);
    // 如果存在，先移除旧的（触发旧的清理逻辑），重新加载
    if (existing) {
        // 尝试触发它上面绑定的清理函数（如果有）
        if (existing.cleanup) existing.cleanup();
        existing.remove();
        return;
    }

    const { Notice } = tp.obsidian;

    // --- 1. 配置 ---
    const CONFIG = {
        RENDER_WINDOW: 60 // 显示前后各60条
    };

    // --- 2. 颜色配置 (保持不变) ---
    const colorOptions = [ "钴蓝", "翡翠绿", "奶油色", "米色", "褐灰色", "中棕色", "亮红色", "亮黄色", "亮蓝色", "深紫色", "橙色", "绿色", "粉红色", "金色", "银色", "铜色", "珊瑚红", "孔雀蓝", "薰衣草紫", "玫瑰金", "青铜色", "深绿色", "勃艮第红", "蒂芙尼蓝", "杏色", "石榴红", "鸭蛋青", "芥末黄", "茄子紫", "橄榄绿", "赭石色", "靛蓝" ];

    const colorMap = {
        "钴蓝": "rgba(0, 71, 171, 0.1)", "翡翠绿": "rgba(80, 200, 120, 0.1)", "奶油色": "rgba(255, 253, 208, 0.1)", "米色": "rgba(245, 245, 220, 0.1)", "褐灰色": "rgba(139, 133, 137, 0.1)", "中棕色": "rgba(165, 42, 42, 0.1)", "亮红色": "rgba(255, 0, 0, 0.1)", "亮黄色": "rgba(255, 255, 0, 0.1)", "亮蓝色": "rgba(0, 0, 255, 0.1)", "深紫色": "rgba(75, 0, 130, 0.1)", "橙色": "rgba(255, 165, 0, 0.1)", "绿色": "rgba(0, 128, 0, 0.1)", "粉红色": "rgba(255, 192, 203, 0.1)", "金色": "rgba(255, 215, 0, 0.1)", "银色": "rgba(192, 192, 192, 0.1)", "铜色": "rgba(184, 115, 51, 0.1)", "珊瑚红": "rgba(255, 127, 80, 0.1)", "孔雀蓝": "rgba(51, 161, 201, 0.1)", "薰衣草紫": "rgba(230, 230, 250, 0.1)", "玫瑰金": "rgba(183, 110, 121, 0.1)", "青铜色": "rgba(205, 127, 50, 0.1)", "深绿色": "rgba(34, 139, 34, 0.1)", "勃艮第红": "rgba(128, 0, 32, 0.1)", "蒂芙尼蓝": "rgba(10, 186, 181, 0.1)", "杏色": "rgba(251, 206, 177, 0.1)", "石榴红": "rgba(196, 30, 58, 0.1)", "鸭蛋青": "rgba(224, 255, 255, 0.1)", "芥末黄": "rgba(255, 219, 88, 0.1)", "茄子紫": "rgba(97, 64, 81, 0.1)", "橄榄绿": "rgba(128, 128, 0, 0.1)", "赭石色": "rgba(160, 82, 45, 0.1)", "靛蓝": "rgba(75, 0, 130, 0.1)", "克莱因蓝": "rgba(0, 47, 167, 0.5)"
    };

    // --- 3. 核心算法 (保持不变) ---
    const Utils = {
        parseNoteName: function(name) {
            const match = name.match(/ATOM@(.*?)- (.+)/);
            if (!match) return null;
            const addressPart = match[1];
            const atIndex = addressPart.lastIndexOf('@');
            if (atIndex !== -1) {
                return { addressType: addressPart.substring(0, atIndex), address: addressPart.substring(atIndex + 1), name: match[2] };
            } else {
                return { addressType: '', address: addressPart, name: match[2] };
            }
        },
        generateColorIndex: function(addressType, address) {
            const fullAddress = (addressType ? addressType + '@' : '') + address;
            let hash = 0;
            for (let i = 0; i < fullAddress.length; i++) {
                hash = ((hash << 5) - hash) + fullAddress.charCodeAt(i);
                hash = hash & hash;
            }
            return Math.abs(hash) % colorOptions.length;
        },
        isSameFormat: function(addr1, addr2, ignoreDotsUpTo = 0) {
            const processAddr = (addr, ignoreDots) => {
                let processed = addr.substring(0, ignoreDots).replace(/\./g, '') + addr.substring(ignoreDots);
                return processed.replace(/[A-Z]/g, 'L').replace(/\d+/g, 'N');
            };
            return processAddr(addr1, ignoreDotsUpTo) === processAddr(addr2, ignoreDotsUpTo);
        },
        containsDot: function(address) { return address.includes('.'); },
        getLengthBeforeFirstDot: function(address) {
            const dotIndex = address.indexOf('.');
            return dotIndex === -1 ? address.length : dotIndex;
        },
        compareAddresses: function(addr1, addr2) {
            const parts1 = addr1.split('.');
            const parts2 = addr2.split('.');
            const minLen = Math.min(parts1.length, parts2.length);
            for (let i = 0; i < minLen; i++) {
                if (parts1[i] !== parts2[i]) {
                    if (/^\d+$/.test(parts1[i]) && /^\d+$/.test(parts2[i])) {
                        return parseInt(parts1[i]) - parseInt(parts2[i]);
                    }
                    return parts1[i].localeCompare(parts2[i]);
                }
            }
            return parts1.length - parts2.length;
        }
    };

    // --- 4. 全量数据计算 (可重建) ---
    let pageInfo = [];
    let addressFormats = new Map();

    const buildData = () => {
        const allFiles = app.vault.getMarkdownFiles().filter(file => {
            if (!file.name.startsWith("ATOM@")) return false;
            const cache = app.metadataCache.getFileCache(file);
            return cache && cache.frontmatter && cache.frontmatter.antinet === "atom";
        });

        allFiles.sort((a, b) => a.name.localeCompare(b.name, undefined, { numeric: true, sensitivity: 'base' }));

        const infoList = allFiles.map(file => {
            const noteName = Utils.parseNoteName(file.name);
            if (!noteName) return null;

            const fullAddress = (noteName.addressType ? noteName.addressType + '@' : '') + noteName.address;
            const cache = app.metadataCache.getFileCache(file);

            return {
                file: file, path: file.path, name: file.name, noteName: noteName, fullAddress: fullAddress,
                hasDot: Utils.containsDot(fullAddress),
                indexedKeywords: cache?.frontmatter?.['索引关键词已录入'] === true,
                treeStructure: cache?.frontmatter?.['树的结构'] === true
            };
        }).filter(Boolean);

        // 颜色计算 Phase 1 & 2
        let colorIndex = 0;
        let formats = new Map();
        let uncoloredPages = [];

        infoList.forEach((info) => {
            if (info.hasDot) { uncoloredPages.push(info); return; }
            let found = false;
            for (let [existingAddress, existingColor] of formats) {
                if (Utils.isSameFormat(existingAddress, info.fullAddress)) {
                    formats.set(info.fullAddress, existingColor);
                    found = true; break;
                }
            }
            if (!found) {
                formats.set(info.fullAddress, colorIndex);
                colorIndex = (colorIndex + 1) % colorOptions.length;
            }
        });

        if (uncoloredPages.length > 0) {
            uncoloredPages.sort((a, b) => Utils.compareAddresses(a.fullAddress, b.fullAddress));
            while (uncoloredPages.length > 0) {
                const firstPage = uncoloredPages.shift();
                const ignoreDotsUpTo = Utils.getLengthBeforeFirstDot(firstPage.fullAddress);
                const colorIdx = Utils.generateColorIndex(firstPage.noteName.addressType, firstPage.noteName.address);
                formats.set(firstPage.fullAddress, colorIdx);
                let remaining = [];
                uncoloredPages.forEach(info => {
                    if (Utils.isSameFormat(firstPage.fullAddress, info.fullAddress, ignoreDotsUpTo)) {
                        formats.set(info.fullAddress, colorIdx);
                    } else { remaining.push(info); }
                });
                uncoloredPages = remaining;
            }
        }

        return { pageInfo: infoList, addressFormats: formats };
    };

    const rebuildData = () => {
        const data = buildData();
        pageInfo = data.pageInfo;
        addressFormats = data.addressFormats;
    };

    rebuildData();

    // --- 5. 构建 Widget 骨架 (空内容) ---
    const widget = document.createElement("div");
    widget.id = WIDGET_ID;
    widget.style.cssText = `
        position: fixed;
        top: 30px;
        right: 40px;
        width: 550px;
        height: 90vh;
        max-height: 1200px;
        background: var(--background-primary);
        border: 1px solid var(--background-modifier-border);
        box-shadow: 0 8px 30px rgba(0,0,0,0.4);
        border-radius: 8px;
        z-index: 9999;
        display: flex;
        flex-direction: column;
        overflow: hidden;
        font-family: var(--font-text);
        animation: fadeIn 0.2s ease;
    `;

    const styleTag = document.createElement("style");
    styleTag.textContent = `
        @keyframes fadeIn { from { opacity: 0; transform: translateY(10px); } to { opacity: 1; transform: translateY(0); } }
        #${WIDGET_ID} .eva-header { padding: 8px 12px; background: var(--background-secondary); border-bottom: 1px solid var(--background-modifier-border); cursor: move; display: flex; justify-content: space-between; align-items: center; user-select: none; flex-shrink: 0; }
        #${WIDGET_ID} .eva-refresh:hover { color: var(--text-accent) !important; }
        #${WIDGET_ID} .eva-close:hover { color: var(--text-error) !important; }
        #${WIDGET_ID} .eva-scroll { flex: 1; overflow-y: auto; overflow-x: hidden; }
        .eva-table { width: 100%; border-collapse: collapse; font-size: 15px; table-layout: fixed; }
        .eva-table tr { min-height: 34px; border-bottom: 1px solid transparent; cursor: pointer; transition: background 0.1s; }
        .eva-table tr:hover { filter: brightness(0.96); }
        .eva-table td { padding: 6px 8px; line-height: 1.4; vertical-align: middle; }
        .eva-num { width: 45px; text-align: right; color: var(--text-muted); font-family: monospace; font-size: 0.9em; vertical-align: top !important; padding-top: 8px !important; }
        .eva-name { white-space: normal !important; word-break: break-word !important; overflow: visible !important; }
        .eva-current-file { border-left: 5px solid #000; font-weight: bold; }
        .eva-tree-mark { box-shadow: inset 4px 0 0 rgba(255, 0, 0, 0.6); }
    `;
    widget.appendChild(styleTag);

    // Header (Title placeholder)
    const header = document.createElement("div");
    header.className = "eva-header";
    const headerTitle = document.createElement("span");
    headerTitle.style.fontWeight = "600";
    headerTitle.textContent = "⚓ EVA 助手 (加载中...)";
    header.appendChild(headerTitle);

    const headerActions = document.createElement("div");
    headerActions.style.cssText = "display:flex; align-items:center; gap:6px;";

    const refreshBtn = document.createElement("div");
    refreshBtn.className = "eva-refresh";
    refreshBtn.innerHTML = "⟳";
    refreshBtn.style.cssText = "cursor:pointer; padding:0 6px; color:var(--text-muted); font-size:14px;";

    const closeBtn = document.createElement("div");
    closeBtn.className = "eva-close";
    closeBtn.innerHTML = "✕";
    closeBtn.style.cssText = "cursor:pointer; padding:0 6px; color:var(--text-muted); font-size:14px;";

    headerActions.appendChild(refreshBtn);
    headerActions.appendChild(closeBtn);
    header.appendChild(headerActions);
    widget.appendChild(header);

    // Content Area
    const scrollArea = document.createElement("div");
    scrollArea.className = "eva-scroll";
    const table = document.createElement("table");
    table.className = "eva-table";
    const tbody = document.createElement("tbody");
    table.appendChild(tbody);
    scrollArea.appendChild(table);
    widget.appendChild(scrollArea);

    // --- 5.5 滚动控制标志 ---
    let isClickFromWidget = false;  // 标记是否从 widget 内部点击
    let currentRenderedStart = 0;   // 当前渲染的起始索引
    let currentRenderedEnd = 0;     // 当前渲染的结束索引

    // --- 6. 核心渲染函数 (支持被事件调用) ---
    const updateView = (activeFile, forceRebuild = false) => {
        let currentIndex = 0;
        if (activeFile) currentIndex = pageInfo.findIndex(p => p.path === activeFile.path);
        if (currentIndex === -1) currentIndex = 0;
        if (forceRebuild) isClickFromWidget = false;

        // 检查新文件是否在当前渲染范围内
        const isInCurrentRange = currentIndex >= currentRenderedStart && currentIndex < currentRenderedEnd;

        // 如果是 widget 内部点击且在范围内，只更新高亮，不重建表格
        if (isClickFromWidget && isInCurrentRange && !forceRebuild) {
            // 移除旧的高亮
            const oldCurrent = tbody.querySelector(".eva-current-file");
            if (oldCurrent) {
                oldCurrent.classList.remove("eva-current-file");
            }
            // 添加新的高亮
            const rowIndex = currentIndex - currentRenderedStart;
            const newCurrent = tbody.children[rowIndex];
            if (newCurrent) {
                newCurrent.classList.add("eva-current-file");
            }
            isClickFromWidget = false;  // 重置标志
            return;
        }

        const start = Math.max(0, currentIndex - CONFIG.RENDER_WINDOW);
        const end = Math.min(pageInfo.length, currentIndex + CONFIG.RENDER_WINDOW);

        // 记录当前渲染范围
        currentRenderedStart = start;
        currentRenderedEnd = end;

        // 更新标题
        headerTitle.textContent = `⚓ EVA 助手 (${end - start}/${pageInfo.length})`;

        // 清空现有表格
        tbody.innerHTML = '';

        const displayRows = pageInfo.slice(start, end);
        displayRows.forEach((info, idx) => {
            const colorIdx = addressFormats.get(info.fullAddress) || 0;
            const colorName = colorOptions[colorIdx % colorOptions.length];
            const bgColor = info.indexedKeywords ? colorMap["克莱因蓝"] : colorMap[colorName];
            const isCurrent = (activeFile && info.path === activeFile.path);

            const tr = document.createElement("tr");
            tr.style.backgroundColor = bgColor;
            if (isCurrent) tr.classList.add("eva-current-file");
            if (info.treeStructure) tr.classList.add("eva-tree-mark");

            // 左键点击 - 打开笔记
            tr.onclick = async () => {
                isClickFromWidget = true;  // 标记是从 widget 点击的
                await app.workspace.openLinkText(info.path, "", false);
            };

            // 右键点击 - 滚动树状图到对应位置
            tr.oncontextmenu = (e) => {
                e.preventDefault();
                let scrolledCount = 0;

                // 滚动程序①（整库树状图）
                if (window.evaTreeScrollToFile && window.evaTreeScrollToFile.viewType === 'EVA_FULL') {
                    try {
                        window.evaTreeScrollToFile(info.name);
                        scrolledCount++;
                    } catch (err) {
                        console.log('程序①滚动失败:', err);
                    }
                }

                // 滚动程序②（关键词过滤版）
                if (window.evaTreeScrollToFile_KEYWORD) {
                    try {
                        window.evaTreeScrollToFile_KEYWORD(info.name);
                        scrolledCount++;
                    } catch (err) {
                        console.log('程序②滚动失败:', err);
                    }
                }

                if (scrolledCount > 0) {
                    new Notice(`已滚动 ${scrolledCount} 个树状图到: ${info.noteName.name.substring(0, 30)}...`);
                } else {
                    new Notice('未找到可用的树状图视图');
                }
            };

            const tdNum = document.createElement("td");
            tdNum.className = "eva-num";
            tdNum.textContent = start + idx + 1;

            const tdName = document.createElement("td");
            tdName.className = "eva-name";
            tdName.textContent = info.name;

            tr.appendChild(tdNum);
            tr.appendChild(tdName);
            tbody.appendChild(tr);
        });

        // 自动定位到当前项（无动画、无可见滚动过程）
        if (!isClickFromWidget) {
            requestAnimationFrame(() => {
                const current = widget.querySelector(".eva-current-file");
                if (current) {
                    const top = current.offsetTop;
                    const height = scrollArea.clientHeight;
                    const rowHeight = current.clientHeight;
                    const target = Math.max(top - height / 2 + rowHeight / 2, 0);
                    scrollArea.scrollTop = target;
                }
            });
        } else {
            isClickFromWidget = false;  // 重置标志，不自动定位
        }
    };

    // --- 6.5 数据重建与自动刷新 ---
    let refreshTimer = null;
    const scheduleRebuild = (delayMs = 0) => {
        if (refreshTimer) window.clearTimeout(refreshTimer);
        refreshTimer = window.setTimeout(() => {
            rebuildData();
            updateView(app.workspace.getActiveFile(), true);
        }, delayMs);
    };

    // --- 7. 初始化与事件绑定 ---
    // 先隐藏，渲染并定位后再显示，避免用户看到任何滚动过程
    widget.style.visibility = "hidden";
    document.body.appendChild(widget);

    // 初次渲染
    updateView(app.workspace.getActiveFile());

    widget.style.visibility = "visible";

    // 手动刷新按钮
    refreshBtn.onclick = () => scheduleRebuild(0);

    // 自动刷新（有轻微防抖，避免频繁重算）
    const vaultEventRefs = [];
    const cacheEventRefs = [];
    const AUTO_REFRESH_DEBOUNCE = 300;

    vaultEventRefs.push(app.vault.on("create", () => scheduleRebuild(AUTO_REFRESH_DEBOUNCE)));
    vaultEventRefs.push(app.vault.on("delete", () => scheduleRebuild(AUTO_REFRESH_DEBOUNCE)));
    vaultEventRefs.push(app.vault.on("rename", () => scheduleRebuild(AUTO_REFRESH_DEBOUNCE)));
    vaultEventRefs.push(app.vault.on("modify", () => scheduleRebuild(AUTO_REFRESH_DEBOUNCE)));
    cacheEventRefs.push(app.metadataCache.on("changed", () => scheduleRebuild(AUTO_REFRESH_DEBOUNCE)));

    // --- 8. 清理逻辑 (重要！) ---
    const cleanup = () => {
        if (refreshTimer) window.clearTimeout(refreshTimer);
        vaultEventRefs.forEach(ref => app.vault.offref(ref));
        cacheEventRefs.forEach(ref => app.metadataCache.offref(ref));
        if(widget.parentElement) widget.remove();
    };

    // 绑定到 DOM 对象上，方便再次打开时调用
    widget.cleanup = cleanup;

    closeBtn.onclick = cleanup;

    // --- 9. 拖拽逻辑 (保持不变) ---
    let isDragging = false, dragX = 0, dragY = 0;
    header.onmousedown = (e) => {
        if (e.target === closeBtn) return;
        isDragging = true;
        const rect = widget.getBoundingClientRect();
        dragX = e.clientX - rect.left;
        dragY = e.clientY - rect.top;
        widget.style.right = "auto";
        widget.style.left = rect.left + "px";
        widget.style.top = rect.top + "px";
        document.addEventListener('mousemove', onMouseMove);
        document.addEventListener('mouseup', onMouseUp);
        e.preventDefault();
    };
    const onMouseMove = (e) => {
        if (!isDragging) return;
        widget.style.left = (e.clientX - dragX) + "px";
        widget.style.top = (e.clientY - dragY) + "px";
    };
    const onMouseUp = () => {
        isDragging = false;
        document.removeEventListener('mousemove', onMouseMove);
        document.removeEventListener('mouseup', onMouseUp);
    };

})();
