// 主程序执行
async function scheduleSprints() {
    try {
        const goals = await readLongTermGoals();
        const processedGoals = processGoalData(goals);
        const scheduledGoals = iterativeScheduling(processedGoals);
        const updatedGoals = await updateYAMLFrontmatter(scheduledGoals);
        showResultPopup(updatedGoals);
        displayTimeline(scheduledGoals);
        return "Sprints scheduled successfully!";
    } catch (error) {
        console.error("An error occurred:", error);
        new Notice("安排Sprints时发生错误，请查看控制台以获取详细信息。", 5000);
        throw error;
    }
}

async function readLongTermGoals() {
    const files = app.vault.getMarkdownFiles();
    const goals = [];

    for (const file of files) {
        const content = await app.vault.read(file);
        const frontmatter = app.metadataCache.getFileCache(file)?.frontmatter;

        if (frontmatter && 
            frontmatter.qishiriqidate && 
            frontmatter.duedate && 
            (frontmatter.gaocengji === true || frontmatter.gaocengji === 'true')) {
            const goalSteps = extractGoalSteps(content);
            const sprintDates = {};
            
            for (let i = 1; i <= goalSteps.length; i++) {
                if (frontmatter[`sprint${i}`]) {
                    sprintDates[i] = new Date(frontmatter[`sprint${i}`]);
                }
            }

            goals.push({
                file: file,
                startDate: new Date(frontmatter.qishiriqidate),
                dueDate: new Date(frontmatter.duedate),
                steps: goalSteps,
                existingSprints: frontmatter.sprints || [],
                sprintDates: sprintDates
            });
        }
    }

    return goals;
}

function extractGoalSteps(content) {
    const lines = content.split('\n');
    const steps = [];
    let currentStep = null;
    let isInGoalSection = false;

    for (const line of lines) {
        if (line.trim() === '# 接下来的分解步骤，以及抵达的路点') {
            isInGoalSection = true;
            continue;
        }

        if (isInGoalSection) {
            if (line.startsWith('- goal:')) {
                if (currentStep) steps.push(currentStep);
                currentStep = { goal: line.substring(7).trim(), duration: 0, weight: 1 };
            } else if (line.includes('预计使用时间:') && currentStep) {
                currentStep.duration = parseInt(line.split(':')[1].trim()) || 0;
                currentStep.weight = Math.log(currentStep.duration + 1) + 1;
                steps.push(currentStep);
                currentStep = null;
            } else if (line.trim().startsWith('#')) {
                break;
            }
        }
    }

    if (currentStep) steps.push(currentStep);
    return steps;
}

function processGoalData(goals) {
    return goals.map(goal => ({
        ...goal,
        totalDuration: goal.steps.reduce((sum, step) => sum + step.duration, 0),
        totalWeight: goal.steps.reduce((sum, step) => sum + step.weight, 0),
        availableDays: Math.floor((goal.dueDate - goal.startDate) / (1000 * 60 * 60 * 24))
    }));
}

function calculateDynamicImportance(sprint, goal, currentDate) {
    const totalDuration = (goal.dueDate - goal.startDate) / (1000 * 60 * 60 * 24);
    const remainingTime = (goal.dueDate - currentDate) / (1000 * 60 * 60 * 24);
    const timeRatio = 1 - (remainingTime / totalDuration);
    
    const baseImportance = sprint.weight;
    const timeFactor = Math.exp(timeRatio * 2) - 1;
    const dynamicImportance = baseImportance * (1 + timeFactor);
    
    return dynamicImportance;
}

function iterativeScheduling(goals) {
    let scheduledGoals = goals.map(goal => ({ ...goal, sprints: [] }));
    let maxIterations = 20;
    let iteration = 0;

    while (iteration < maxIterations) {
        let unscheduledSprints = [];
        const timeline = createTimeline(scheduledGoals);

        // 尝试安排所有的 sprints
        for (const goal of scheduledGoals) {
            let currentDate = new Date(goal.startDate);
            let sprintIndex = 1;

            for (const step of goal.steps) {
                const sprintDuration = step.duration;
                const sprintDeadline = new Date(Math.min(
                    goal.sprintDates[sprintIndex] || goal.dueDate,
                    goal.dueDate
                ));
                
                const dynamicImportance = calculateDynamicImportance(step, goal, currentDate);
                
                let bestStart = findBestStartDate(goal, timeline, sprintDuration, currentDate, sprintDeadline, dynamicImportance);

                if (bestStart === null) {
                    unscheduledSprints.push({ goal, step, currentDate, sprintDeadline, dynamicImportance });
                } else {
                    const sprintEnd = new Date(bestStart.getTime() + sprintDuration * 24 * 60 * 60 * 1000);

                    goal.sprints.push({
                        start: bestStart.toISOString().split('T')[0],
                        end: sprintEnd.toISOString().split('T')[0],
                        goal: step.goal,
                        weight: step.weight,
                        dynamicImportance: dynamicImportance,
                        type: 'sprint'
                    });

                    updateTimeline(timeline, bestStart, sprintDuration, dynamicImportance, goal.file.basename);
                    currentDate = new Date(sprintEnd.getTime() + 3 * 24 * 60 * 60 * 1000); // 添加 3 天休息时间
                }
                sprintIndex++;
            }
        }

        if (unscheduledSprints.length === 0) {
            break; // 所有 sprints 都已安排，退出循环
        }

        // 如果还有未安排的 sprints，尝试向前移动已安排的 sprints
        let adjusted = false;
        for (const goal of scheduledGoals) {
            if (adjustSprintsForward(goal, timeline)) {
                adjusted = true;
            }
        }

        if (!adjusted) {
            console.log("无法进一步调整 sprints 位置");
            break;
        }

        iteration++;
    }

    return scheduledGoals;
}

function createTimeline(goals) {
    const startDate = new Date(Math.min(...goals.map(g => g.startDate)));
    const endDate = new Date(Math.max(...goals.map(g => g.dueDate)));
    const timeline = [];
    for (let d = new Date(startDate); d <= endDate; d.setDate(d.getDate() + 1)) {
        timeline.push({ date: new Date(d), load: 0, goals: [] });
    }
    return timeline;
}



function findBestStartDate(goal, timeline, duration, lastSprintEnd, sprintDeadline, dynamicImportance) {
    let bestStart = null;
    let minOverlap = Infinity;

    const latestPossibleStart = new Date(Math.min(
        sprintDeadline.getTime() - duration * 24 * 60 * 60 * 1000,
        goal.dueDate.getTime() - duration * 24 * 60 * 60 * 1000
    ));

    for (let start = new Date(Math.max(goal.startDate, lastSprintEnd)); start <= latestPossibleStart; start.setDate(start.getDate() + 1)) {
        const overlap = calculateOverlap(timeline, start, duration, goal.file.basename);
        if (overlap < minOverlap) {
            minOverlap = overlap;
            bestStart = new Date(start);
        }
    }

    return bestStart;
}

function calculateOverlap(timeline, start, duration, goalName) {
    let overlap = 0;
    for (let i = 0; i < duration; i++) {
        const date = new Date(start.getTime() + i * 24 * 60 * 60 * 1000);
        const day = timeline.find(d => d.date.getTime() === date.getTime());
        if (day) {
            overlap += day.goals.filter(g => g !== goalName).length;
        }
    }
    return overlap;
}

function adjustSprintsForward(goal, timeline) {
    let adjusted = false;
    for (let i = 0; i < goal.sprints.length; i++) {
        const sprint = goal.sprints[i];
        const currentStart = new Date(sprint.start);
        const earliestPossibleStart = i === 0 ? goal.startDate : new Date(goal.sprints[i-1].end);
        earliestPossibleStart.setDate(earliestPossibleStart.getDate() + 3); // 确保 3 天休息时间

        const newStart = findEarliestAvailableDate(timeline, earliestPossibleStart, new Date(sprint.end), sprint.goal, goal.file.basename);

        if (newStart && newStart < currentStart) {
            // 更新 timeline
            updateTimeline(timeline, currentStart, (new Date(sprint.end) - currentStart) / (24 * 60 * 60 * 1000), -sprint.weight, goal.file.basename);
            updateTimeline(timeline, newStart, (new Date(sprint.end) - currentStart) / (24 * 60 * 60 * 1000), sprint.weight, goal.file.basename);

            // 更新 sprint 日期
            const newEnd = new Date(newStart.getTime() + (new Date(sprint.end) - currentStart));
            sprint.start = newStart.toISOString().split('T')[0];
            sprint.end = newEnd.toISOString().split('T')[0];

            adjusted = true;
        }
    }
    return adjusted;
}

function findEarliestAvailableDate(timeline, start, end, sprintGoal, goalName) {
    const duration = (end - start) / (24 * 60 * 60 * 1000);
    for (let d = new Date(start); d <= end; d.setDate(d.getDate() + 1)) {
        if (isTimeSlotAvailable(timeline, d, duration, goalName)) {
            return d;
        }
    }
    return null;
}

function isTimeSlotAvailable(timeline, start, duration, goalName) {
    for (let i = 0; i < duration; i++) {
        const date = new Date(start.getTime() + i * 24 * 60 * 60 * 1000);
        const day = timeline.find(d => d.date.getTime() === date.getTime());
        if (day && day.goals.some(g => g !== goalName)) {
            return false;
        }
    }
    return true;
}

function updateTimeline(timeline, start, duration, weight, goalName) {
    for (let i = 0; i < duration; i++) {
        const date = new Date(start.getTime() + i * 24 * 60 * 60 * 1000);
        const day = timeline.find(d => d.date.getTime() === date.getTime());
        if (day) {
            day.load += weight;
            if (!day.goals.includes(goalName)) {
                day.goals.push(goalName);
            }
        }
    }
}

async function updateYAMLFrontmatter(scheduledGoals) {
    const updatedGoals = [];
    for (const goal of scheduledGoals) {
        const content = await app.vault.read(goal.file);
        const frontmatter = app.metadataCache.getFileCache(goal.file)?.frontmatter;

        if (frontmatter) {
            const filteredSprints = goal.sprints.filter(sprint => sprint.type === 'sprint');
            
            const updatedSprints = filteredSprints.map(sprint => ({
                start: sprint.start,
                end: sprint.end,
                goal: sprint.goal
            }));

            const updatedFrontmatter = { ...frontmatter, sprints: updatedSprints };
            const updatedContent = updateYAMLContent(content, updatedFrontmatter);
            await app.vault.modify(goal.file, updatedContent);
            updatedGoals.push(goal.file.basename);
        }
    }
    return updatedGoals;
}

function updateYAMLContent(content, frontmatter) {
    const yamlSeparator = '---\n';
    const [, oldFrontmatter, afterFrontmatter] = content.split(yamlSeparator);
    const newFrontmatter = Object.entries(frontmatter)
        .map(([key, value]) => `${key}: ${JSON.stringify(value)}`)
        .join('\n');
    return `${yamlSeparator}${newFrontmatter}\n${yamlSeparator}${afterFrontmatter}`;
}

function showResultPopup(updatedGoals) {
    const message = `已更新以下长期目标的Sprints:\n${updatedGoals.join('\n')}`;
    new Notice(message, 10000); // 显示10秒
}

function displayTimeline(scheduledGoals) {
    const allDates = scheduledGoals.flatMap(goal => 
        goal.sprints.flatMap(sprint => [new Date(sprint.start), new Date(sprint.end)])
    );
    const startDate = new Date(Math.min(...allDates));
    const endDate = new Date(Math.max(...allDates));

    let timeline = "";
    for (let d = new Date(startDate); d <= endDate; d.setDate(d.getDate() + 1)) {
        const date = d.toISOString().split('T')[0];
        const activeGoals = scheduledGoals.filter(goal => 
            goal.sprints.some(sprint => 
                date >= sprint.start && date <= sprint.end
            )
        );
        timeline += `${date}: ${activeGoals.map(g => g.file.basename).join(', ')}\n`;
    }
    console.log(timeline);
}
