async function findLatestActiveNote() {
    const files = app.vault.getMarkdownFiles();
    for (const file of files) {
        const content = await app.vault.read(file);
        if (content.includes("青蛙: true")) {
            return file;
        }
    }
    return null;
}

async function updateViewFileYAML(activeNote) {
    // 指定视图文件路径
    const viewFilePath = "项目/18-战略视图迭代/18E-GTD夹子/18E1-下一步代办事项清单文件夹/下一步代办的liner chain/青蛙清单.md";
    const viewFile = app.vault.getAbstractFileByPath(viewFilePath);
    
    if (!viewFile) {
        new Notice("未找到视图文件！");
        return;
    }

    try {
        // 读取视图文件内容
        const content = await app.vault.read(viewFile);
        
        // 更新YAML
        let newContent;
        if (content.startsWith('---')) {
            // 如果已经存在YAML前置元数据
            const yamlEndIndex = content.indexOf('---', 3);
            if (yamlEndIndex !== -1) {
                const yamlContent = content.substring(0, yamlEndIndex);
                const restContent = content.substring(yamlEndIndex);
                
                // 更新或添加最新的事件记录的笔记名属性
                if (yamlContent.includes('最新的事件记录的笔记名:')) {
                    newContent = yamlContent.replace(/最新的事件记录的笔记名:.*$/m, `最新的事件记录的笔记名: ${activeNote.basename}`) + restContent;
                } else {
                    newContent = yamlContent.trim() + `\n最新的事件记录的笔记名: ${activeNote.basename}\n` + restContent;
                }
            }
        } else {
            // 如果不存在YAML前置元数据，添加新的
            newContent = `---\n最新的事件记录的笔记名: ${activeNote.basename}\n---\n${content}`;
        }

        // 写入更新后的内容
        await app.vault.modify(viewFile, newContent);

        // 在新标签页中打开视图文件
        const newLeaf = app.workspace.getLeaf('tab');
        await newLeaf.openFile(viewFile);
        
        // 设置为预览模式
        if (newLeaf.view.getMode() !== 'preview') {
            await newLeaf.setViewState({
                type: 'markdown',
                state: { mode: 'preview' }
            });
        }

        // 锁定标签页
        if (newLeaf.view.leaf) {
            newLeaf.view.leaf.setPinned(true);
        }

        new Notice('已更新视图文件并打开（预览模式）');
    } catch (error) {
        console.error('Error:', error);
        new Notice(`更新YAML时发生错误：${error.message}`);
    }
}

// 执行主函数
async function main() {
    const activeNote = await findLatestActiveNote();
    if (!activeNote) {
        new Notice("未找到带有'工作进程记录'的笔记！");
        return;
    }

    await updateViewFileYAML(activeNote);
}

await main();
