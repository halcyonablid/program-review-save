<%*
async function toggleRightSidebarTabs() {
    // Get the right sidebar container
    const rightSidebar = app.workspace.rightSplit;
    
    // Ensure the right sidebar is visible
    if (!rightSidebar.expanded) {
        rightSidebar.expand();
    }
    
    // Get all leaves in the right sidebar
    const rightLeaves = rightSidebar.children[0].children;
    
    // Check if we have at least 2 tabs
    if (!rightLeaves || rightLeaves.length < 2) {
        new Notice('右侧边栏需要至少两个标签才能切换');
        return;
    }
    
    // Get the currently active leaf
    const activeLeaf = app.workspace.activeLeaf;
    
    // Find the index of the active leaf in the right sidebar
    const activeIndex = rightLeaves.indexOf(activeLeaf);
    
    // If active leaf is in right sidebar (index >= 0)
    if (activeIndex >= 0) {
        // If first tab is active, switch to second
        if (activeIndex === 0) {
            await app.workspace.setActiveLeaf(rightLeaves[1], true);
            new Notice('已切换到右侧边栏第二个标签');
        } 
        // If second tab is active, switch to first
        else if (activeIndex === 1) {
            await app.workspace.setActiveLeaf(rightLeaves[0], true);
            new Notice('已切换到右侧边栏第一个标签');
        }
    } else {
        // If no tab is active in right sidebar, switch to first tab
        await app.workspace.setActiveLeaf(rightLeaves[0], true);
        new Notice('已切换到右侧边栏第一个标签');
    }
}

// 执行函数
await toggleRightSidebarTabs();
%>
