<%*
// 主处理函数
function addTodoIndexProperty() {
    // 获取当前编辑器实例
    const editor = app.workspace.activeEditor?.editor;
    if (!editor) return;

    // 获取当前文件的全部内容
    const content = editor.getValue();

    // 检查是否有YAML前置元数据
    if (!content.startsWith('---')) {
        // 如果没有YAML，直接返回
        return;
    }

    // 查找第二个'---'的位置
    const secondDashIndex = content.indexOf('---', 3);
    if (secondDashIndex === -1) return;

    // 提取YAML部分
    const yamlContent = content.substring(0, secondDashIndex);
    const restContent = content.substring(secondDashIndex);

    // 检查是否已经有"待办事项在索引上"属性
    if (!yamlContent.includes('待咨询内容:')) {
        // 在YAML末尾添加新属性（在第二个'---'之前）
        const newContent = yamlContent.trimEnd() + '\n待咨询内容: true\n' + restContent;
        
        // 更新编辑器内容
        editor.setValue(newContent);
    }
}

// 执行函数
addTodoIndexProperty();

%>
